48    	 * Moves the ILM and DLM code section of the specified project binary into
49    	 * CPU0's ILM and DLM
50    	 *
51    	 * @param p_project_binary_header
52    	 *  The location in memory of the Project Binary
53    	 */
54    	void jump_pram(const PROJECT_BINARY_HEADER *p_project_binary_header)
55    	{
56    	    U64 project_binary_address;
57    	    U64 cpu0_atcm_address, cpu0_btcm_address;
58    	    U32 cpu0_atcm_size, cpu0_btcm_size;
59    	
60    	    LOAD_CODE_STATUS *p_load_code_status;
61    	    const PROJECT_BINARY_HEADER *ptr_project_binary_header;
62    	
63    	    gpFwSimCtrl1Reg->jump_to_fw = 1;
64    	
65    	    update_dbuf_for_fw();
66    	    project_binary_address = (U64)(U32)p_project_binary_header;
67    	
68    	    p_load_code_status = (LOAD_CODE_STATUS *)(DBUF_API_BOOTCODE_MISC_BASE + DBUF_API_BOOTCODE_MISC_OFFSET_LOAD_CODE_STATUS);
69    	    project_binary_address += (p_load_code_status->key_manifest_size);
70    	    ptr_project_binary_header = (const PROJECT_BINARY_HEADER *)((U32)(project_binary_address & 0x00000000FFFFFFFF));
71    	
72    	    cpu0_atcm_address = project_binary_address + (SECTOR_SIZE * ptr_project_binary_header->project_binary_header_sector_size);
73    	    cpu0_atcm_size = (ptr_project_binary_header->section_sector_size[0] << SECTOR_SIZE_SHIFT);
74    	
75    	    cpu0_btcm_address = cpu0_atcm_address + cpu0_atcm_size;
76    	    cpu0_btcm_size = (ptr_project_binary_header->section_sector_size[1] << SECTOR_SIZE_SHIFT);
77    	
78    	#if MI_OOB_SMBUS_EN
79    	    // SMBS force NACK and disable INT prior to jump code
80    	    // JIRA: https://jira.phison.com:8443/browse/SDKBOOT-1127
81    	    phal_smbs_clr_int_en(SMBUS0, SMBS_PROTOCOL_INT_BITMAP | SMBS_ERROR_INT_BITMAP);
82    	    phal_smbs_set_force_nack(SMBUS0);
83    	    phal_smbs_rx_vld_clr(SMBUS0);
84    	    hal_smbs_reset_2(SMBUS0);
85    	#endif
86    	
87    	    phal_pmu_switch_jump(cpu0_atcm_address,  MEMORY_MAP_CPU0_ATCM_BASE, cpu0_atcm_size, cpu0_btcm_address, MEMORY_MAP_CPU0_B0TCM_BASE, cpu0_btcm_size);
88    	}

96    	 *
97    	 * @param p_project_binary_header
98    	 *  The location of the Project Binary to be checked
99    	 *
100    	 * @return
101    	 *  16-bit bitmap that indicates the errors that have occurred,
102    	 *      BIT0 (PROJECT_BINARY_ERROR_HEADER_CODE_SIGNATURE_MARKING)
103    	 *          The Code Signature Marking field (offset 0x00, size 0x10) is incorrect
104    	 *      BIT1 (PROJECT_BINARY_ERROR_HEADER_IC_VERSION)
105    	 *          The IC Version field (offset 0x18, Size 0x08) is incorrect

108    	static U32 check_project_binary_header(const PROJECT_BINARY_HEADER *p_project_binary_header)
109    	{
110    	    U32 check_project_binary_error;
111    	
112    	    const U8 a_code_signature_marking[PROJECT_BINARY_HEADER_CODE_SIGNATURE_MARKING_SIZE] =
113    	    {'C', 'O', 'D', 'E', ' ', 'B', 'L', 'O', 'C', 'K', 0x55, 0xAA, 0xAA, 0x55, 0x00, 0xFF};
114    	
115    	    check_project_binary_error = 0;
116    	
117    	    Dump_SEC(tprintf("Check Header"));
118    	    Dump_SEC(tprintf("marking addr :%x",p_project_binary_header->a_code_signature_marking));
119    	    //Check the Code Signature Marking
120    	    if (memcmp((void *)(p_project_binary_header->a_code_signature_marking), (void *)a_code_signature_marking, sizeof(p_project_binary_header->a_code_signature_marking))) {
121    	        Dump_SEC(tprintf("!!!Check Fail %L: ", (U64)PROJECT_BINARY_ERROR_HEADER_CODE_SIGNATURE_MARKING));
122    	        Dump_SEC(tprintf("Code Signature Marking"));
123    	        check_project_binary_error |= PROJECT_BINARY_ERROR_HEADER_CODE_SIGNATURE_MARKING;
124    	    }
125    	    Dump_SEC(tprintf("version addr :%x",p_project_binary_header->a_ic_version));
126    	    //Check the IC Version Marking
127    	    if (memcmp((void *)(p_project_binary_header->a_ic_version), (void *)ga_ic_version, sizeof(p_project_binary_header->a_ic_version))) {
128    	        Dump_SEC(tprintf("!!!Check Fail %L: ", (U64)PROJECT_BINARY_ERROR_HEADER_IC_VERSION));
129    	        Dump_SEC(tprintf("IC Version"));
130    	        check_project_binary_error |= PROJECT_BINARY_ERROR_HEADER_IC_VERSION;
131    	    }
132    	
133    	    // Check file format - only support FMT5
134    	    if (p_project_binary_header->file_format != PROJECT_BINARY_HEADER_SIGNATURE_FILE_FORMAT_KEY_SIGNATURE)
135    	    {
136    	        Dump_SEC(tprintf("!!!Chk F"));
137    	        Dump_SEC(tprintf("File Format %d not supported", p_project_binary_header->file_format));
138    	        return PROJECT_BINARY_ERROR_HEADER_FILE_FORMAT;
139    	    }
140    	
141    	    //TODO add checking section sizes
142    	    //needs at least 1 sector for atcm
143    	    //atcm and btcm should not exceed the memory map definition
144    	
145    	    return check_project_binary_error;
146    	}

148    	static U32 check_project_binary_security_setting(const PROJECT_BINARY_HEADER *p_project_binary_header, const EFUSE_SECURITY_SETTING *p_efuse_security_setting, U32 project_binary_source)
149    	{
150    	    U32 check_project_binary_error;
151    	
152    	    check_project_binary_error = 0;
153    	
154    	    //Checks the CIDM
155    	    //The CIDM in the Project Binary must match a valid CIDM in eFuse
156    	    //Fails if ALL of the below conditions are NOT true
157    	    //1. Project Binary CIDM is 0x0000 (can be accepted by all eFuse CIDM)
158    	    //2. EFuse CIDM is valid and matches Project Binary's CIDM
159    	    //3. EFuse CIDM has not been specified
160    	    //4. CIDM checking can be bypassed
161    	    if (!(
162    	            (0x0000 == p_project_binary_header->cidm) ||
163    	            ((0xFFFF == (p_efuse_security_setting->cidm ^ p_efuse_security_setting->cidm_inverse)) &&
164    	                (p_efuse_security_setting->cidm == p_project_binary_header->cidm)) ||
165    	            ((0x0000 == p_efuse_security_setting->cidm) &&
166    	                (0x0000 == p_efuse_security_setting->cidm_inverse)) ||
167    	            ((0 == p_efuse_security_setting->cidm_bypass_disable) &&
168    	                (0xFFFF == p_efuse_security_setting->cidm) &&
169    	                (0xFFFF == p_efuse_security_setting->cidm_inverse)))) {
170    	
171    	        Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_SECURITY_SETTING_CIDM));
172    	        Dump_SEC(tprintf("CIDM"));
173    	        Dump_SEC(tprintf("CIDM Project Binary: 0x%x", p_project_binary_header->cidm));
174    	        Dump_SEC(tprintf("CIDM eFuse: 0x%x, ", p_efuse_security_setting->cidm));
175    	        Dump_SEC(tprintf("CIDM eFuse Inverse: 0x%x", p_efuse_security_setting->cidm_inverse));
176    	        Dump_SEC(tprintf("CIDM Bypass Disable: %d", p_efuse_security_setting->cidm_bypass_disable));
177    	
178    	        //TODO this isn't a good way to use production_mode, actually modify CIDM value instead so that CIDM checks actually passes
179    	        if (gBCodeGlobalStruct.production_mode) {
180    	            check_project_binary_error |= PROJECT_BINARY_ERROR_SECURITY_SETTING_CIDM;
181    	        }
182    	    }
183    	
184    	    //Checks the Security Version based on how the Project Binary is obtained
185    	    //The highest bit of the Security Version in the Project Binary must be
186    	    //  greater than or equal to the highest bit of the Security Version in eFuse
187    	    //Fails if ALL of the below conditions are NOT true
188    	    //1. Obtained from NAND and Security Version in the is lower
189    	    if (!(
190    	            ((PROJECT_BINARY_SOURCE_NAND == project_binary_source) &&
191    	                (0 == check_field_valid(p_efuse_security_setting->security_version_nand, p_project_binary_header->security_version))) ||
192    	            ((PROJECT_BINARY_SOURCE_HOST == project_binary_source) &&
193    	                (0 == check_field_valid(p_efuse_security_setting->security_version_host, p_project_binary_header->security_version))))) {
194    	
195    	        Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_SECURITY_SETTING_SECURITY_VERSION));
196    	        Dump_SEC(tprintf("Security Version"));
197    	        Dump_SEC(tprintf("Security Version Project Binary: 0x%x", p_project_binary_header->security_version));
198    	        Dump_SEC(tprintf("Security Version NAND: 0x%x", p_efuse_security_setting->security_version_nand));
199    	        Dump_SEC(tprintf("Security Version Host: 0x%x", p_efuse_security_setting->security_version_host));
200    	
201    	        //TODO this isn't a good way to use production_mode, actually modify security version instead so tht the check passes
202    	        if (gBCodeGlobalStruct.production_mode) {
203    	            check_project_binary_error |= PROJECT_BINARY_ERROR_SECURITY_SETTING_SECURITY_VERSION;
204    	        }
205    	    }
206    	
207    	    //Code Signature Check is required, but the Project Binary is not signed
208    	    if (0 == p_project_binary_header->code_sign_finish) {
209    	        if (0 != p_efuse_security_setting->check_code_signature_enable) {
210    	
211    	            Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_SECURITY_SETTING_NOT_SIGNED));
212    	            Dump_SEC(tprintf("Code Signature Required"));
213    	            Dump_SEC(tprintf("eFuse Code Signature Check: %d", p_efuse_security_setting->check_code_signature_enable));
214    	            Dump_SEC(tprintf("Binary Code Signature Finish: %d", p_project_binary_header->code_sign_finish));
215    	
216    	            //Code Signature related Security Settings are checked when checking Code Signature
217    	            if (gBCodeGlobalStruct.production_mode) {
218    	                check_project_binary_error |= PROJECT_BINARY_ERROR_SECURITY_SETTING_NOT_SIGNED;
219    	            }
220    	        }
221    	    }
222    	    else {
223    	        //Check the accepted algorithm(0x17)
224    	        switch (p_project_binary_header->algorithm_type) {
225    	        case PROJECT_BINARY_HEADER_ALGORITHM_TYPE_RSA4096_SHA512_PKCS1_V2_1:
226    	            if (0 != gEFuseAPI.efuse_security_setting.rsa4096_pkcs2_1_disable) {
227    	
228    	                Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE));
229    	                Dump_SEC(tprintf("Code Signature Algorithm Disabled"));
230    	                Dump_SEC(tprintf("Algorithm 0x%x (RSA4096 SHA512 PKCS1 V2.1)", PROJECT_BINARY_HEADER_ALGORITHM_TYPE_RSA4096_SHA512_PKCS1_V2_1));
231    	
232    	                if (gBCodeGlobalStruct.production_mode) {
233    	                    check_project_binary_error |= PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE;
234    	                }
235    	            }
236    	            break;
237    	
238    	        case PROJECT_BINARY_HEADER_ALGORITHM_TYPE_SW_SM2_SM3:
239    	            if (0 != gEFuseAPI.efuse_security_setting.sm2_sm3_disable) {
240    	
241    	                Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE));
242    	                Dump_SEC(tprintf("Code Signature Algorithm Disabled"));
243    	                Dump_SEC(tprintf("Algorithm 0x%x (SM3 SM2)", PROJECT_BINARY_HEADER_ALGORITHM_TYPE_SW_SM2_SM3));
244    	
245    	                if (gBCodeGlobalStruct.production_mode) {
246    	                    check_project_binary_error |= PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE;
247    	                }
248    	            }
249    	            break;
250    	
251    	        case PROJECT_BINARY_HEADER_ALGORITHM_TYPE_LMS_SHA256_192:
252    	            if (0 != gEFuseAPI.efuse_security_setting.lms_sha256_192_disable) {
253    	
254    	                Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE));
255    	                Dump_SEC(tprintf("Code Signature Algorithm Disabled"));
256    	                Dump_SEC(tprintf("Algorithm 0x%x (LMS SHA256)", PROJECT_BINARY_HEADER_ALGORITHM_TYPE_LMS_SHA256_192));
257    	
258    	                if (gBCodeGlobalStruct.production_mode) {
259    	                    check_project_binary_error |= PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE;
260    	                }
261    	            }
262    	            break;
263    	
264    	        case PROJECT_BINARY_HEADER_ALGORITHM_TYPE_SW_SM9:
265    	            if (0 != gEFuseAPI.efuse_security_setting.sw_sm9_disable) {
266    	
267    	                Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE));
268    	                Dump_SEC(tprintf("Code Signature Algorithm Disabled"));
269    	                Dump_SEC(tprintf("Algorithm 0x%x (SM9)", PROJECT_BINARY_HEADER_ALGORITHM_TYPE_SW_SM9));
270    	
271    	                if (gBCodeGlobalStruct.production_mode) {
272    	                    check_project_binary_error |= PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE;
273    	                }
274    	            }
275    	            break;
276    	
277    	        case PROJECT_BINARY_HEADER_ALGORITHM_TYPE_MLDSA_PREHASH:
278    	            if (0 != gEFuseAPI.efuse_security_setting.mldsa_hash_sha512_disable) {
279    	
280    	                Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE));
281    	                Dump_SEC(tprintf("Code Signature Algorithm Disabled"));
282    	                Dump_SEC(tprintf("Algorithm 0x%x (MLDSA PREHASH)", PROJECT_BINARY_HEADER_ALGORITHM_TYPE_MLDSA_PREHASH));
283    	
284    	                if (gBCodeGlobalStruct.production_mode) {
285    	                    check_project_binary_error |= PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE;
286    	                }
287    	            }
288    	            break;
289    	
290    	        case PROJECT_BINARY_HEADER_ALGORITHM_TYPE_MLDSA_PURE:

292    	
293    	                Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE));

297    	                if (gBCodeGlobalStruct.production_mode) {
298    	                    check_project_binary_error |= PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE;
299    	                }
300    	            }
301    	            break;
302    	
303    	        case PROJECT_BINARY_HEADER_ALGORITHM_TYPE_KAZ:
304    	            if (0 != gEFuseAPI.efuse_security_setting.kaz_disable) {
305    	
306    	                Dump_SEC(tprintf("!!!Check Fail %x: ", (U64)PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE));

322    	    return check_project_binary_error;

327    	/**
328    	 * Checks the CRC (offset 0x9C within the Project Binary Header) of the
329    	 * specified Project Binary. The area of protection provided by the CRC starts
330    	 * from the Project ID within the Project Binary Header (offset 0x80) until the
331    	 * end, not including the Code Signature
332    	 *
333    	 * @param p_project_binary_header
334    	 *  The location of the Project Binary to be checked
335    	 *
336    	 * @return
337    	 *  16-bit bitmap that indicates the errors that have occurred
338    	 *      BIT8 (PROJECT_BINARY_ERROR_CRC)
339    	 *          The CRC is incorrect
340    	 *      BIT13 (PROJECT_BINARY_ERROR_HW_ERROR)
341    	 *          The hardware required to check the CRC has failed
342    	 */
343    	
344    	static U32 check_project_binary_crc(PROJECT_BINARY_HEADER *p_project_binary_header)
345    	{
346    	    U32 check_project_binary_error;
347    	
348    	    volatile U32 *p_crc;
349    	    U32 crc_address, crc_length;
350    	    U32 crc_bin, crc_calc;
351    	
352    	    check_project_binary_error = 0;
353    	
354    	    if (0 == p_project_binary_header->code_sign_finish) {
355    	
356    	        Dump_LoadCode(tprintf("Check CRC: Code Sign CRC"));
357    	
358    	        //Check code_sign_crc Byte 0~511 + Extension
359    	        p_crc = &(p_project_binary_header->code_sign_crc);
360    	
361    	        crc_address = (U32) & (p_project_binary_header->a_code_signature_marking[0]);
362    	        crc_length = (SECTOR_SIZE * p_project_binary_header->project_binary_header_sector_size);
363    	    }
364    	    else {
365    	
366    	        Dump_LoadCode(tprintf("Check CRC: Project Binary Header CRC "));
367    	
368    	        //Check project binary head CRC Byte 128~511 + Extension
369    	        p_crc = &(p_project_binary_header->project_binary_header_crc);
370    	
371    	        crc_address = (U32) & (p_project_binary_header->project_id);
372    	        crc_length = (SECTOR_SIZE * p_project_binary_header->project_binary_header_sector_size) - (crc_address - (U32)p_project_binary_header);
373    	
374    	    }
375    	
376    	    Dump_LoadCode(tprintf("CRC : %x @ %x, Start %x for Size %x", *p_crc, p_crc, crc_address, crc_length));
377    	
378    	    crc_calc = 0;
379    	    crc_bin = *p_crc;
380    	    *p_crc = 0;
381    	
382    	    crc_calc = phal_dmac_crc32((U32)crc_address, crc_length, CRC_CHK_BINARY);
383    	
384    	    //Restore the Binary CRC after calculation has finished
385    	    *p_crc = crc_bin;
386    	
387    	    Dump_LoadCode(tprintf("CRC Bin: %x, CRC Calc: %x", crc_bin, crc_calc));
388    	
389    	    if (crc_calc != crc_bin) {
390    	
391    	        Dump_LoadCode(tprintf("!!!Header CRC Compare Error"));
392    	        check_project_binary_error |= PROJECT_BINARY_ERROR_CRC_HEADER;
393    	    }
394    	
395    	    // Clear crc
396    	    crc_calc = 0;
397    	
398    	    //2. Check CRC of all the sections
399    	    p_crc = &(p_project_binary_header->section_crc);
400    	    crc_bin = *p_crc;
401    	
402    	    crc_address = (U32)(p_project_binary_header) + (SECTOR_SIZE * p_project_binary_header->project_binary_header_sector_size);
403    	    crc_length = (SECTOR_SIZE * (p_project_binary_header->project_binary_sector_size - p_project_binary_header->project_binary_header_sector_size));
404    	
405    	    Dump_LoadCode(tprintf("Check CRC: Section Data CRC %x @ %x, Start %x for Size %x", *p_crc, p_crc, crc_address, crc_length));
406    	
407    	    crc_calc =  phal_dmac_crc32((U32)crc_address, crc_length, CRC_CHK_BINARY);
408    	
409    	    Dump_LoadCode(tprintf("CRC Bin: %x, CRC Calc: %x", (U64)crc_bin, (U64)crc_calc));
410    	
411    	    if (crc_calc != crc_bin) {
412    	        Dump_LoadCode(tprintf("!!!Section Data CRC Compare Error"));
413    	        check_project_binary_error |= PROJECT_BINARY_ERROR_CRC_DATA;
414    	    }
415    	
416    	    // Clear crc
417    	    crc_calc = 0;
418    	
419    	    return check_project_binary_error;
420    	}

422    	static U32 check_code_signature_key_validity(const PROJECT_BINARY_HEADER *p_project_binary_header, U32 calculation_buffer, U8 project_binary_type, U32 project_binary_source)
423    	{
424    	    gSecShaTimeout = 0; // clear flag
425    	
426    	    const EFUSE_CODE_SIGN_KEY_HASH *p_efuse_code_sign_key_hash;
427    	
428    	    // Get All key hash
429    	    U8 compare_key_cnt = 0;
430    	
431    	    //Components of the Project Binary
432    	    U32 data_address, data_size;
433    	    U32 key_address, key_size;
434    	
435    	    U32 key_id;
436    	
437    	    U8 sha2_mode; //mode for hashing public key
438    	    SEC_SHA_PARAMS_T sec_sha_param;
439    	
440    	    U8 is_sha_sm3 = FALSE;
441    	    U8 disable_dmsca = SEC_NO;
442    	
443    	    Dump_SEC(tprintf("Check Code Signature Key Validity"));
444    	    p_efuse_code_sign_key_hash = NULL;
445    	    data_address = (U32)p_project_binary_header;
446    	    data_size = (p_project_binary_header->project_binary_sector_size << SECTOR_SIZE_SHIFT);
447    	
448    	    key_size = get_key_size(p_project_binary_header->algorithm_type);
449    	    if (key_size == 0) {
450    	        return PROJECT_BINARY_ERROR_HEADER_ALGORITHM_TYPE;
451    	    }
452    	
453    	    key_address = data_address + data_size;
454    	
455    	    key_id = *(U32 *)(key_address);
456    	
457    	    Dump_SEC(tprintf("key_address %x", key_address));
458    	    Dump_SEC(tprintf("key_id     %c%c%c%c", ((key_id)&0xFF), ((key_id >> 8)&0xFF), ((key_id >> 16)&0xFF), ((key_id >> 24)&0xFF)));
459    	
460    	    if ((p_project_binary_header->algorithm_type == PROJECT_BINARY_HEADER_ALGORITHM_TYPE_SW_SM2_SM3) || (p_project_binary_header->algorithm_type == PROJECT_BINARY_HEADER_ALGORITHM_TYPE_SW_SM9))
461    	    {
462    	        is_sha_sm3 = TRUE;
463    	    }
464    	
465    	    sal_log_tracker_set(sal_locate(0x40, TYPE_STEP));
466    	
467    	    {
468    	        //Check if the Code Signature Key in the Project Binary is known
469    	        //if type of proj bin is key manifest -> load key from efuse
470    	        //if type of proj bin is loader fw and enable key manifest -> load key from key_manifest_api
471    	        //if type of proj bin is loader fw but disable key manifset -> return error bit map
472    	        if (project_binary_type == FW_PROJECT_BINARY_TYPE) {
473    	            KEY_MANIFEST_API_PTR key_manifest_api_ptr;
474    	
475    	            key_manifest_api_ptr = GET_GP_KEY_MANIFEST_API();
476    	
477    	            if (key_manifest_api_ptr->key_skip == KEY_MANIFEST_NO_SKIP) {
478    	                sync_key_header_info(key_manifest_api_ptr);
479

481    	
482    	                p_efuse_code_sign_key_hash = get_correct_key_hash(key_manifest_api_ptr, ((RSA4096_PUBLIC_KEY *)key_address)->key.id);
483    	                compare_key_cnt = 1;
484    	            }
485    	            else if (key_manifest_api_ptr->key_skip == KEY_MANIFEST_SKIP) {
486    	                sha2_mode = (is_sha_sm3 == FALSE) ? HAL_SEC_SHA2_512_MODE : HAL_SEC_SHA2_SM3_MODE;
487    	                p_efuse_code_sign_key_hash = get_efuse_code_sign_key_hash_valid(project_binary_source, &gEFuseAPI);
488    	                compare_key_cnt = EFUSE_CODE_SIGN_KEY_HASH_COUNT;
489    	            }
490    	            else {
491    	                //else
492    	            }
493    	        }
494    	        else if (project_binary_type == KEY_MANIFEST_PROJECT_BINARY_TYPE) {
495    	            sha2_mode = (is_sha_sm3 == FALSE) ? HAL_SEC_SHA2_512_MODE : HAL_SEC_SHA2_SM3_MODE;
496    	
497    	            p_efuse_code_sign_key_hash = get_efuse_code_sign_key_hash_valid(project_binary_source, &gEFuseAPI);
498    	            compare_key_cnt = EFUSE_CODE_SIGN_KEY_HASH_COUNT;
499    	        }
500    	        else {
501    	            //else
502    	        }
503    	
504    	        if (NULL == p_efuse_code_sign_key_hash) {
505    	
506    	            Dump_SEC(tprintf("!!!Check Fail %x", (U64)PROJECT_BINARY_ERROR_SECURITY_SETTING_UNKNOWN_KEY));
507    	            Dump_SEC(tprintf("Unknown Key"));

511    	            //and key disabled (key state is not valid)
512    	            //Return early here because subsequent checks requires these to be correct
513    	            return PROJECT_BINARY_ERROR_SECURITY_SETTING_UNKNOWN_KEY;
514    	        }
515    	    }
516    	
517    	    sal_log_tracker_set(sal_locate(0x41, TYPE_STEP));
518    	
519    	    {
520    	        //Check if the Code Signature Key in the Project Binary is valid by
521    	        //verifying the hash of the key
522    	        //The hash is performed over the key included in the Project Binary,
523    	        //  before any endian modification
524    	        //This is so that the hash can be calculated straight from the binary
525    	        //  without needing to know how the key looks when it is used
526    	        //The one down side to this is that the hash will change whenever the endian changes
527    	        if ((sha2_mode != HAL_SEC_SHA2_512_MODE) && (sha2_mode != HAL_SEC_SHA2_SM3_MODE)) { // only support sha 512 and sm3
528    	            return PROJECT_BINARY_ERROR_SECURITY_SETTING_INCORRECT_KEY_HASH;
529    	        }
530    	
531    	        memset((void *)&sec_sha_param, 0, sizeof(SEC_SHA_PARAMS_T));
532    	        sec_sha_param.src_addr.all     = (U64)key_address;
533    	        sec_sha_param.src_len          = key_size;
534    	        sec_sha_param.targ_addr.all    = (U64)calculation_buffer;
535    	        sec_sha_param.sha_mode         = sha2_mode;
536    	        sec_sha_param.dummy_en         = FALSE;
537    	        phal_sec_sha_blocking(&sec_sha_param);
538    	
539    	        if (gSecShaTimeout) {
540    	            return PROJECT_BINARY_ERROR_SECURITY_SETTING_TIMEOUT;
541    	        }
542    	
543    	        sal_log_tracker_set(sal_locate(0x42, TYPE_STEP));
544    	
545    	        disable_dmsca = get_efuse_dmsca_compare_disable(&gEFuseAPI);
546    	
547    	        // Compare all efuse key hash //TODO- change to dmsca_def compare
548    	        for (U8 idx = 0; idx < compare_key_cnt; idx++)
549    	        {
550    	            if (HAL_SEC_DMSCA_PASS == phal_sec_verify_signature_compare((U32)(p_efuse_code_sign_key_hash + idx), calculation_buffer, hal_sec_get_digest_len(sha2_mode), disable_dmsca))
551    	            {
552    	                sal_log_tracker_set(sal_locate(0x43, TYPE_STEP));
553    	
554    	                if (EFUSE_PARSE_CPTRA_STATE(gEFuseAPI.efuse_security_setting.u16_union.sqz.cptra_enable))
555    	                {
556    	                    // Truncate 384b for caliptra context
557    	                    phal_sqz_set_pk_hash(calculation_buffer);
558    	                }
559    	                memset((void *)calculation_buffer, 0x00, hal_sec_get_digest_len(sha2_mode));
560    	                Dump_SEC(tprintf("KV OK idx(%d)", idx));
561    	                break;
562    	            }
563    	
564    	            if ((idx + 1) == compare_key_cnt)
565    	            {
566    	                Dump_SEC(tprintf("KV F! cmp(%d)", compare_key_cnt));
567    	                sal_log_tracker_set(sal_locate(0x4F, TYPE_FATAL));
568    	                memset((void *)calculation_buffer, 0x00, hal_sec_get_digest_len(sha2_mode));
569    	                return PROJECT_BINARY_ERROR_SECURITY_SETTING_INCORRECT_KEY_HASH;
570    	            }
571    	        }
572    	    }
573    	
574    	    return 0;
575    	}

858    	    key_endian_buffer       = signature_endian_buffer + SECTOR_SIZE;
859    	
860    	    memcpy((void *)signature_endian_buffer, (void *)signature_address, signature_size);
861    	    memcpy((void *)key_endian_buffer, (void *)key_address, key_size);
862    	
863    	    // Make sure all fields in public key and signature are little endian
864    	    sec_sm2_verify_param_reverse(key_endian_buffer, signature_endian_buffer, p_project_binary_header->pubkey_bledn_sig_bledn);
865    	
866    	    // Init all parameters used in sm2 verify
867    	    code_sign_sw_sm2_init_param(sm2_param_ptr,              // sm2_param_pointer
868    	        key_endian_buffer,       // key_address
869    	        signature_endian_buffer, // sig_address
870    	        hash_buffer,             // hash_buffer
871    	        data_address,            // sm2_msg_addr
872    	        sm2_calculate_buffer);   // sm2_calculate_buffer
873    	
874    	    // step 1. Calculate Z by ID
875    	    code_sign_sw_sm2_calculate_z(sm2_param_ptr, id_address, z_input_address, z_output_address, sha_mode);
876    	
877    	    // step 2. Pad Z with 32 bytes 0
878    	    (void)memset((void *)(z_output_address + 32), 0x00, 32);
879    	
880    	    // step 3. Calculate sm3 value of (z || 0 || code heaer || code body || public key)
881    	    code_sign_sw_sm2_calculate_code_hash(sm2_param_ptr, z_output_address, data_size, key_address, public_key_total_size, sha_mode);
882    	    (void)memset((void *)project_binary_hash, 0x00, HAL_SEC_SHA2_512_DIGEST_LEN);
883    	    (void)memcpy((void *)project_binary_hash, (void *)hash_buffer, HAL_SEC_SM2_BYTE_NUM);
884    	
885    	    sm2_param_ptr->kat_mode = get_efuse_dmsca_compare_disable(&gEFuseAPI); // If kat_mode is set to 1, use memcmp. Otherwise, use dmsca cmp
886    	    // step 4. Verify signature
887    	    if (phal_sec_sm2_verify_algo(sm2_param_ptr) != HAL_SEC_SM2_ERROR(SEC_YES)) {
888    	        Dump_SEC(tprintf("Err"));
889    	        Dump_SEC(tprintf("DSAV F!"));
890    	        memset((void *)project_binary_hash, 0x00, HAL_SEC_SM2_BYTE_NUM);
891    	        memset((void *)hash_buffer, 0x00, HAL_SEC_SM2_BYTE_NUM);
892    	        return PROJECT_BINARY_ERROR_CODE_SIGNATURE;
893    	    }
894    	
895    	    if (gSecAesTimeout | gSecShaTimeout | gSecPkeTimeout) {
896    	        memset((void *)project_binary_hash, 0x00, HAL_SEC_SM2_BYTE_NUM);
897    	        memset((void *)hash_buffer, 0x00, HAL_SEC_SM2_BYTE_NUM);
898    	        return PROJECT_BINARY_ERROR_SECURITY_SETTING_TIMEOUT;
899    	    }
900    	
901    	    Dump_SEC(tprintf("DSAV PA"));
902    	    memset((void *)hash_buffer, 0x00, HAL_SEC_SM2_BYTE_NUM);
903    	    return PROJECT_BINARY_ERROR_NO_ERROR;
904    	}

906    	static void sec_mldsa_pure_batch_sha_calculate(U32 src_addr, U32 tar_addr, U32 pad_size, U32 msg_addr, U32 msg_size)
907    	{
908    	    U32 remain_data_len   = 0U;
909    	    U32 max_msg_addr_size = HAL_SEC_SHA3_SHAKE_256_BLOCK_LEN * 0x787; //0x3FFB8
910    	    U32 sha_copy_addr     = 0U;
911    	    U32 sha_copy_size     = 0U;
912    	
913    	    SEC_SHA_PARAMS_T  sec_sha_param;
914    	
915    	    // CRH(64B) + 2B + CTX length(max 255B) --> align block len (136B)
916    	    sha_copy_addr = msg_addr;
917    	    sha_copy_size = max_msg_addr_size - pad_size;
918    	    remain_data_len = msg_size - sha_copy_size;
919    	
920    	    (void)memcpy((void *)(U32 *)(src_addr + pad_size), (void *)sha_copy_addr, sha_copy_size);
921    	    sha_copy_addr += sha_copy_size;
922    	
923    	    memset((void *)&sec_sha_param, 0, sizeof(SEC_SHA_PARAMS_T));
924    	    sec_sha_param.src_addr.low   = (U32)src_addr;
925    	    sec_sha_param.src_len        = pad_size + sha_copy_size;
926    	    sec_sha_param.targ_addr.low  = (U32)tar_addr;
927    	    sec_sha_param.sha_mode       =  HAL_SEC_SHA3_SHAKE_256_MODE;
928    	    sec_sha_param.shake_out_type = 3U;
929    	    sec_sha_param.timeout_val    = HAL_SEC_BLOCKING_TIME_VAL;
930    	    sec_sha_param.dummy_en       = FALSE;
931    	    sec_sha_param.phase          = HAL_SEC_SHA_FIRST_MODE;
932    	    phal_sec_sha_blocking(&sec_sha_param);
933    	
934    	    while (remain_data_len > max_msg_addr_size)
935    	    {
936    	        (void)memcpy((void *)(U32 *)(src_addr), (void *)sha_copy_addr, max_msg_addr_size);
937    	        sha_copy_addr += max_msg_addr_size;
938    	
939    	        memset((void *)&sec_sha_param, 0, sizeof(SEC_SHA_PARAMS_T));
940    	        sec_sha_param.src_addr.low   = (U32)src_addr;
941    	        sec_sha_param.src_len        = max_msg_addr_size;
942    	        sec_sha_param.targ_addr.low  = (U32)tar_addr;
943    	        sec_sha_param.sha_mode       =  HAL_SEC_SHA3_SHAKE_256_MODE;
944    	        sec_sha_param.shake_out_type = 3U;
945    	        sec_sha_param.timeout_val    = HAL_SEC_BLOCKING_TIME_VAL;
946    	        sec_sha_param.dummy_en       = FALSE;
947    	        sec_sha_param.phase          = HAL_SEC_SHA_MIDDLE_MODE;
948    	        phal_sec_sha_blocking(&sec_sha_param);
949    	
950    	        remain_data_len -= max_msg_addr_size;
951    	    }
952    	
953    	    (void)memcpy((void *)(U32 *)(src_addr), (void *)sha_copy_addr, remain_data_len);
954    	
955    	    memset((void *)&sec_sha_param, 0, sizeof(SEC_SHA_PARAMS_T));
956    	    sec_sha_param.src_addr.low   = (U32)src_addr;
957    	    sec_sha_param.src_len        = remain_data_len;
958    	    sec_sha_param.targ_addr.low  = (U32)tar_addr;
959    	    sec_sha_param.sha_mode       =  HAL_SEC_SHA3_SHAKE_256_MODE;
960    	    sec_sha_param.shake_out_type = 3U;
961    	    sec_sha_param.timeout_val    = HAL_SEC_BLOCKING_TIME_VAL;
962    	    sec_sha_param.dummy_en       = FALSE;
963    	    sec_sha_param.phase          = HAL_SEC_SHA_FINAL_MODE;
964    	    phal_sec_sha_blocking(&sec_sha_param);
965    	}

1127    	        Dump_SEC(tprintf("DSAV F!"));

1129    	        memset((void *)hash_buffer, 0x00, HAL_SEC_SHA2_512_DIGEST_LEN);
1130    	        return PROJECT_BINARY_ERROR_CODE_SIGNATURE;

1134    	        memset((void *)project_binary_hash, 0x00, HAL_SEC_SHA2_512_DIGEST_LEN);

1142    	    return PROJECT_BINARY_ERROR_NO_ERROR;

1799    	/**
1800    	 * Calculates the Chunk Info and Chunk Wordline offset from the
1801    	 * first Chunk Info of the copy and the Code Wordline.
1802    	 *
1803    	 * @param fw_chunk_info_start
1804    	 *  The first Chunk Info the copy
1805    	 * @param copy_wordline
1806    	 *  The Wordline within the copy
1807    	 */
1808    	
1809    	static U32 get_fw_chunk_info_from_slot_wordline(const FW_CHUNK_INFO *a_fw_chunk_info_start, U32 *p_chunk, U32 *p_wordline)
1810    	{
1811    	    U32 start_chunk;
1812    	    U32 wordline_offset;
1813    	
1814    	    U8 chunk;
1815    	
1816    	    Dump_LoadCode(tprintf("Slot Wordline %d ", (U16)*p_wordline));
1817    	
1818    	    //The chunk at input is the starting chunk of this copy
1819    	    start_chunk = *p_chunk;
1820    	    //The wordline at input is the logical slot wordline to look for
1821    	    wordline_offset = *p_wordline;
1822    	
1823    	    for (chunk = start_chunk; chunk < FW_SLOT_CHUNK_COUNT; chunk++) {
1824    	        //If the wordline offset, relative to the start of this chunk fits within this chunk
1825    	        if (a_fw_chunk_info_start[chunk].wordline_count > wordline_offset) {
1826    	            //The chunk at output is now the chunk that contains the input slot wordline
1827    	            *p_chunk = chunk;
1828    	            //The wordline at output is now the wordline within this chunk
1829    	            *p_wordline = wordline_offset;
1830    	
1831    	            Dump_LoadCode(tprintf("is Chunk %d Wordline %d", chunk, (U16)wordline_offset));
1832    	            return (chunk - start_chunk);
1833    	        }
1834    	
1835    	        //Decrease the wordline offset by the number of wordlines in this chunk
1836    	        //This makes the wordline offset now relative to the new chunk
1837    	        wordline_offset -= a_fw_chunk_info_start[chunk].wordline_count;
1838    	    }
1839    	
1840    	    Dump_LoadCode(tprintf("does not exist, exceeded by %d wordline", (U16)wordline_offset));
1841    	
1842    	    //There are no more chunks left to check
1843    	    return FW_SLOT_CHUNK_COUNT;
1844    	}

1846    	static U32 check_fw_slot_info(const FW_SLOT_INFO *p_fw_slot_info)
1847    	{
1848    	    U32 fw_slot_info_error;
1849    	    U32 copy;
1850    	
1851    	    U32 chunk, wordline, relative_chunk;
1852    	
1853    	    Dump_LoadCode(tprintf("Checking FW Slot Info @ %L", (U64)p_fw_slot_info));
1854    	
1855    	    fw_slot_info_error = 0;
1856    	
1857    	    //Check the Node per Wordline
1858    	    if (FLH_MAX_LC_PER_PAGE < p_fw_slot_info->node_per_wordline) {
1859    	        Dump_LoadCode(tprintf("Invalid %d Nodes per Wordline, ", p_fw_slot_info->node_per_wordline));
1860    	        Dump_LoadCode(tprintf("exceed %d", FLH_MAX_LC_PER_PAGE));
1861    	        fw_slot_info_error |= FW_SLOT_INFO_ERROR_INVALID_NODE_PER_WORDLINE;
1862    	    }
1863    	
1864    	    //Check that the FW Slot Size does not exceed the available buffer size
1865    	    if (DBUF_API_BOOTCODE_DATA_BUFFER_SIZE < ((p_fw_slot_info->total_wordline_count * p_fw_slot_info->node_per_wordline) << LC_BYTE_SIZE_SHIFT)) {
1866    	        Dump_LoadCode(tprintf("Invalid FW Slot size %x, ", ((p_fw_slot_info->total_wordline_count * p_fw_slot_info->node_per_wordline) << LC_BYTE_SIZE_SHIFT)));
1867    	        Dump_LoadCode(tprintf("exceed %x", DBUF_API_BOOTCODE_DATA_BUFFER_SIZE));
1868    	        fw_slot_info_error |= FW_SLOT_INFO_ERROR_INVALID_FW_SLOT_SIZE;
1869    	    }
1870    	
1871    	    //Check the Copy Count
1872    	    if ((FW_SLOT_COPY_COUNT < p_fw_slot_info->copy_count) || (0 == p_fw_slot_info->copy_count)) {
1873    	        Dump_LoadCode(tprintf("Invalid Copy Count %d, ", p_fw_slot_info->copy_count));
1874    	        fw_slot_info_error |= FW_SLOT_INFO_ERROR_INVALID_COPY_COUNT;
1875    	    }
1876    	
1877    	    for (copy = 0; copy < p_fw_slot_info->copy_count; copy++) {
1878    	        // check copy cnt
1879    	        if (copy >= FW_SLOT_COPY_COUNT) {
1880    	            // invalid copy cnt, not do anything
1881    	        }
1882    	        //Check the Chunk Offset
1883    	        else if ((copy < FW_SLOT_COPY_COUNT) && (FW_SLOT_CHUNK_COUNT < p_fw_slot_info->a_chunk_offset[copy])) {
1884    	            Dump_LoadCode(tprintf("Invalid Chunk Offset for %d ", p_fw_slot_info->a_chunk_offset[copy]));
1885    	            Dump_LoadCode(tprintf("Copy %d, ", (U16)copy));
1886    	            Dump_LoadCode(tprintf("exceed %d", FW_SLOT_CHUNK_COUNT));
1887    	            fw_slot_info_error |= FW_SLOT_INFO_ERROR_INVALID_CHUNK_OFFSET;
1888    	        }
1889    	        else {
1890    	            //Check that the last Wordline of each Copy can be found within the Chunk Infos
1891    	            chunk = p_fw_slot_info->a_chunk_offset[copy];
1892    	            wordline = p_fw_slot_info->total_wordline_count - 1;
1893    	            relative_chunk = get_fw_chunk_info_from_slot_wordline(p_fw_slot_info->a_chunk_info, &chunk, &wordline);
1894    	            if (FW_SLOT_CHUNK_COUNT <= relative_chunk) {
1895    	                Dump_LoadCode(tprintf("Invalid Starting Chunk Info %d ", (U16)chunk));
1896    	                Dump_LoadCode(tprintf("for Copy %d, ", (U16)copy));
1897    	                Dump_LoadCode(tprintf("can not find last wordline %d", (U16)wordline));
1898    	                fw_slot_info_error |= FW_SLOT_INFO_ERROR_INVALID_CHUNK_INFO;
1899    	            }
1900    	        }
1901    	    }
1902    	
1903    	    return fw_slot_info_error;
1904    	}
