36    	    SEC_RSA_ENCODING_PARAMS_T_PTR sqz_param = (SEC_RSA_ENCODING_PARAMS_T_PTR)phal_sqz_get_param_buf_base();
37    	    memset((void *)sqz_param, 0, sizeof(SEC_RSA_ENCODING_PARAMS_T));
38    	    memcpy((void *)sqz_param, params, sizeof(SEC_RSA_ENCODING_PARAMS_T));
39    	
40    	    MR_CQ_PARAM cq;
41    	    MR_SQ_PARAM sq = form_sq(sqz_param, MR_CODE_SIGN, RSA_PSS_VERIFY, RSA_PSS_VERIFY_TAG);
42    	    hal_db_push_block(&sq, 0, MR_QID_CPU0_SQ_SQZ_CMD);
