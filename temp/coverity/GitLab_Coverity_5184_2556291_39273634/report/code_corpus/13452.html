461    	static void mctp_message_handle_nvme_mi_command(MCTP_MESSAGE_TRANSMIT_HANDLER *p_transmit_handler)
462    	{
463    	    HOST_API_STRUCT_PTR host = (HOST_API_STRUCT_PTR)&gHostApiStruct;
464    	
465    	    NVME_MI_CMD_REQ_MSG *p_nvme_mi_cmd_req_msg;
466    	    NVME_MI_CMD_RESP_MSG *p_nvme_mi_cmd_resp_msg;
467    	
468    	    DATA_DESCRIPTOR request_data;
469    	    DATA_DESCRIPTOR response_data;
470    	
471    	    U32 message_response_status;
472    	
473    	    U8 opcode;
474    	
475    	    Dump_SMBUS(tprintf(" [MCTP NVME-MI] Receive Message Count : %d", p_transmit_handler->request_message_handler.byte_count));
476    	
477    	    //TODO should actually set the p_nvme_mi_cmd_resp_msg and response_data_base into the response buffer
478    	    //But that requires changing the underlying structure of constructing the response messages
479    	    p_nvme_mi_cmd_req_msg = (NVME_MI_CMD_REQ_MSG *)(p_transmit_handler->request_message_handler.buffer.p_base);
480    	    p_nvme_mi_cmd_resp_msg = (NVME_MI_CMD_RESP_MSG *)(p_transmit_handler->reponse_packet_handler.buffer.p_base);
481    	
482    	    memset((void *)p_nvme_mi_cmd_resp_msg, 0, sizeof(NVME_MI_CMD_RESP_MSG));
483    	
484    	    //This assumes that all NVME MI commands' response data are handled in host->buffer_base
485    	    response_data.p_base = (U8 *)(host->buffer_base);
486    	    response_data.size = 0;
487    	
488    	    message_response_status = NVME_MI_RSP_MSG_STS_SUCCESS;
489    	
490    	    //The minimum number of bytes required to form a NVME MI Command requires is
491    	    // (16 byte) : mi command request format
492    	    // (N  byte) : Request Data (Optional)
493    	    // (4  byte) : Message Integrity Check (MIC)
494    	    if (p_transmit_handler->request_message_handler.byte_count >= (sizeof(NVME_MI_CMD_REQ_MSG) + 4U)) {
495    	        // NVMe Request Data have data
496    	        request_data.p_base = (p_transmit_handler->request_message_handler.buffer.p_base + sizeof(NVME_MI_CMD_REQ_MSG));
497    	        request_data.size = (p_transmit_handler->request_message_handler.byte_count - (sizeof(NVME_MI_CMD_REQ_MSG) + 4U));
498    	
499    	        //the copy is essentially useless, its basically to match the admin command handle in case it handles things in the host buffer
500    	        //The NVMe-MI message command request data in the host->buffer_base handle
501    	        memcpy((void *)(host->buffer_base), (void *) & (request_data.p_base[0U]), request_data.size);
502    	    }
503    	    else {
504    	        //Not enough data to properly form the command
505    	        message_response_status |= NVME_MI_RSP_MSG_STS_INV_CMD_SIZE;
506    	    }
507    	
508    	    if (NVME_MI_RSP_MSG_STS_SUCCESS == message_response_status) {
509    	        // Init response header + format (8 byte)
510    	        memset((void *)p_nvme_mi_cmd_resp_msg, 0, sizeof(NVME_MI_CMD_RESP_MSG));
511    	
512    	        opcode = p_nvme_mi_cmd_req_msg->opcode;
513    	
514    	        switch (opcode) {
515    	        case NVME_MI_CMD_READ_MI_STRUCT:     // 0x00     (M)
516    	            mctp_message_handle_nvme_mi_message_read_mi_struct(p_nvme_mi_cmd_req_msg, p_nvme_mi_cmd_resp_msg, (U32)(response_data.p_base));
517    	            if (NVME_MI_RSP_MSG_STS_SUCCESS == p_nvme_mi_cmd_resp_msg->dw1.generic_error.status) {
518    	                response_data.size = p_nvme_mi_cmd_resp_msg->dw1.nvme_mi_command.nmresp;
519    	            }
520    	            break;
521    	        case NVME_MI_CMD_SUBSYS_HEALTH_POLL: // 0x01     (M)
522    	            // bootcode not support
523    	            Dump_SMBUS(tprintf("    >>>>> Subsystem Health Status Poll (Bootcode not support)"));
524    	            message_response_status |= NVME_MI_RSP_MSG_STS_INV_CMD_OPC;
525    	            break;
526    	        case NVME_MI_CMD_CTRL_HEALTH_POLL:   // 0x02     (M)
527    	            // bootcode not support
528    	            Dump_SMBUS(tprintf("    >>>>> Controller Health Status Poll (Bootcode not support)"));
529    	            message_response_status |= NVME_MI_RSP_MSG_STS_INV_CMD_OPC;
530    	            break;
531    	        case NVME_MI_CMD_CONFIG_SET:         // 0x03     (M)
532    	            mctp_message_handle_nvme_mi_message_configuration_set(p_nvme_mi_cmd_req_msg, p_nvme_mi_cmd_resp_msg, p_transmit_handler->message_info.p_transport);
533    	            break;
534    	        case NVME_MI_CMD_CONFIG_GET:         // 0x04     (M)
535    	            mctp_message_handle_nvme_mi_message_configuration_get(p_nvme_mi_cmd_req_msg, p_nvme_mi_cmd_resp_msg, p_transmit_handler->message_info.p_transport);
536    	            break;
537    	        default:
538    	            message_response_status |= NVME_MI_RSP_MSG_STS_INV_CMD_OPC;
539    	            break;
540    	        }
541    	    }
542    	
543    	    p_nvme_mi_cmd_resp_msg->dw1.nvme_mi_command.status |= message_response_status;
544    	
545    	    //Reset the Transmit Handler to idle
546    	    mctp_message_transmit_handler_reset(p_transmit_handler);
547    	
548    	    mctp_message_handle_nvme_mi_message_response(
549    	        &p_transmit_handler->message_info,
550    	        p_nvme_mi_cmd_req_msg->mi_header.csi,
551    	        NMIMT_NVME_MI_COMMAND,
552    	        (U8 *)p_nvme_mi_cmd_resp_msg,
553    	        sizeof(NVME_MI_CMD_RESP_MSG),
554    	        response_data.p_base,
555    	        response_data.size);
556    	}

564    	
565    	    DATA_DESCRIPTOR request_data;
566    	    DATA_DESCRIPTOR response_data;
567    	
568    	    U32 message_response_status;
569    	
570    	    U8 request_data_exists = FALSE;
571

581    	
582    	    message_response_status = NVME_MI_RSP_MSG_STS_SUCCESS;
583    	
584    	    // (68 byte) : admin command request format
585    	    // (N  byte) : NVMe Request Data (Optional)
586    	    // (4  byte) : Message Integrity Check (MIC)
587    	    if (p_transmit_handler->request_message_handler.byte_count >= (sizeof(NVME_MI_NVME_ADMIN_CMD_REQ) + 4U)) {

595    	        request_data.size = 0UL;
596    	    }

602    	        Dump_SMBUS(tprintf("    >>>>> Admin Command Size : %d", p_transmit_handler->request_message_handler.byte_count));
603    	
604    	    }
605    	    else if (chk_mctp_mi_adm_prohibit(p_nvme_admin_cmd_req->sqdw0.opcode) == TRUE) {
606    	        message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
607    	        p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_OPC;
608    	    }

610    	    else if (chk_mctp_mi_adm_support(p_nvme_admin_cmd_req->sqdw0.opcode) == FALSE) {
611    	        message_response_status |= NVME_MI_RSP_MSG_STS_INV_CMD_OPC;
612    	        p_nvme_admin_cmd_resp->dw1.generic_resp.status = message_response_status;
613    	    }
614    	    else if ((chk_mctp_mi_adm_need_data_out(p_nvme_admin_cmd_req->sqdw0.opcode) == TRUE) && (p_nvme_admin_cmd_req->dlen == 0U)) {
615    	        // need return an Invalid Parameter Error Response
616    	        message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
617    	        p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_DLEN;
618    	        Dump_SMBUS(tprintf(" [MCTP NVME-MI] Admin Command need transmit data, but DLEN is 0!"));
619    	    }
620    	
621    	    //Check Command Flags Bit 0 and the DLEN fields
622    	    //Figure 112: NVMe Admin Command Request Description
623    	    //--Byte 05
624    	    //--Byte 35:32
625    	    //if (0U == (p_nvme_admin_cmd_req->sqdw0.cmdflag & BIT0))   //X2 follow MI version 1.2 don't check cmdflag bit 0 & bit 1
626    	    //{
627    	    //    // Bit 0, if cleared to ���0���, then the DLEN field shall be cleared to 0h.
628    	    //    if (0U != p_nvme_admin_cmd_req->dlen)
629    	    //    {
630    	    //        // need return an Invalid Parameter Error Response
631    	    //        message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
632    	    //        Dump_SMBUS(tprintf(" [MCTP NVME-MI] DLEN != 0"));
633    	    //        Dump_SMBUS(tprintf("    >>>>> DLEN : %d", p_nvme_admin_cmd_req->dlen));
634    	    //    }
635    	    //}
636    	    //else
637    	    // Bits 0 and 1 of this field shall be cleared to ���0���
638    	    else if ((p_nvme_admin_cmd_req->dlen & (BIT1 | BIT0)) != 0U) {
639    	        // need return an Invalid Parameter Error Response
640    	        message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
641    	        p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_DLEN;
642    	        Dump_SMBUS(tprintf(" [MCTP NVME-MI] DLEN BIT1, BIT0 != 0"));
643    	        Dump_SMBUS(tprintf("    >>>>> DLEN : %d", p_nvme_admin_cmd_req->dlen));
644    	    }
645    	
646    	    // This field shall be less than or equal to 4,096.
647    	    else if ((p_nvme_admin_cmd_req->dofst + p_nvme_admin_cmd_req->dlen) > NVME_MI_ADMIN_COMMAND_DLEN_MAX) {
648    	        // need return an Invalid Parameter Error Response
649    	        message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
650    	        p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_DLEN;
651    	        Dump_SMBUS(tprintf(" [MCTP NVME-MI] DLEN > ", NVME_MI_ADMIN_COMMAND_DLEN_MAX));
652    	        Dump_SMBUS(tprintf("    >>>>> DLEN : %d", p_nvme_admin_cmd_req->dlen));
653    	    }
654    	
655    	    //Figure 112: NVMe Admin Command Request Description
656    	    //--Byte 05
657    	    //--Byte 31:28
658    	    //Check Command Flags Bit 1 and the DOFST fields
659    	    //if (0U == (p_nvme_admin_cmd_req->sqdw0.cmdflag & BIT1))   //X2 follow MI version 1.2 don't check cmdflag bit 0 & bit 1
660    	    //{
661    	    //    // Bit 1, if cleared to ���0���, then the DOFST field shall be cleared to 0h.
662    	    //    if (0U != p_nvme_admin_cmd_req->dofst)
663    	    //    {
664    	    //        // need return an Invalid Parameter Error Response
665    	    //        message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
666    	    //        Dump_SMBUS(tprintf(" [MCTP NVME-MI] DOFST != 0"));
667    	    //        Dump_SMBUS(tprintf("    >>>>> DOFST : %d", p_nvme_admin_cmd_req->dofst));
668    	    //    }
669    	    //}
670    	    //else
671    	    // Bits 0 and 1 of this field shall be cleared to ���0���
672    	    else if ((p_nvme_admin_cmd_req->dofst & (BIT1 | BIT0)) != 0U) {
673    	        // need return an Invalid Parameter Error Response
674    	        message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
675    	        p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_DOFST;
676    	        Dump_SMBUS(tprintf(" [MCTP NVME-MI] DOFST BIT1, BIT0 != 0"));
677    	        Dump_SMBUS(tprintf("    >>>>> DOFST : %d", p_nvme_admin_cmd_req->dofst));
678    	    }
679    	
680    	    //else if (p_nvme_admin_cmd_req->sqdw0.ctrlrid == 0U)
681    	    //{
682    	    //   // TODO
683    	    //   // This field specifies the Controller ID of the Controller that this command targets.
684    	    //}
685    	
686    	    else if (p_nvme_admin_cmd_req->sqdw0.opcode == ADM_OPC_GET_LOG_PAGE) {
687    	        U32 numd = ((p_nvme_admin_cmd_req->sqdw11 << 16U) | (U32)(p_nvme_admin_cmd_req->sqdw10 >> 16U));
688    	        U16 byte_cnt = 4U * (numd + 1U); // dw: 0's base, 1DW = 4 bytes
689    	        // check if dlen is greater than number of dword
690    	        if (p_nvme_admin_cmd_req->dlen > byte_cnt) {
691    	            // need return an Invalid Parameter Error Response
692    	            message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
693    	            p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_DLEN;
694    	            Dump_SMBUS(tprintf(" [MCTP NVME-MI] DLEN is greater than number of dword!"));
695    	            Dump_SMBUS(tprintf("    >>>>> DLEN : %d", p_nvme_admin_cmd_req->dlen));
696    	            Dump_SMBUS(tprintf("    >>>>> Number of Dword : %d", numd));
697    	        }
698    	        // check if dofst is greater than number of dword
699    	        else if (p_nvme_admin_cmd_req->dofst > byte_cnt) {
700    	            // need return an Invalid Parameter Error Response
701    	            message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
702    	            p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_DOFST;
703    	            Dump_SMBUS(tprintf(" [MCTP NVME-MI] DOFST is greater than number of dword!"));
704    	            Dump_SMBUS(tprintf("    >>>>> DOFST : %d", p_nvme_admin_cmd_req->dofst));
705    	            Dump_SMBUS(tprintf("    >>>>> Number of Dword : %d", numd));
706    	        }
707    	        // check if dlen+dofst is greater than number of dword
708    	        else if ((p_nvme_admin_cmd_req->dlen + p_nvme_admin_cmd_req->dofst) > byte_cnt) {
709    	            // need return an Invalid Parameter Error Response
710    	            message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
711    	            p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_DLEN;
712    	            Dump_SMBUS(tprintf(" [MCTP NVME-MI] DLEN+DOFST is greater than number of dword!"));
713    	            Dump_SMBUS(tprintf("    >>>>> DLEN : %d", p_nvme_admin_cmd_req->dlen));
714    	            Dump_SMBUS(tprintf("    >>>>> DOFST : %d", p_nvme_admin_cmd_req->dofst));
715    	            Dump_SMBUS(tprintf("    >>>>> Number of Dword : %d", numd));
716    	        }
717    	        else {
718    	            /*do nothing*/
719    	        }
720    	    }
721    	    //If any request data is given, assume that this is Admin Command is expecting it
722    	    //Actually checking what the Admin Command wants against what is given is not performed.
723    	    else if (request_data.size != 0U) {
724    	        Dump_SMBUS(tprintf(" [MCTP NVME-MI] Request Data Size"));
725    	        Dump_SMBUS(tprintf("    >>>>> Request Data Size   : %d", request_data.size));
726    	        Dump_SMBUS(tprintf("    >>>>> Request Buffer Base : %x", (U32)(request_data.p_base)));
727    	
728    	        request_data_exists = TRUE;
729    	
730    	        // For commands that transmit data from the Management Controller to the Management Endpoint
731    	        // (i.e., the Request Data field in the Request Message has non-zero length) or do not transmit data,
732    	        // this field shall be cleared to 0h. If this field is not 0h, then the Management Endpoint shall return an Invalid Parameter Error Response.
733    	        if (p_nvme_admin_cmd_req->dofst != 0U) {
734    	            // need return an Invalid Parameter Error Response
735    	            message_response_status |= NVME_MI_RSP_MSG_STS_INV_PARAM;
736    	            p_nvme_admin_cmd_resp->dw1.invld_param_err_resp.all = PEL_DOFST;
737    	            Dump_SMBUS(tprintf(" [MCTP NVME-MI] Request Data Size != 0 + DOFST != 0"));
738    	            Dump_SMBUS(tprintf("    >>>>> Request Data Size : %d", request_data.size));
739    	            Dump_SMBUS(tprintf("    >>>>> DOFST             : %d", p_nvme_admin_cmd_req->dofst));
740    	        }
741    	        else if (p_nvme_admin_cmd_req->dlen != request_data.size) {
742    	            // need return an Invalid Parameter Error Response
743    	            message_response_status |= NVME_MI_RSP_MSG_STS_INV_CMD_INPUT_DATA_SIZE;
744    	            p_nvme_admin_cmd_resp->dw1.generic_resp.status = message_response_status;
745    	            Dump_SMBUS(tprintf(" [MCTP NVME-MI] Request Data Size != DLEN"));
746    	            Dump_SMBUS(tprintf("    >>>>> Request Data Size : %d", request_data.size));
747    	            Dump_SMBUS(tprintf("    >>>>> DLEN              : %d", p_nvme_admin_cmd_req->dlen));
748    	        }
749    	        else {
750    	            /*do nothing*/
751    	        }
752    	    }
753    	    else {
754    	        request_data_exists = FALSE;
755    	    }
756    	
757    	    if (NVME_MI_RSP_MSG_STS_SUCCESS == message_response_status) {
758    	        mctp_message_handle_set_fake_admin_param(host, p_nvme_admin_cmd_req, request_data_exists, &request_data);
759    	
760    	        nvme_adm_cmd_handler(host);
761    	
762    	        mctp_message_handle_get_admin_cq_param(host, p_nvme_admin_cmd_req, p_nvme_admin_cmd_resp, request_data_exists, &response_data);
763    	    }
764    	
765    	    p_nvme_admin_cmd_resp->dw1.generic_resp.status = message_response_status;
766    	
767    	    //Reset the Transmit Handler to idle
768    	    mctp_message_transmit_handler_reset(p_transmit_handler);
769    	
770    	    mctp_message_handle_nvme_mi_message_response(
771    	        &p_transmit_handler->message_info,
772    	        p_nvme_admin_cmd_req->mi_header.csi,
773    	        NMIMT_NVME_ADMIN_COMMAND,
774    	        (U8 *)p_nvme_admin_cmd_resp,
775    	        sizeof(NVME_MI_NVME_ADMIN_CMD_RESP),
776    	        (U8 *)response_data.p_base,
777    	        response_data.size);
778    	
779    	}

781    	U32 mctp_message_handle_nvme_mi_message(MCTP_MESSAGE_TRANSMIT_HANDLER *p_transmit_handler)
782    	{
783    	    U32 crc_32c;
784    	    U32 message_response_status;
785    	    NVME_MI_MSG_HEADER_PTR p_nvme_mi_msg_header = (NVME_MI_MSG_HEADER_PTR)p_transmit_handler->request_message_handler.buffer.p_base;
