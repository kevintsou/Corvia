69    	static U8 xmodem_rcv_check_packet(XMOD_STRUCT_PTR xmodem)
70    	{
71    	    U8 result = XMODEM_PACKET_CHK_PASS;
72    	    U8 chk_sum;
73    	    U16 crc;
74    	
75    	    /*
76    	     *  To prevent overflow issue, (xmodem->packet_id + 1) should type cast to U8
77    	     */
78    	    if (xmodem->token[PACKET_TOK] == (U8)(xmodem->packet_id)) {
79    	        result = XMODEM_PACKET_CHK_PACKET_RESEND;
80    	        Dump_XModem(tprintf("Packet ID: 0x%x is resent 0x%x", xmodem->token[PACKET_TOK], (xmodem->packet_id)));
81    	    }
82    	    else if (xmodem->token[PACKET_TOK] != (U8)(xmodem->packet_id + 1)) {
83    	        result = XMODEM_PACKET_CHK_PACKET_FAIL;
84    	        Dump_XModem(tprintf("Packet ID: 0x%x Fail, should be 0x%x", xmodem->token[PACKET_TOK], (xmodem->packet_id + 1)));
85    	    }
86    	    else {
87    	        // correct packet ID
88    	    }
89    	
90    	    if ((xmodem->token[PACKET_TOK] + xmodem->token[N_PACKET_TOK]) != 0xFF) {
(1) Event misra_c_2012_rule_10_4_violation: Essential type of the left hand operand "xmodem->token[PACKET_TOK] + xmodem->token[N_PACKET_TOK]" (unsigned) is not the same as that of the right operand "255"(signed).
91    	        result = XMODEM_PACKET_CHK_N_PACKET_FAIL;
92    	        Dump_XModem(tprintf("Packet N ID: 0x%x Fail, should be 0x%x", xmodem->token[N_PACKET_TOK], (0xFF - xmodem->token[PACKET_TOK])));
93    	    }
94    	
95    	    if (XMOD_CRC16 == xmodem->protect_mode) {
96    	        /*
97    	         *  Check CRC
98    	         */
99    	        crc = calculate_crc16((U32)xmodem->xmod_buf, xmodem->packet_size, 0);
100    	        if (crc != ((xmodem->token[CRC_1_TOK] << 8) | xmodem->token[CRC_2_TOK])) {
101    	            result = XMODEM_PACKET_CHK_CRC_FAIL;
102    	            Dump_XModem(tprintf("CRC: 0x%x Fail, should be 0x%x", ((xmodem->token[CRC_1_TOK] << 8) | xmodem->token[CRC_2_TOK]), crc));
103    	        }
104    	    }
105    	    else {
106    	        /*
107    	         *  Check SUM
108    	         */
109    	        chk_sum = calculate_chksum((U32)xmodem->xmod_buf, xmodem->packet_size);
110    	        if (chk_sum != xmodem->token[CRC_1_TOK]) {
111    	            result = XMODEM_PACKET_CHK_CHK_SUM_FAIL;
112    	            Dump_XModem(tprintf("Check Sum: 0x%x Fail, should be 0x%x", xmodem->token[CRC_1_TOK], chk_sum));
113    	        }
114    	    }
115    	
116    	    return result;
117    	}
