144    	 *  REG_OFFSET [         0        ][         1        ] => right shift itc_id by 2 to retrieve register offset
145    	 *             [       0 : 31     ][       0 : 31     ]
146    	 *  itc_id     [ 0 ][ 1 ][ 2 ][ 3 ][ 4 ][ 5 ][ 6 ][ 7 ] => module itc_id by 4 and left shift 3 to retrieve itc_id offset
147    	 *             [0:7][0:7][0:7][0:7][0:7][0:7][0:7][0:7]
148    	 *
149    	 *  Ex. itc_id: 6 => right shift itc_id by 2 to retrieve register offset, 6 >> 2 => reg_offset 1
150    	 *      itc_id: 6 => module itc_id by 4 and left shift 3 to retrieve itc_id offset (6 % 4) << 3 => itc_id offset 16
151    	 *
152    	 */
153    	void dal_gic_set_target(U16 itc_id, U8 itc_target)
154    	{
155    	    ASSERT(4445U, (CHK_ONE_BIT(itc_target, GIC_INT_TARGET_TO_CPU0) == TRUE || CHK_ONE_BIT(itc_target, GIC_INT_TARGET_TO_CPU1) == TRUE  || CHK_ONE_BIT(itc_target, GIC_INT_TARGET_TO_CPU2) == TRUE));
156    	    r8_GICD[GICDB_ITARGETSRN + itc_id] = itc_target;
157    	}
158    	void dal_gic_clr_target(U16 itc_id)
159    	{
160    	    dal_gic_set_target(itc_id, 0U);
161    	}

164    	 *  Level-sensitive or edge-triggerd is configured by bit[1].
165    	 *  If bit[1] is 0, interrupt is triggered by level-sensitive.
166    	 *  If bit[1] is 1, interrupt is triggered by edge-triggered.
167    	 */

(1) Event misra_c_2012_rule_10_7_violation: The width of the composite expression "itc_id >> 4U" (16 bits) is less that the left hand operand "768UL" (32 bits).
