410    	    /*
411    	     *  We call this function while device is !healthy and could not be recovered internally in each stage (e.g. bootcode, loader, fw)
412    	     *  The device will then stay in bootcode and update device status by cases prior to OCP recovery process
413    	     */
414    	
415    	    ACROT_DATA *pACRotData = GetACRotDataPtr();
416    	
417    	    Dump_OCP(tprintf("[OCP] Update Device Status (Current: %x)", pACRotData->sDeviceStatus.ubDeviceStatus));
418    	
419    	    /*
420    	     *  There are several cases by device status
421    	     *
422    	     *  1. DS_BOOT_FAILURE
423    	     *     This is the only status that can populate a recovery reason code
424    	     *     and the RCV value indicates if the device can be recovered based on error code
425    	     *     (1) RCV = Y
426    	     *          - Switch device status to recovery mode (0x3)
427    	     *     (2) RCV = N/M
428    	     *          - Let host to determine if need to forced recovery
429    	     *
430    	     *  2. DS_DEVICE_ERROR/DS_FATAL_ERROR
431    	     *     These two status don't populate recovery reason code
432    	     *     - Let host to determine if need to forced recovery
433    	     */
434    	    switch (pACRotData->sDeviceStatus.ubDeviceStatus)
435    	    {
436    	        case DS_BOOT_FAILURE:
437    	        case DS_DEVICE_ERROR:
438    	        case DS_FATAL_ERROR:
439    	            if (DS_BOOT_FAILURE == pACRotData->sDeviceStatus.ubDeviceStatus)

440    	            {
441    	                if (M_IS_RRC_RCV_Y(pACRotData->sDeviceStatus.uwRecoveryReasonCode))
442    	                {
443    	                    RpDeviceStatusManagement(pACRotData, E_RP_STATE_UPDATE_STATUS_BOOT_FAILURE_WITH_RCV_Y);
444    	                }
445    	                /* coverity[unsigned_compare] */
446    	                else if (M_IS_RRC_RCV_N_M(pACRotData->sDeviceStatus.uwRecoveryReasonCode))
447    	                {
448    	                    RpDeviceStatusManagement(pACRotData, E_RP_STATE_UPDATE_STATUS_BOOT_FAILURE_WITH_RCV_N_M);
449    	                }
450    	            }
451    	            else if ((DS_DEVICE_ERROR == pACRotData->sDeviceStatus.ubDeviceStatus) ||
452    	                        (DS_FATAL_ERROR  == pACRotData->sDeviceStatus.ubDeviceStatus))
453    	            {
454    	                // device error & fatal error
455    	                RpDeviceStatusManagement(pACRotData, E_RP_STATE_UPDATE_STATUS_DEVICE_ERROR_FATAL_ERROR);
456    	            }
457    	            OcpRecoveryUpdateSubsystemManagementData(pACRotData);
458    	            break;
459    	
460    	        case DS_RUNNING_RECOVERY_IMAGE:
461    	            RpDeviceStatusManagement(pACRotData, E_RP_STATE_UPDATE_STATUS_RUNNING_RECOVERY_IMAGE);
462    	            break;
463    	    }
464    	}
