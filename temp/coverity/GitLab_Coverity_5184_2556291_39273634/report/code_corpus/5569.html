224    	   @brief      Check if the recieved command is a valid ocp recovery command and can be accepted
225    	
226    	   @param      ubCmdType         type of command (RO/RW) issued by RA
227    	   @param      ubCmdCode         recieved command code
228    	   @param      pSmbsTx           pointer to SMBus slave TX buffer
229    	   @return     BOOL
230    	 */
231    	BOOL CheckIsOcpRecoveryCmdCode(U8 ubCmdType, U8 ubCmdCode, SMBUS_BUFFER_T *pSmbsTx)
232    	{
233    	    BOOL bIsCmdValid = FALSE;
234    	
235    	    ACROT_DATA *pACRotData = GetACRotDataPtr();
236    	
237    	    Dump_OCP(tprintf(""));
238    	    Dump_OCP(tprintf("[OCP] Receive OCP %s Command (CmdCode: %d)", ((RP_CMD_TYPE_WRITE == ubCmdType) ? "Write" : "Read"), ubCmdCode));
239    	
240    	    // There are 3 prerequisites to be a valid command:
241    	    do {
242    	        // 1. It's a recovery command (cmdcode = 34h ~ 44h)
243    	        // 2. The protocol must match the command type (RW for both block write/read while RO is only for block read)
244    	        if ((!M_IS_OCP_RECOVERY_COMMAND(ubCmdCode)) ||
245    	            ((RP_CMD_TYPE_WRITE == ubCmdType) && M_IS_OCP_RECOVERY_RO_COMMAND(ubCmdCode)))
246    	        {
247    	            Dump_OCP(tprintf("  - Unsupported or write RO command"));
248    	            M_OCP_RECOVERY_SET_PROTOCOL_ERROR(pACRotData, PE_UNSUPPORTED_OR_WRITE_COMMAND);
249    	            break;
250    	        }
251    	
252    	        // 3. The command should be supported in PROT_CAP
253    	        if (((RECOVERY_CMD_DEVICE_ID == ubCmdCode)      && (FALSE == pACRotData->sProtCap.sCapabilities.sAgent.btIdentification))       ||
(1) Event misra_c_2012_rule_10_4_violation: Essential type of the left hand operand "0" (boolean) is not the same as that of the right operand "pACRotData->sProtCap.sCapabilities.sAgent.btIdentification"(unsigned).
254    	            ((RECOVERY_CMD_DEVICE_STATUS == ubCmdCode)  && (FALSE == pACRotData->sProtCap.sCapabilities.sAgent.btDeviceStatus))         ||
255    	            ((RECOVERY_CMD_HW_STATUS == ubCmdCode)      && (FALSE == pACRotData->sProtCap.sCapabilities.sAgent.btHwStatus))             ||
256    	            ((RECOVERY_CMD_INDIRECT_CTRL == ubCmdCode)  && (FALSE == pACRotData->sProtCap.sCapabilities.sAgent.btRecoveryMemoryAccess)) ||
257    	            ((RECOVERY_CMD_VENDOR == ubCmdCode)         && (FALSE == pACRotData->sProtCap.sCapabilities.sAgent.btVendorsCommand)))
258    	        {
259    	            Dump_OCP(tprintf("  - Unsupported in PROT_CAP"));
260    	            M_OCP_RECOVERY_SET_PROTOCOL_ERROR(pACRotData, PE_UNSUPPORTED_OR_WRITE_COMMAND);
261    	
262    	            if (RP_CMD_TYPE_READ == ubCmdType)
263    	            {
264    	                bRpReadUnsupportedCmd = TRUE;
265    	            }
266    	            break;
267    	        }
268    	
269    	        // 4. In cases of R-scope commands, they're only accepted when device status = !healthy
270    	        if (M_IS_OCP_RECOVERY_R_SCOPE_COMMAND(ubCmdCode) && (DS_STATUS_PENDING == pACRotData->sDeviceStatus.ubDeviceStatus))
271    	        {
272    	            Dump_OCP(tprintf("  - R-scope command"));
273    	
274    	            if (RP_CMD_TYPE_READ == ubCmdType)
275    	            {
276    	                bRpReadUnsupportedCmd = TRUE;
277    	            }
278    	            break;
279    	        }
280    	
281    	        bIsCmdValid = TRUE;
282    	    } while(0);
283    	
284    	    return (bIsCmdValid | bRpReadUnsupportedCmd);
