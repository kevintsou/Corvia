23    	static U32 mctp_message_handle_mctp_control_set_eid(MCTP_MESSAGE_TRANSMIT_HANDLER *p_transmit_handler)
24    	{
25    	    MCTP_CONTROL_PTR p_mctp_control_req = (MCTP_CONTROL_PTR)p_transmit_handler->request_message_handler.buffer.p_base;
26    	    MCTP_CONTROL_PTR p_mctp_control_resp = (MCTP_CONTROL_PTR)p_transmit_handler->reponse_packet_handler.buffer.p_base;
(1) Event misra_c_2012_rule_11_3_violation: The object pointer expression "p_transmit_handler->reponse_packet_handler.buffer.p_base" of type "U8 *" is cast to type "MCTP_CONTROL_PTR".
27    	    U32 reponse_data_size = 0U;
28    	
29    	    Dump_SMBUS(tprintf(" [MCTP CONTROL] Set EID"));
30    	    Dump_SMBUS(tprintf("    >>>>> Receive Message Count : %d", p_transmit_handler->request_message_handler.byte_count));
31    	
32    	    switch (p_mctp_control_req->req_data.set_eid.operation)
33    	    {
34    	        // 0xFF, 0x00 = illegal.
35    	        // Endpoints are not allowed to be assigned the broadcast or null EIDs. It is
36    	        // recommended that the endpoint return an ERROR_INVALID_DATA
37    	        // completion code if it receives either of these values.
38    	        case MCTP_CTRL_SET_EID_OP_SET_EID :
39    	        case MCTP_CTRL_SET_EID_OP_FORCE_EID :
40    	            if ((MCTP_NULL_EID == p_mctp_control_req->req_data.set_eid.eid) || (MCTP_BROADCAST_EID == p_mctp_control_req->req_data.set_eid.eid)) {
41    	                Dump_SMBUS(tprintf("    >>>>> EID == NULL or EID == Broadcast EID"));
42    	                p_mctp_control_resp->resp_data.set_eid.cpl_code |= MCTP_CTRL_CPL_ERROR_INVALID_DATA;
43    	            }
44    	            else if ((MCTP_NULL_EID != p_transmit_handler->message_info.p_transport->endpoint.eid) && (MCTP_CTRL_SET_EID_OP_FORCE_EID != p_mctp_control_req->req_data.set_eid.operation)) {
45    	                Dump_SMBUS(tprintf("    >>>>> EID has already been assigned and the Set Endpoint ID command is issued from a different bus without forcing an EID assignment"));
46    	                Dump_SMBUS(tprintf("    >>>>> Already been assigned EID : %x", p_transmit_handler->message_info.p_transport->endpoint.eid));
47    	                p_mctp_control_resp->resp_data.set_eid.assign_status = MCTP_CTRL_SET_EID_STATUS_REJECT;
48    	            }
49    	            else {
50    	                Dump_SMBUS(tprintf("    >>>>> EID : %x", p_mctp_control_req->req_data.set_eid.eid));
51    	                p_mctp_control_resp->resp_data.set_eid.assign_status = MCTP_CTRL_SET_EID_STATUS_ACCEPT;
52    	                p_mctp_control_resp->resp_data.set_eid.eid_setting = p_mctp_control_req->req_data.set_eid.eid;
53    	                p_transmit_handler->message_info.p_transport->endpoint.eid = p_mctp_control_req->req_data.set_eid.eid;
54    	            }
55    	            break;
56    	
57    	        default :
