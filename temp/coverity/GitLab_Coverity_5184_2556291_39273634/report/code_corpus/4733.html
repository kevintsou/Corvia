39    	    U8 ret = SEC_YES;
40    	    SEC_HMAC_PARAMS_T sec_hmac_sha_param;
41    	    memset((void *)&sec_hmac_sha_param, 0x00, sizeof(SEC_HMAC_PARAMS_T));
42    	    sec_hmac_sha_param.key_addr.all         = (U64)key_addr;
43    	    sec_hmac_sha_param.key_len              = key_size;
44    	    sec_hmac_sha_param.sha.sha_mode         = sha_mode;
45    	    sec_hmac_sha_param.sha.src_addr.all     = (U64)src_addr;
46    	    sec_hmac_sha_param.sha.src_len          = src_size;
47    	    sec_hmac_sha_param.sha.targ_addr.all    = (U64)targ_addr;
48    	    sec_hmac_sha_param.sha.r1_enable        = FALSE;
49    	    sec_hmac_sha_param.sha.output_r1_enable = FALSE;
50    	
51    	    ret = phal_sec_hmac_sha_blocking(&sec_hmac_sha_param);
52    	    if (ret == SEC_NO) {
53    	        Dump_SEC(tprintf("HMAC F!"));
54    	    }
55    	    return ret;
