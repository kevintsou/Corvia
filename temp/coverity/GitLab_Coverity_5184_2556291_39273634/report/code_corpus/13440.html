495    	static void hal_nand_ctrl_api_write_data_dobule_byte(U8 ch, U8 *data, U16 size)
496    	{
497    	    U16 i;
498    	
499    	    if (dal_fip_get_io_type(ch) == DAL_FIP_IO_TYPE_LEGACY) { // sdr
500    	        for (i = 0U; i < size; i++) {
501    	            // datw
502    	            r16_w_FCH_PIO_DAT(ch, 0, (U16)data[i] | ((U16)data[i] << 8U));
503    	        }
504    	    }
505    	    else { // ddr
506    	        for (i = 0U; i < size; i++) {
507    	            // set dqs feature
508    	            if (i == 0U) {
509    	                //dal_fip_set_pio_dat_first(ch);
510    	                w_PIO_DAT_FIRST(ch, 1);
511    	                w_PIO_DAT_LAST(ch, 0);
512    	            }
513    	            else if (i == (size - 1U)) {
514    	                //dal_fip_set_pio_dat_last(ch);
515    	                w_PIO_DAT_FIRST(ch, 0);
516    	                w_PIO_DAT_LAST(ch, 1);
517    	            }
518    	            else {
519    	                //dal_fip_set_pio_dat_midden(ch);
520    	                w_PIO_DAT_FIRST(ch, 0);
521    	                w_PIO_DAT_LAST(ch, 0);
522    	            }
523    	
524    	            // datw
525    	            r16_w_FCH_PIO_DAT(ch, 0, (U16)data[i] | ((U16)data[i] << 8U));
526    	        }
527    	
528    	        // restore dqs feature
529    	        //dal_fip_set_pio_dat_default(ch);
530    	        w_PIO_DAT_FIRST(ch, 1);
531    	        w_PIO_DAT_LAST(ch, 1);
532    	    }
533    	}

535    	U32 hal_nand_ctrl_api_read_data_skip_highbyte(U8 ch, U8 *data, U16 size, U32 *ac_timing)
536    	{
537    	    U16 i;
538    	    U32 status;
539    	
540    	    status = 0U;
541    	
542    	    if (dal_fip_get_io_type(ch) == DAL_FIP_IO_TYPE_LEGACY) { // sdr
543    	        for (i = 0U; i < size; i++) {
544    	            // datr
545    	            data[i] = (U8)r32_r_FCH_PIO_DAT(ch);
546    	        }
547    	    }
548    	    else { // ddr
549    	        if ((U8)r_SCA_PROT(ch) == 0U) {
550    	            for (i = 0U; i < size; i++) {
551    	                // set dqs feature
552    	                if (i == 0U) {
553    	                    //dal_fip_set_pio_dat_first(ch);
554    	                    w_PIO_DAT_FIRST(ch, 1);
555    	                    w_PIO_DAT_LAST(ch, 0);
556    	                }
557    	                else if (i == (size - 1U)) {
558    	                    //dal_fip_set_pio_dat_last(ch);
559    	                    w_PIO_DAT_FIRST(ch, 0);
560    	                    w_PIO_DAT_LAST(ch, 1);
561    	                }
562    	                else {
563    	                    //dal_fip_set_pio_dat_midden(ch);
564    	                    w_PIO_DAT_FIRST(ch, 0);
565    	                    w_PIO_DAT_LAST(ch, 0);
566    	                }
567    	
568    	                // datr (trim high byte)
569    	                data[i] = (U8)r32_r_FCH_PIO_DAT(ch);
570    	            }
571    	
572    	            //Check dqs time out & dqs vaild
573    	            if ((((U8)r_PIOR_DQS_TIMEOUT_FLAG(ch)) == TRUE) || (((U8)r_PIO_DQS_VALID(ch)) == FALSE)) {
574    	                FIP_DBG_INFO_PRINTF("[PIO]ch = %d, flh tmo = 0x%x, dqs vld = 0x%x\r\n", ch, (U8)r_PIOR_DQS_TIMEOUT_FLAG(ch), (U8)r_PIO_DQS_VALID(ch));
575    	                status |= HAL_STATUS_PIO_ABORT_HANDLER;
576    	                //Clear time out flag
577    	                w_PIOR_DQS_TIMEOUT_FLAG(ch, 0);
578    	            }
579    	        }
580    	        else { // sca mode (single byte output)
581    	            for (i = 0U; i < size; i++) {
582    	                // datr (trim high byte)
583    	                data[i] = (U8)r32_r_FCH_PIO_DAT(ch);
584    	
585    	                // signle byte output terminate
586    	                //dal_fip_pio_ce_disable(ch, 0U);
587    	                r32_FLH_REG_TX(ch)[r32_FCH_FCE_ENB] &= (U32)(~(BIT(0)));
588    	                dal_fip_delay(ac_timing[HAL_DLY_TCEH2]);
589    	                //dal_fip_pio_ce_enable(ch, 0U);
590    	                r32_FLH_REG_TX(ch)[r32_FCH_FCE_ENB] |= (U32)(BIT(0));
591    	                w_DQ_BIT_REV_IDX(ch, 0);
592    	            }
593    	
594    	            //Check dqs time out & dqs vaild
595    	            if (((U8)r_PIOR_ALE_TIMEOUT_FLAG(ch)) == TRUE) {
596    	                FIP_DBG_INFO_PRINTF("[PIO]ch = %d, flh tmo = 0x%x\r\n", ch, (U8)r_PIOR_ALE_TIMEOUT_FLAG(ch));
597    	                status |= HAL_STATUS_PIO_ABORT_HANDLER;
598    	                //Clear time out flag
599    	                w_PIOR_ALE_TIMEOUT_FLAG(ch, 0);
600    	            }
601    	        }
602    	
603    	        // restore dqs feature
604    	        //dal_fip_set_pio_dat_default(ch);
605    	        w_PIO_DAT_FIRST(ch, 1);
606    	        w_PIO_DAT_LAST(ch, 1);
607    	    }
608    	    return status;
609    	}

611    	U32 hal_fip_api_set_feature(U8 ch, U8 ce, U8 cmd, U8 feat_adr, U8 *dat, U8 cmd_dqs_h, U8 odt_terminated, U32 *ac_timing, U32 chk_rdy)
612    	{
613    	    U32 status;
614    	    U32 tFEAT;
615    	    U8 is_sca = (U8)r_SCA_PROT(ch);
616    	
617    	    tFEAT = ((feat_adr < 0x80U) ? (ac_timing[HAL_DLY_TFEAT_1]) : (ac_timing[HAL_DLY_TFEAT_2]));
618    	    tFEAT = (ac_timing[HAL_DLY_TWB] + tFEAT);
619    	
620    	    status      = 0U;
621    	    // set feature
622    	    if (is_sca == 0U) {
623    	        w_PIO_DQS_OE(ch, 1);
624    	        w_PIO_DQS(ch, cmd_dqs_h);
625    	    }
626    	    else {
627    	        //set fip fetch pio data from ca bus
628    	        w_PIO_DATA_MODE(ch, 0);
629    	    }
630    	
631    	    //dal_fip_pio_ce_enable(ch, ce);
632    	    r32_FLH_REG_TX(ch)[r32_FCH_FCE_ENB] |= (U32)(BIT(ce));
633    	    w_DQ_BIT_REV_IDX(ch, ce);
634    	    //tCS
635    	    dal_fip_delay(ac_timing[HAL_DLY_TCS]);
636    	
637    	    if (odt_terminated == 1U) {
638    	        // set controller odt disable
639    	        //dal_fip_set_odt_en(ch, 0U, 0U);
640    	        w_FORCE_ODT_EN(ch, 0U);
641    	        w_DYNAMIC_ODT_EN(ch, 0U);
642    	
643    	        //tCSVOL
644    	        dal_fip_delay(ac_timing[HAL_DLY_TCSVOL]);
645    	        // set nand odt disable
646    	        w_PIO_CMD(ch, HAL_NAND_CMD_ODT_DISABLE);
647    	        //tODT on/off
648    	        dal_fip_delay(ac_timing[HAL_DLY_TODT_ON_OFF]);
649    	    }
650    	
651    	    w_PIO_CMD(ch, cmd);
652    	    w_PIO_ADR(ch, feat_adr);
653    	    //tADL
654    	    dal_fip_delay(ac_timing[HAL_DLY_TADL]);
655    	    if (is_sca == 0U) {
656    	        //dal_fip_pio_dqs(ch, 0U);
657    	        w_PIO_DQS_OE(ch, 1);
658    	        w_PIO_DQS(ch, 0U);
659    	    }
660    	
661    	    hal_nand_ctrl_api_write_data_dobule_byte(ch, dat, 4U);
662    	
663    	    if (odt_terminated == 1U) {
664    	        // restore nand odt
665    	        w_PIO_CMD(ch, HAL_NAND_CMD_ODT_ENABLE);
666    	        //tODT on/off
667    	        dal_fip_delay(ac_timing[HAL_DLY_TODT_ON_OFF]);
668    	
669    	        // restore controller odt
670    	        //dal_fip_set_odt_en(ch, 1U, 1U);
671    	        w_FORCE_ODT_EN(ch, 1U);
672    	        w_DYNAMIC_ODT_EN(ch, 1U);
673    	    }
674    	
675    	    // wait nand busy
676    	    dal_fip_delay(tFEAT);
677    	
678    	    //Check R/B pin before read
679    	    U32 mabiao_addr = MABIAO_SET_MS(MABIAO_RB_PIN_CHK, chk_rdy);
680    	    do {
681    	        if (MABIAO_CHK(mabiao_addr)) {
682    	            status |= HAL_STATUS_PIO_POLL_RDY_PIN_TIMEOUT;
683    	            break;
684    	        }
685    	    } while (hal_fip_chk_rby_sts(ch) == FALSE);
686    	
687    	    //tCH
688    	    dal_fip_delay(ac_timing[HAL_DLY_TCH]);
689    	    //dal_fip_pio_ce_disable(ch, ce); // micron need exit data input by ce high when set feature to feature addres 02h
690    	    r32_FLH_REG_TX(ch)[r32_FCH_FCE_ENB] &= (U32)(~(BIT(ce)));
691    	
692    	    //tprintf("ch = %d, ce = %d, set feature (%xh) - %x %x %x %x", ch, ce, feat_adr, dat[0], dat[1], dat[2], dat[3]);
693    	    //tprintf("hal_fip_api_set_feature status = %x", status);
694    	    return  status;
695    	}

975    	    //dal_fip_pio_ce_disable(ch, ce);
976    	    r32_FLH_REG_TX(ch)[r32_FCH_FCE_ENB] &= (U32)(~(BIT(ce)));
977

981    	static U32 hal_nand_api_read_data(U8 ch, U8 *data, U16 size)
982    	{
983    	    U16 i;
984    	    U16 pio_cycle;
985    	    U32 status;
986    	
987    	    status = 0U;
988    	
989    	    if (dal_fip_get_io_type(ch) == DAL_FIP_IO_TYPE_LEGACY) { // sdr
990    	        pio_cycle = size;
991    	        for (i = 0U; i < pio_cycle; i++) {
992    	            // datr
993    	            data[i] = (U8)r32_r_FCH_PIO_DAT(ch);
994    	        }
995    	    }
996    	    else { // ddr
997    	        pio_cycle = (size >> 1U);
998    	        U16 *p16_data = (U16 *)((void *)data);
999    	        if ((U8)r_SCA_PROT(ch) == 0U) {
1000    	            for (i = 0U; i < pio_cycle; i++) {
1001    	                // set dqs feature
1002    	                if (i == 0U) {
1003    	                    //dal_fip_set_pio_dat_first(ch);
1004    	                    w_PIO_DAT_FIRST(ch, 1);
1005    	                    w_PIO_DAT_LAST(ch, 0);
1006    	                }
1007    	                else if (i == (pio_cycle - 1U)) {
1008    	                    //dal_fip_set_pio_dat_last(ch);
1009    	                    w_PIO_DAT_FIRST(ch, 0);
1010    	                    w_PIO_DAT_LAST(ch, 1);
1011    	                }
1012    	                else {
1013    	                    //dal_fip_set_pio_dat_midden(ch);
1014    	                    w_PIO_DAT_FIRST(ch, 0);
1015    	                    w_PIO_DAT_LAST(ch, 0);
1016    	                }
1017    	
1018    	                // datr
1019    	                p16_data[i] = (U16)r32_r_FCH_PIO_DAT(ch);
1020    	            }
1021    	
1022    	            //Check dqs time out & dqs vaild
1023    	            if ((((U8)r_PIOR_DQS_TIMEOUT_FLAG(ch)) == TRUE) || (((U8)r_PIO_DQS_VALID(ch)) == FALSE)) {
1024    	                FIP_DBG_INFO_PRINTF("[PIO]ch = %d, flh tmo = 0x%x, dqs vld = 0x%x\r\n", ch, (U8)r_PIOR_DQS_TIMEOUT_FLAG(ch), (U8)r_PIO_DQS_VALID(ch));
1025    	                status |= HAL_STATUS_PIO_ABORT_HANDLER;
1026    	                //Clear time out flag
1027    	                w_PIOR_DQS_TIMEOUT_FLAG(ch, 0);
1028    	            }
1029    	        }
1030    	        else { // sca mode (single byte output)
1031    	            //set fip fetch pio data from ca bus
1032    	            w_PIO_DATA_MODE(ch, 0);
1033    	
1034    	            // set SCE pkt header and body
1035    	            //hal_flash_pio_pkt_header(ch, 0x1, 0x7);
1036    	            w_FCTL_PKT_DATA_HEADER(ch, 0x7);
1037    	            w_FCTL_PKT_LEN_HEADER(ch, 0x1);
1038    	
1039    	            //hal_flash_pio_pkt_body(ch, 0x2, 0x0);
1040    	            w_FCTL_PKT_DATA_BODY(ch, 0x0);
1041    	            w_FCTL_PKT_LEN_BODY(ch, 0x2);
1042    	
1043    	            //hal_flash_pio_pkt(ch, 0x1);
1044    	            w_PIO_PKT(ch, 0x1);
1045    	
1046    	            //set fip fetch pio data from dq bus
1047    	            w_PIO_DATA_MODE(ch, 1);
1048    	
1049    	            for (i = 0U; i < pio_cycle; i++) {
1050    	                // set dqs feature
1051    	                if (i == 0U) {
1052    	                    //dal_fip_set_pio_dat_first(ch);
1053    	                    w_PIO_DAT_FIRST(ch, 1);
1054    	                    w_PIO_DAT_LAST(ch, 0);
1055    	                }
1056    	                else if (i == (pio_cycle - 1U)) {
1057    	                    //dal_fip_set_pio_dat_last(ch);
1058    	                    w_PIO_DAT_FIRST(ch, 0);
1059    	                    w_PIO_DAT_LAST(ch, 1);
1060    	                }
1061    	                else {
1062    	                    //dal_fip_set_pio_dat_midden(ch);
1063    	                    w_PIO_DAT_FIRST(ch, 0);
1064    	                    w_PIO_DAT_LAST(ch, 0);
1065    	                }
1066    	
1067    	                // datr
1068    	                p16_data[i] = (U16)r32_r_FCH_PIO_DAT(ch);
1069    	            }
1070    	
1071    	            //set fip fetch pio data from ca bus
1072    	            w_PIO_DATA_MODE(ch, 0);
1073    	            // set SCT pkt header and body
1074    	            //hal_flash_pio_pkt_header(ch, 0x1, 0xF);
1075    	            w_FCTL_PKT_DATA_HEADER(ch, 0xF);
1076    	            w_FCTL_PKT_LEN_HEADER(ch, 0x1);
1077    	
1078    	            //hal_flash_pio_pkt_body(ch, 0x2, 0x0);
1079    	            w_FCTL_PKT_DATA_BODY(ch, 0x0);
1080    	            w_FCTL_PKT_LEN_BODY(ch, 0x2);
1081    	
1082    	            //hal_flash_pio_pkt(ch, 0x1);
1083    	            w_PIO_PKT(ch, 0x1);
1084    	
1085    	            // Pull DQS_OE and RE high
1086    	            w_GPO_DQS_OE(ch, 1);
1087    	            w_GPO_RE(ch, 1);
1088    	            w_GPO_WE(ch, 0);
1089    	
1090    	            //Check dqs time out & dqs vaild
1091    	            if (((U8)r_PIOR_ALE_TIMEOUT_FLAG(ch)) == TRUE) {
1092    	                FIP_DBG_INFO_PRINTF("[PIO]ch = %d, flh tmo = 0x%x\r\n", ch, (U8)r_PIOR_ALE_TIMEOUT_FLAG(ch));
1093    	                status |= HAL_STATUS_PIO_ABORT_HANDLER;
1094    	                //Clear time out flag
1095    	                w_PIOR_ALE_TIMEOUT_FLAG(ch, 0);
1096    	            }
1097    	        }
1098    	
1099    	        // restore dqs feature
1100    	        //dal_fip_set_pio_dat_default(ch);
1101    	        w_PIO_DAT_FIRST(ch, 1);
1102    	        w_PIO_DAT_LAST(ch, 1);
1103    	    }
1104    	    return status;
1105    	}
