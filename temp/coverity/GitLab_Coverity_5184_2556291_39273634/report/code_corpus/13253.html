393    	static void host_nud_dma(U32 buf_addr, U32 spr_addr, U16 ctag, U32 byte_cnt, U8 op)
394    	{
395    	    HAL_APU_FW_BCMD_SQ_HW apu_sq;
396    	    HAL_APU_TRIG_DMA_PARAM fw_buf_ready;
397    	
398    	    U32 cur_buf_addr  = buf_addr;
399    	    U32 cur_4k_cnt    = 0;
400    	    U32 total_4k_cnt  = get_4k_cnt(byte_cnt);
401    	    U32 remain_4K_cnt = total_4k_cnt;
402    	
403    	    hal_cmgr_set_ct_rlbn(ctag, total_4k_cnt);
404    	
405    	    U32 mabiao_addr = MABIAO_SET_MS(MABIAO_DMA_RDY, 200);
406    	
407    	    while ( remain_4K_cnt ) {
408    	        memset((void *)&fw_buf_ready, 0x0U, sizeof(HAL_APU_TRIG_DMA_PARAM));
409    	
410    	        fw_buf_ready.ctag              = ctag;
411    	        fw_buf_ready.ud                = 0U;
412    	        fw_buf_ready.fst               = (cur_4k_cnt == 0U) ? TRUE : FALSE;                // first 4k
413    	        fw_buf_ready.lst               = (cur_4k_cnt == total_4k_cnt - 1) ? TRUE : FALSE;  // last 4K
414    	        fw_buf_ready.dir               = (op == NUD_DMA_WRITE) ? NUD_DMA_WRITE : NUD_DMA_READ;
415    	        fw_buf_ready.node_tag          = 0U;
416    	        fw_buf_ready.buf_addr_l        = (U32)(cur_buf_addr);
417    	        fw_buf_ready.buf_addr_h        = 0;
418    	        fw_buf_ready.spr_addr_l        = spr_addr;
419    	        fw_buf_ready.spr_addr_h        = 0;
420    	        fw_buf_ready.fw_bcmd.tbc       = byte_cnt;
421    	        fw_buf_ready.fw_bcmd.offset    = 0U;
422    	        fw_buf_ready.fw_bcmd.dma_done  = 0U;
423    	        fw_buf_ready.fw_bcmd.err_info  = 0U;
424    	        fw_buf_ready.fw_bcmd.acpl      = 0U;
425    	
426    	        hal_apu_fill_fw_bcmd_sq(&apu_sq, &fw_buf_ready);
427    	        Dump_NVME(tprintf("NUD DMA %x %x %x %x %x",apu_sq.header, apu_sq.all[0], apu_sq.all[1], apu_sq.all[2], apu_sq.all[3]));
428    	
429    	        WAIT_UNTIL_TIMEOUT((hal_db_api_get_full(HOST_MR_QID_CPU0_SQ_NVME_NUD_DMA_P0) == TRUE), NVME_CMD_TIMEOUT_THRESHOLD, ERR_DB_FULL_TIMEOUT, MABIAO_DB_WAIT_SQ);
430    	        WAIT_UNTIL_TIMEOUT((hal_db_push(&apu_sq, sizeof(HAL_APU_FW_BCMD_SQ_HW), HOST_MR_QID_CPU0_SQ_NVME_NUD_DMA_P0) == FAIL), NVME_CMD_TIMEOUT_THRESHOLD, ERR_DB_FULL_TIMEOUT, MABIAO_DB_WAIT_SQ);
431    	
432    	        cur_buf_addr = cur_buf_addr + LC_BYTE_SIZE;
433    	        --remain_4K_cnt;
434    	        ++cur_4k_cnt;
435    	
436    	        if (MABIAO_CHK(mabiao_addr)) {
437    	            break;
438    	        }
439    	    }
440    	}
