501    	    hal_sec_sm9_sm3_update(wbuf, HAL_SEC_SM9_Z256_WBUF_LEN, &sm9_kem_buf->comm_buf);
502    	    hal_sec_sm9_sm3_update((U8 *)id, id_len, &sm9_kem_buf->comm_buf);
503    	    hal_sec_sm9_sm3_kdf_finish(key, key_len, &sm9_kem_buf->comm_buf);
504    	
505    	    if (hal_sec_sm9_mem_is_zero(key, key_len) == 1U)
506    	    {
507    	        return SEC_NO;
508    	    }
509    	
510    	    (void)memset(&r, 0, sizeof(r));
511    	    (void)memset(&w, 0, sizeof(w));
512    	    (void)memset(wbuf, 0, HAL_SEC_SM9_Z256_WBUF_LEN);
513    	
514    	    // A7: output (K, C)
515    	    return SEC_YES;
516    	}

526    	    // B1: check C in G1

539    	
540    	    if (hal_sec_sm9_mem_is_zero(key, key_len) == 1U)
541    	    {
542    	        return SEC_NO;
543    	    }
544    	
545    	    (void)memset(&w, 0, sizeof(w));
546    	    (void)memset(wbuf, 0, HAL_SEC_SM9_Z256_WBUF_LEN);
547    	
548    	    // B4: output K
549    	    return SEC_YES;

591    	U8 hal_sec_sm9_exch_a1(HAL_SEC_SM9_KEY_EXCH_PAC_T_PTR exch_t, U8 *nonce, HAL_SEC_SM9_KEYEXCH_BUF_T_PTR keyexch_buf)
592    	{
593    	    HAL_SEC_SM9_Z256_T temp = {0ULL};
594    	    HAL_SEC_SM9_Z256_POINT_T ra_point;
595    	    SM9_EXCH_MASTER_KEY emk_t;
596    	
597    	    // retrieve key
598    	    if (hal_sec_sm9_z256_point_from_raw(&emk_t.Ppube, exch_t->mpk) != SEC_YES)
599    	    {
600    	        return SEC_NO;
601    	    }
602    	
603    	    // A1 : Q = H1(ID_B||hid, N) * P1 + Ppube
604    	    HAL_SEC_SM9_Z256_POINT_T sm9_z256_mont_p1;
605    	    memcpy((void *)&sm9_z256_mont_p1, (void *)&SM9_Z256_MONT_P1, sizeof(SM9_Z256_MONT_P1));
606    	
607    	    hal_sec_sm9_z256_hash1(temp, exch_t->identityB, exch_t->iblen, HAL_SEC_SM9_EXCH_HID, &keyexch_buf->comm_buf);
608    	    hal_sec_sm9_z256_point_mul(&ra_point, temp, &sm9_z256_mont_p1, &keyexch_buf->comm_buf);
609    	    hal_sec_sm9_z256_point_add(&ra_point, &ra_point, &emk_t.Ppube);
610    	
611    	    //A2 : rand_val rA in [1, N-1]
612    	    hal_sec_sm9_z256_from_bytes(temp, nonce);
613    	
614    	    //RA = rA * Q
615    	    hal_sec_sm9_z256_point_mul(&ra_point, temp, &ra_point, &keyexch_buf->comm_buf);
616    	    hal_sec_sm9_z256_point_to_bytes(&ra_point, exch_t->Ra);
617    	
618    	    (void)memcpy(&exch_t->ra, temp, sizeof(HAL_SEC_SM9_Z256_T));
619    	
620    	    return SEC_YES;
621    	}

623    	U8 hal_sec_sm9_exch_b1(HAL_SEC_SM9_KEY_EXCH_PAC_T_PTR exch_t, U8 out[], U32 out_len, U8 *msg, HAL_SEC_SM9_KEYEXCH_BUF_T_PTR keyexch_buf)
624    	{
625    	    HAL_SEC_SM9_Z256_T temp = {0ULL};
626    	
627    	    U8 ta[65], tb[65];
628    	    HAL_SEC_SM9_Z256_POINT_T rb_point, ra_point;
629    	    SM9_EXCH_MASTER_KEY emk;
630    	    HAL_SEC_SM9_EXCH_KEY_T ek_t;
631    	    HAL_SEC_SM9_Z256_T rb;
632    	
633    	    HAL_SEC_SM9_Z256_FP12_T *G1 = (HAL_SEC_SM9_Z256_FP12_T *)(void *)(U32 *)(keyexch_buf->keyexch_G1_addr);
634    	    HAL_SEC_SM9_Z256_FP12_T *G2 = (HAL_SEC_SM9_Z256_FP12_T *)(void *)(U32 *)(keyexch_buf->keyexch_G2_addr);
635    	    HAL_SEC_SM9_Z256_FP12_T *G3 = (HAL_SEC_SM9_Z256_FP12_T *)(void *)(U32 *)(keyexch_buf->keyexch_G3_addr);
636    	    U8 *g1 = (U8 *)(void *)(U32 *)(keyexch_buf->keyexch_g1_addr);
637    	    U8 *g2 = (U8 *)(void *)(U32 *)(keyexch_buf->keyexch_g2_addr);
638    	    U8 *g3 = (U8 *)(void *)(U32 *)(keyexch_buf->keyexch_g3_addr);
639    	
640    	    hal_sec_sm9_z256_from_bytes(rb, msg);
641    	
642    	    if (hal_sec_sm9_z256_point_from_raw(&ra_point, exch_t->Ra) != SEC_YES)
643    	    {
644    	        return SEC_NO;
645    	    }
646    	
647    	    // retrieve key
648    	    if (hal_sec_sm9_z256_point_from_raw(&emk.Ppube, exch_t->mpk) != SEC_YES)
649    	    {
650    	        return SEC_NO;
651    	    }
652    	
653    	    if (hal_sec_sm9_z256_point_from_raw(&ek_t.Ppube, exch_t->mpk) != SEC_YES)
654    	    {
655    	        return SEC_NO;
656    	    }
657    	
658    	    if (hal_sec_sm9_z256_twist_point_from_raw(&ek_t.de, exch_t->sk, &keyexch_buf->comm_buf) != SEC_YES)
659    	    {
660    	        return SEC_NO;
661    	    }
662    	
663    	    // B1: Q = H1(ID_A||hid,N) * P1 + Ppube
664    	    HAL_SEC_SM9_Z256_POINT_T sm9_z256_mont_p1;
665    	    memcpy((void *)&sm9_z256_mont_p1, (void *)&SM9_Z256_MONT_P1, sizeof(SM9_Z256_MONT_P1));
666    	
667    	    hal_sec_sm9_z256_hash1(temp, exch_t->identityA, exch_t->ialen, HAL_SEC_SM9_EXCH_HID, &keyexch_buf->comm_buf);
668    	    hal_sec_sm9_z256_point_mul(&rb_point, temp, &sm9_z256_mont_p1, &keyexch_buf->comm_buf);
669    	    hal_sec_sm9_z256_point_add(&rb_point, &rb_point, &emk.Ppube);
670    	
671    	    // B3: RB = rB * Q
672    	    hal_sec_sm9_z256_point_mul(&rb_point, rb, &rb_point, &keyexch_buf->comm_buf);
673    	
674    	    // B4: check RA on curve; G1 = e(RA, deB), G2 = e(Ppube, P2) ^ rB, G3 = G1 ^ rB
675    	    if (hal_sec_sm9_z256_point_is_on_curve(&ra_point) == SEC_NO)
676    	    {
677    	        return SEC_NO;
678    	    }
679    	
680    	    HAL_SEC_SM9_Z256_TWIST_POINT_T sm9_z256_mont_p2;
681    	    memcpy((void *)&sm9_z256_mont_p2, (void *)&SM9_Z256_MONT_P2, sizeof(SM9_Z256_MONT_P2));
682    	
683    	    hal_sec_sm9_z256_pairing(*G1, &ek_t.de, &ra_point, &keyexch_buf->comm_buf);
684    	    hal_sec_sm9_z256_pairing(*G2, &sm9_z256_mont_p2, &emk.Ppube, &keyexch_buf->comm_buf);
685    	    hal_sec_sm9_z256_fp12_pow(*G2, *G2, rb, &keyexch_buf->comm_buf);
686    	    hal_sec_sm9_z256_fp12_pow(*G3, *G1, rb, &keyexch_buf->comm_buf);
687    	
688    	    hal_sec_sm9_z256_point_to_uncompressed_octets(&ra_point, ta);
689    	    hal_sec_sm9_z256_point_to_uncompressed_octets(&rb_point, tb);
690    	    hal_sec_sm9_z256_fp12_to_bytes(*G1, g1);
691    	    hal_sec_sm9_z256_fp12_to_bytes(*G2, g2);
692    	    hal_sec_sm9_z256_fp12_to_bytes(*G3, g3);
693    	
694    	    // B5: sk = KDF(ID_A || ID_B || RA || RB || g1 || g2 || g3, klen)
695    	    hal_sec_sm9_sm3_init(&keyexch_buf->comm_buf);
696    	    hal_sec_sm9_sm3_update(exch_t->identityA, exch_t->ialen, &keyexch_buf->comm_buf);
697    	    hal_sec_sm9_sm3_update(exch_t->identityB, exch_t->iblen, &keyexch_buf->comm_buf);
698    	    hal_sec_sm9_sm3_update(ta + 1, 64, &keyexch_buf->comm_buf);
699    	    hal_sec_sm9_sm3_update(tb + 1, 64, &keyexch_buf->comm_buf);
700    	    hal_sec_sm9_sm3_update(g1, 32 * 12, &keyexch_buf->comm_buf);
701    	    hal_sec_sm9_sm3_update(g2, 32 * 12, &keyexch_buf->comm_buf);
702    	    hal_sec_sm9_sm3_update(g3, 32 * 12, &keyexch_buf->comm_buf);
703    	    hal_sec_sm9_sm3_kdf_finish(out, out_len, &keyexch_buf->comm_buf);
704    	
705    	    if (hal_sec_sm9_mem_is_zero(out, out_len) == 1U)
706    	    {
707    	        return SEC_NO;
708    	    }
709    	
710    	    // B6: SB = Hash(0x82 || g1 || Hash(g2 || g3 || ID_A || ID_B || RA || RB)) [optional]
711    	
712    	    (void)memset(temp, 0, sizeof(temp));
713    	    (void)memset(G1, 0, sizeof(*G1));
714    	    (void)memset(G2, 0, sizeof(*G2));
715    	    (void)memset(G3, 0, sizeof(*G3));
716    	    (void)memset(g1, 0, 32 * 12);
717    	    (void)memset(g2, 0, 32 * 12);
718    	    (void)memset(g3, 0, 32 * 12);
719    	    (void)memset(ta, 0, sizeof(ta));
720    	    (void)memset(tb, 0, sizeof(tb));
721    	
722    	    // B7: Output RB
723    	    hal_sec_sm9_z256_point_to_bytes(&rb_point, exch_t->Rb);
724    	
725    	    return SEC_YES;
726    	}

911    	void hal_sec_sm9_sm3_kdf_finish(U8 *kbuf, U32 klen, HAL_SEC_SM9_COMM_BUF_T_PTR comm_buffer)
912    	{
913    	    U32 ct = 0x00000001UL;
914    	    U32 i, m, n;
915    	
916    	    SEC_SHA_PARAMS_T sm3_param;
917    	    U32 kbuf_tmp = (U32)kbuf;
918    	
919    	    m = klen / 32UL;
920    	    n = klen % 32UL;
921    	
922    	    if (n != 0UL)
923    	    {
924    	        m++;
925    	    }
926    	
927    	    for (i = 0UL; i < m; i++)
928    	    {
929    	        kbuf = (U8 *)(U32 *)(kbuf_tmp) + (i * 32U);
930    	
931    	        ((U8 *)(comm_buffer->hash_src_addr))[comm_buffer->hash_buf_len]       = (U8)((ct >> 24U) & 0xFFU);
932    	        ((U8 *)(comm_buffer->hash_src_addr))[comm_buffer->hash_buf_len + 1UL] = (U8)((ct >> 16U) & 0xFFU);
933    	        ((U8 *)(comm_buffer->hash_src_addr))[comm_buffer->hash_buf_len + 2UL] = (U8)((ct >> 8U)  & 0xFFU);
934    	        ((U8 *)(comm_buffer->hash_src_addr))[comm_buffer->hash_buf_len + 3UL] = (U8)((ct)        & 0xFFU);
935    	
936    	        (void)memset((void *)&sm3_param, 0, sizeof(SEC_SHA_PARAMS_T));
937    	        sm3_param.src_addr.low  = comm_buffer->hash_src_addr;
938    	        sm3_param.src_len       = comm_buffer->hash_buf_len + 4UL;
939    	        sm3_param.targ_addr.low = comm_buffer->hash_targ_addr;
940    	        sm3_param.sha_mode      = HAL_SEC_SHA2_SM3_MODE;
941    	
942    	        (void)phal_sec_sha_blocking(&sm3_param);
943    	
944    	        if ((i == (m - 1UL)) && (n != 0UL))
945    	        {
946    	            (void)memcpy((void *)(kbuf), (void *)(U32 *)(comm_buffer->hash_targ_addr), n);
947    	        }
948    	        else
949    	        {
950    	            (void)memcpy((void *)(kbuf), (void *)(U32 *)(comm_buffer->hash_targ_addr), HAL_SEC_SHA2_SM3_DIGEST_LEN);
951    	        }
952    	
953    	        ct ++;
954    	    }
