78    	{
79    	    if (unlikely(port_id >= cfg->port_cnt) || unlikely(pf_id > cfg->port[port_id].pf_cnt))
80    	    {
81    	        return FALSE;
82    	    }
83    	
84    	    // check bar setting
85    	    if (bar->mem_type & 0x1)

86    	    {
87    	        return FALSE;
88    	    }
89    	    if ((bar->bar > 5) || ((bar->mem_type == 0x2) && (bar->bar == 5)))
90    	    {
91    	        // no available bar
92    	        return FALSE;

102    	    dal_set_bar_type(port_id, pf_id, (U32)bar->bar, (U32)bar->mem_type);
103    	    dal_set_bar_prefetch(port_id, pf_id, (U32)bar->bar, (U32)bar->mem_prefetch);
