152    	            sal_reverse_copy(	(void *)pInputKey->RsaPublicKey.ubN,		    (void *)info.pubKeyN, 	VUA_RSA_BYTE_CNT);

157    	        }

162    	        // todo check SHA_CAL_BUF_SIZE

171    	        // Dump_SEC(tprintf("Key Data"));
172    	        // Dump_SEC(tprintm((U32)pInputKey, ulKeySize));

174    	        Dump_SEC(tprint8h((U8 *)gVUALock.ulHashCalBufBase, VUA_KEY_HASH_SIZE));
175    	#endif
176    	
177    	        // use gVUALock.ulHashCalBufBase to verify key hash
178    	        if (!MyAssert(SECResult_PASS == VUA_verify_key_hash(gVUALock.ulHashCalBufBase))) {
179    	            // Compare eFuse Key Hash Fail
180    	            ulStatus   = INVALID_STATUS_VERIFICATION_ERROR;
181    	            ulFailCode = STATUS_VERIFY_FAIL;
182    	            Dump_SEC(tprintf("VK F!"));
183    	            break;
184    	        }
185    	
186    	        memcpy((void *)&gVUALock.VUAPublicKey, (void *)pInputKey, sizeof(PublicKey_t));
187    	        memcpy(&gVUALock.aubN_v, pInput->ucN_v, 64);

190    	        ulFailCode = STATUS_DEFAULT;
191    	        gVUALock.ubVUCLockState = VUA_VERIFY_STATE_ENGINEERING;
192    	
193    	        Dump_SEC(tprintf("send key ok!"));
194    	        return result;
195    	    } while (0);
196    	
197    	    VUA_FailHandle(&result, ulStatus, ulFailCode);
198    	    Dump_SEC(tprintf("send key F!"));
199    	
200    	    return result;
201    	}
