911    	void hal_sec_sm9_sm3_kdf_finish(U8 *kbuf, U32 klen, HAL_SEC_SM9_COMM_BUF_T_PTR comm_buffer)
912    	{
913    	    U32 ct = 0x00000001UL;
914    	    U32 i, m, n;
915    	
916    	    SEC_SHA_PARAMS_T sm3_param;
917    	    U32 kbuf_tmp = (U32)kbuf;
918    	
919    	    m = klen / 32UL;
920    	    n = klen % 32UL;
921    	
922    	    if (n != 0UL)
923    	    {
924    	        m++;
925    	    }
926    	
927    	    for (i = 0UL; i < m; i++)
928    	    {
929    	        kbuf = (U8 *)(U32 *)(kbuf_tmp) + (i * 32U);
930    	
931    	        ((U8 *)(comm_buffer->hash_src_addr))[comm_buffer->hash_buf_len]       = (U8)((ct >> 24U) & 0xFFU);
932    	        ((U8 *)(comm_buffer->hash_src_addr))[comm_buffer->hash_buf_len + 1UL] = (U8)((ct >> 16U) & 0xFFU);
933    	        ((U8 *)(comm_buffer->hash_src_addr))[comm_buffer->hash_buf_len + 2UL] = (U8)((ct >> 8U)  & 0xFFU);
934    	        ((U8 *)(comm_buffer->hash_src_addr))[comm_buffer->hash_buf_len + 3UL] = (U8)((ct)        & 0xFFU);
935    	
936    	        (void)memset((void *)&sm3_param, 0, sizeof(SEC_SHA_PARAMS_T));
937    	        sm3_param.src_addr.low  = comm_buffer->hash_src_addr;
938    	        sm3_param.src_len       = comm_buffer->hash_buf_len + 4UL;
939    	        sm3_param.targ_addr.low = comm_buffer->hash_targ_addr;
940    	        sm3_param.sha_mode      = HAL_SEC_SHA2_SM3_MODE;
941    	
942    	        (void)phal_sec_sha_blocking(&sm3_param);
943    	
944    	        if ((i == (m - 1UL)) && (n != 0UL))
945    	        {
946    	            (void)memcpy((void *)(kbuf), (void *)(U32 *)(comm_buffer->hash_targ_addr), n);
947    	        }
948    	        else
949    	        {
950    	            (void)memcpy((void *)(kbuf), (void *)(U32 *)(comm_buffer->hash_targ_addr), HAL_SEC_SHA2_SM3_DIGEST_LEN);
951    	        }
952    	
953    	        ct ++;
954    	    }
