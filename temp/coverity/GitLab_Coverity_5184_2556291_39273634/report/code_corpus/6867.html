101    	void nvme_send_cpl_normal(HOST_API_STRUCT_PTR host)
102    	{
103    	    //    HAL_APU_CPL_SQ_PARAM_PTR hcpl = (HAL_APU_CPL_SQ_PARAM_PTR)&host->hcpl;
104    	    HAL_APU_CPL_SQ_PARAM_PTR hcpl = (HAL_APU_CPL_SQ_PARAM_PTR)&host->hcpl;
105    	    U16                      ctag = hcpl->ctag;
106    	
107    	    if (dal_cmgr_get_ct_vld(ctag) == FALSE) {
108    	        BC_APP_ASSERT(BCODE_ERROR_NVME_SEND_CPL_FLAG_VALID, 0);
109    	        return;
110    	    }
111    	
112    	    // if AER cmd: release cmd table resource
113    	    if (dal_cmgr_get_ct_adm(ctag) &&                       // (host->cmd_tbl_ptr[ctag].adm == 1) &&
114    	        (dal_cmgr_get_ct_opc(ctag) == ADM_OPC_AER) &&      // (host->cmd_tbl_ptr[ctag].sq_cmd.opc == ADM_OPC_AER) &&
(1) Event misra_c_2012_rule_10_4_violation: Essential type of the left hand operand "dal_cmgr_get_ct_opc(ctag)" (unsigned) is not the same as that of the right operand "12"(signed).
115    	        (nvme_check_cpl_err_status(host)) == FALSE) {           // AER cmd is successfully finished
116    	        Dump_NVME(tprintf("AER cmd handle.\n\r"));
117    	
118    	        // if AER cmd completes successfully
119    	        dal_cmgr_set_fw_clr_cvld_ctag(PCIE_PORT_0, ctag);  // set the ctag we want to clear
120    	        dal_cmgr_set_fw_clr_cvld_req(PCIE_PORT_0);
121    	
122    	        U32 mabiao_addr = MABIAO_SET_MS(MABIAO_CTAG_CLR, 500);
123    	        while (dal_cmgr_get_fw_clr_cvld_req(PCIE_PORT_0)) {
124    	            if (MABIAO_CHK(mabiao_addr)) {
125    	                BC_APP_ASSERT( BCODE_ERROR_CLEAR_AER_CTAG_TIMEOUT, 0 );
126    	                return;
127    	            }
128    	        }
129    	    }
130    	    else {
131    	        Dump_NVME(tprintf("Send Completion Normal.\n\r"));
132    	
133    	        nvme_send_cpl(host);
134    	    }
135    	
136    	    // unlock CMGR for sync cmd
137    	    nvme_clr_cmgr_hw_lock();
138    	}
