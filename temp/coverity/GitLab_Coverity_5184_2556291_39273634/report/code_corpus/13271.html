476    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->sign.sign_local_var_s.SK, HAL_SEC_PQC_KAZ_DS_SK_LEN, sizeof(unsigned char), HAL_SEC_PQC_KAZ_BIG_ENDIAN, (U8 *)(sk));
477    	
478    	    //2) Compute HASHValue(m)
479    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->sign.sign_local_var_s.hashValue, HAL_SEC_PQC_KAZ_SYS_LEN, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, digest);
480    	
481    	    lenG1qQ = hal_sec_pqc_kaz_mp_size_in_base(kaz_buf_ptr->sign.sign_local_var_s.G1qQ, 2UL);
482    	
483    	    //3) Generate random prime r, random OMEGA2
484    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->sign.sign_local_var_s.r1, HAL_SEC_PQC_KAZ_DS_SP_RAN, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, &kaz_buf_ptr->sign.sign_buf_s.sign_rand_val[rand_val_offset]);
485    	    rand_val_offset += (U8)HAL_SEC_PQC_KAZ_DS_SP_RAN;
486    	
487    	    hal_sec_pqc_kaz_pke_find_next_prime(kaz_buf_ptr->sign.sign_local_var_s.r1, kaz_buf_ptr->sign.sign_local_var_s.r1, &kaz_buf_ptr->sign.sign_buf_s.next_prime_buf, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
488    	
489    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->sign.sign_local_var_s.OMEGA2, HAL_SEC_PQC_KAZ_DS_SP_RAN, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, &kaz_buf_ptr->sign.sign_buf_s.sign_rand_val[rand_val_offset]);
490    	    rand_val_offset += (U8)HAL_SEC_PQC_KAZ_DS_SP_RAN;
491    	
492    	    //4) Compute BETA
493    	    (void)hal_sec_pqc_kaz_pke_mpy64(kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.OMEGA2, kaz_buf_ptr->sign.sign_local_var_s.phiG1);
494    	    hal_sec_pqc_kaz_pke_powm(kaz_buf_ptr->sign.sign_local_var_s.BETA, kaz_buf_ptr->sign.sign_local_var_s.r1, kaz_buf_ptr->sign.sign_local_var_s.phiphiG1, kaz_buf_ptr->sign.sign_local_var_s.tmp, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
495    	
496    	    //5) Generate random prime r, random OMEGA2
497    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->sign.sign_local_var_s.r2, HAL_SEC_PQC_KAZ_DS_SP_RAN, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, &kaz_buf_ptr->sign.sign_buf_s.sign_rand_val[rand_val_offset]);
498    	    rand_val_offset += (U8)HAL_SEC_PQC_KAZ_DS_SP_RAN;
499    	
500    	    hal_sec_pqc_kaz_pke_find_next_prime(kaz_buf_ptr->sign.sign_local_var_s.r2, kaz_buf_ptr->sign.sign_local_var_s.r2, &kaz_buf_ptr->sign.sign_buf_s.next_prime_buf, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
501    	
502    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->sign.sign_local_var_s.OMEGA3, HAL_SEC_PQC_KAZ_DS_SP_RAN, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, &kaz_buf_ptr->sign.sign_buf_s.sign_rand_val[rand_val_offset]);
503    	    rand_val_offset += (U8)HAL_SEC_PQC_KAZ_DS_SP_RAN;
504    	
505    	    //6) Compute BETA
506    	    (void)hal_sec_pqc_kaz_pke_mpy64(kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.OMEGA3, kaz_buf_ptr->sign.sign_local_var_s.phiG1);
507    	    hal_sec_pqc_kaz_pke_powm(kaz_buf_ptr->sign.sign_local_var_s.BETA2, kaz_buf_ptr->sign.sign_local_var_s.r2, kaz_buf_ptr->sign.sign_local_var_s.phiphiG1, kaz_buf_ptr->sign.sign_local_var_s.tmp, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
508    	
509    	    for (;;)
510    	    {
511    	        //7) Compute Signature
512    	        (void)hal_sec_pqc_kaz_pke_mpy64(kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.phiqQ, kaz_buf_ptr->sign.sign_local_var_s.BETA);
513    	        hal_sec_pqc_kaz_pke_powm(kaz_buf_ptr->sign.sign_local_var_s.S1, kaz_buf_ptr->sign.sign_local_var_s.hashValue, kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.G1qQ, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
514    	
515    	        (void)hal_sec_pqc_kaz_pke_mpy64(kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.phiqQ, kaz_buf_ptr->sign.sign_local_var_s.BETA2);
516    	        hal_sec_pqc_kaz_pke_powm(kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.hashValue, kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.G1qQ, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
517    	        (void)hal_sec_pqc_kaz_pke_add(kaz_buf_ptr->sign.sign_local_var_s.S1, kaz_buf_ptr->sign.sign_local_var_s.S1, kaz_buf_ptr->sign.sign_local_var_s.tmp);
518    	
519    	        (void)hal_sec_pqc_kaz_pke_div(kaz_buf_ptr->sign.sign_local_var_s.S1, kaz_buf_ptr->sign.sign_local_var_s.S1, kaz_buf_ptr->sign.sign_local_var_s.G1qQ, HAL_SEC_PQC_KAZ_PKE_MOD);
520    	
521    	        (void)hal_sec_pqc_kaz_pke_mpy64_mod(kaz_buf_ptr->sign.sign_local_var_s.S1, kaz_buf_ptr->sign.sign_local_var_s.S1, kaz_buf_ptr->sign.sign_local_var_s.SK, kaz_buf_ptr->sign.sign_local_var_s.G1qQ);
522    	
523    	        lenS = hal_sec_pqc_kaz_mp_size_in_base(kaz_buf_ptr->sign.sign_local_var_s.S1, 2UL);
524    	
525    	        //FILTER 3
526    	        hal_sec_pqc_kaz_pke_powm(kaz_buf_ptr->sign.sign_local_var_s.Y1, kaz_buf_ptr->sign.sign_local_var_s. V1, kaz_buf_ptr->sign.sign_local_var_s.phiQ, kaz_buf_ptr->sign.sign_local_var_s.G1Q, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
527    	        hal_sec_pqc_kaz_pke_powm(kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.hashValue, kaz_buf_ptr->sign.sign_local_var_s.phiqQ, kaz_buf_ptr->sign.sign_local_var_s.G1Q, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
528    	        (void)hal_sec_pqc_kaz_pke_shf(kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.tmp, HAL_SEC_PKE_SHIFT_L1);
529    	        (void)hal_sec_pqc_kaz_pke_div(kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.G1Q, HAL_SEC_PQC_KAZ_PKE_MOD);
530    	        (void)hal_sec_pqc_kaz_pke_mpy64_mod(kaz_buf_ptr->sign.sign_local_var_s.Y1, kaz_buf_ptr->sign.sign_local_var_s.Y1, kaz_buf_ptr->sign.sign_local_var_s.tmp, kaz_buf_ptr->sign.sign_local_var_s.G1Q);
531    	
532    	        hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->sign.sign_local_var_s.x[0], (U32 *)(void *)(kaz_buf_ptr->sign.sign_buf_s.sign_x0), HAL_SEC_PQC_KAZ_SIGN_X0_LEN);
533    	        hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->sign.sign_local_var_s.x[1], (U32 *)(void *)(kaz_buf_ptr->sign.sign_buf_s.sign_x1), HAL_SEC_PQC_KAZ_SIGN_X1_LEN);
534    	        hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->sign.sign_local_var_s.modulus[0], (U32 *)(void *)(kaz_buf_ptr->sign.sign_buf_s.sign_modulus0), HAL_SEC_PQC_KAZ_SIGN_MODULUS0_LEN);
535    	        hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->sign.sign_local_var_s.modulus[1], (U32 *)(void *)(kaz_buf_ptr->sign.sign_buf_s.sign_modulus1), HAL_SEC_PQC_KAZ_SIGN_MODULUS1_LEN);
536    	
537    	        (void)hal_sec_pqc_kaz_pke_div(kaz_buf_ptr->sign.sign_local_var_s.x[0], kaz_buf_ptr->sign.sign_local_var_s.V2, kaz_buf_ptr->sign.sign_local_var_s.Q, HAL_SEC_PQC_KAZ_PKE_DIV);
538    	        hal_sec_pqc_kaz_mp_set(kaz_buf_ptr->sign.sign_local_var_s.x[1], kaz_buf_ptr->sign.sign_local_var_s.Y1);
539    	
540    	        hal_sec_pqc_kaz_mp_set(kaz_buf_ptr->sign.sign_local_var_s.modulus[0], kaz_buf_ptr->sign.sign_local_var_s.q);
541    	        hal_sec_pqc_kaz_mp_set(kaz_buf_ptr->sign.sign_local_var_s.modulus[1], kaz_buf_ptr->sign.sign_local_var_s.G1Q);
542    	
543    	        (void)hal_sec_pqc_kaz_pke_crt_mpz(kaz_buf_ptr->sign.sign_local_var_s.x, kaz_buf_ptr->sign.sign_local_var_s.modulus, kaz_buf_ptr->sign.sign_local_var_s.SF1, &kaz_buf_ptr->sign.sign_buf_s.powm_buf);
544    	
545    	        if ((lenS == lenG1qQ) && (hal_sec_pqc_kaz_mp_cmp(kaz_buf_ptr->sign.sign_local_var_s.S1, kaz_buf_ptr->sign.sign_local_var_s.SF1) != 0))
546    	        {
547    	            break;
548    	        }
549    	        else
550    	        {
551    	            S2 = S2 + 1U;
552    	            (void)hal_sec_pqc_kaz_pke_add_ui(kaz_buf_ptr->sign.sign_local_var_s.hashValue, kaz_buf_ptr->sign.sign_local_var_s.hashValue, 1UL);
553    	        }
554    	
555    	        if (S2 > 0xFFU) // max 1 byte
556    	        {
557    	            return SEC_NO;
558    	        }
559    	    }
560    	
561    	    //6) Set signature=(S, m)
562    	    SSIZE = hal_sec_pqc_kaz_mp_size_in_base(kaz_buf_ptr->sign.sign_local_var_s.S1, 16UL);
563    	
564    	    hal_sec_pqc_kaz_mp_export(kaz_buf_ptr->sign.sign_local_var_s.SBYTE, &SSIZE, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, kaz_buf_ptr->sign.sign_local_var_s.S1);
565    	
566    	    for (int i = 0; i < HAL_SEC_PQC_KAZ_DS_SIG_LEN; i++)
567    	    {
568    	        signature[i] = 0;
569    	    }
570    	
571    	    int je = HAL_SEC_PQC_KAZ_DS_SIG_LEN - 1;
572    	    signature[je] = (U8)(S2 & 0xFFU);
573    	
574    	    je = HAL_SEC_PQC_KAZ_DS_SIG_LEN - 2;
575    	
576    	    for (int i = (int)SSIZE - 1; i >= 0; i--)
577    	    {
578    	        signature[je] = kaz_buf_ptr->sign.sign_local_var_s.SBYTE[i];
579    	        je--;
580    	    }
581    	
582    	    return SEC_YES;
583    	}

585    	U32 hal_sec_pqc_kaz_verify(U8 *digest, U8 *sm, U8 *pk, HAL_SEC_PQC_KAZ_BUF_S_PTR kaz_buf_ptr)
586    	{
587    	    U32 salt = 0UL;
588    	    U32 lenS = 0UL;
589    	    U32 lenG1qQ = 0UL;
590    	    U32 W1 = 0UL;
591    	
592    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.V1,         (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_v1),        HAL_SEC_PQC_KAZ_VERIFY_V1_LEN);
593    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.V2,         (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_v2),        HAL_SEC_PQC_KAZ_VERIFY_V2_LEN);
594    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.S1,         (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_s1),        HAL_SEC_PQC_KAZ_VERIFY_S1_LEN);
595    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.hashValue,  (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_hash_val),  HAL_SEC_PQC_KAZ_VERIFY_HASH_VAL_LEN);
596    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.tmp,        (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_temp),      HAL_SEC_PQC_KAZ_VERIFY_TEMP_LEN);
597    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.tmp2,       (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_temp2),     HAL_SEC_PQC_KAZ_VERIFY_TEMP2_LEN);
598    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.SF1,        (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_sf1),       HAL_SEC_PQC_KAZ_VERIFY_SF1_LEN);
599    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.SF2,        (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_sf2),       HAL_SEC_PQC_KAZ_VERIFY_SF2_LEN);
600    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.W4,         (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_w4),        HAL_SEC_PQC_KAZ_VERIFY_W4_LEN);
601    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.W5,         (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_w5),        HAL_SEC_PQC_KAZ_VERIFY_W5_LEN);
602    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.Y1,         (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_y1),        HAL_SEC_PQC_KAZ_VERIFY_Y1_LEN);
603    	    hal_sec_pqc_kaz_mp_init_addr(kaz_buf_ptr->verify.verify_local_var_s.Y2,         (U32 *)(void *)(kaz_buf_ptr->verify.verify_buf_s.verify_y2),        HAL_SEC_PQC_KAZ_VERIFY_Y2_LEN);
604    	
605    	    // Get all system parameters and precomputed parameters
606    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.G0,          sizeof(G0_CHAR),         sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)G0_CHAR);
607    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.R,           sizeof(R_CHAR),          sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)R_CHAR);
608    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.G1,          sizeof(G1_CHAR),         sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)G1_CHAR);
609    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.q,           sizeof(q_CHAR),          sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)q_CHAR);
610    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.Q,           sizeof(Q_CHAR),          sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)Q_CHAR);
611    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.G1Q,         sizeof(G1Q_CHAR),        sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)G1Q_CHAR);
612    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.G1qQ,        sizeof(G1qQ_CHAR),       sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)G1qQ_CHAR);
613    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.phiQ,        sizeof(PHIQ_CHAR),       sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)PHIQ_CHAR);
614    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.qQ,          sizeof(qQ_CHAR),         sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)qQ_CHAR);
615    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.phiqQ,       sizeof(PHIqQ_CHAR),      sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)PHIqQ_CHAR);
616    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.qQ_div_e,    sizeof(qQ_div_e_CHAR),   sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)qQ_div_e_CHAR);
617    	    hal_sec_pqc_kaz_mp_set_char(kaz_buf_ptr->verify.verify_local_var_s.G1qQ_div_e,  sizeof(G1qQ_div_e_CHAR), sizeof(U8), HAL_SEC_PQC_KAZ_LITTLE_ENDIAN, (U8 *)G1qQ_div_e_CHAR);
618    	
619    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->verify.verify_local_var_s.V1, HAL_SEC_PQC_KAZ_DS_V1_LEN, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, (U8 *) &pk[0]);
620    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->verify.verify_local_var_s.V2, HAL_SEC_PQC_KAZ_DS_V2_LEN, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, (U8 *) &pk[HAL_SEC_PQC_KAZ_DS_V1_LEN]);
621    	
622    	    //2) Get signature=(S, m)
623    	    salt = (U32)(sm[HAL_SEC_PQC_KAZ_DS_SIG_LEN - 1]);
624    	
625    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->verify.verify_local_var_s.S1, HAL_SEC_PQC_KAZ_DS_SIG_LEN - 1, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, (U8 *) &sm[0]);
626    	
627    	    //3) Compute the hash value of the message
628    	    hal_sec_pqc_kaz_mp_import(kaz_buf_ptr->verify.verify_local_var_s.hashValue, HAL_SEC_PQC_KAZ_SYS_LEN, sizeof(U8), HAL_SEC_PQC_KAZ_BIG_ENDIAN, digest);
629    	
630    	    (void)hal_sec_pqc_kaz_pke_add_ui(kaz_buf_ptr->verify.verify_local_var_s.hashValue, kaz_buf_ptr->verify.verify_local_var_s.hashValue, salt);
631    	
632    	    //4) Filtering Procedures
633    	    if (salt > 255UL)
634    	    {
635    	        return HAL_SEC_PQC_KAZ_ERROR(HAL_SEC_PQC_KAZ_FILITER_0_ERR);
636    	    }
637    	
638    	    //FILTER 1
639    	    (void)hal_sec_pqc_kaz_pke_div(kaz_buf_ptr->verify.verify_local_var_s.tmp, kaz_buf_ptr->verify.verify_local_var_s.S1, kaz_buf_ptr->verify.verify_local_var_s.G1qQ, HAL_SEC_PQC_KAZ_PKE_MOD);
640    	
641    	    if (hal_sec_pqc_kaz_mp_cmp(kaz_buf_ptr->verify.verify_local_var_s.tmp, kaz_buf_ptr->verify.verify_local_var_s.S1) != 0)
642    	    {
643    	        return HAL_SEC_PQC_KAZ_ERROR(HAL_SEC_PQC_KAZ_FILITER_1_ERR);
644    	    }
645    	
646    	    //FILTER 2
647    	    lenS = hal_sec_pqc_kaz_mp_size_in_base(kaz_buf_ptr->verify.verify_local_var_s.S1, 2U);
648    	    lenG1qQ = hal_sec_pqc_kaz_mp_size_in_base(kaz_buf_ptr->verify.verify_local_var_s.G1qQ, 2U);
649    	    W1 = lenS - lenG1qQ;
650    	
651    	    if (W1 != 0UL)
652    	    {
653    	        return HAL_SEC_PQC_KAZ_ERROR(HAL_SEC_PQC_KAZ_FILITER_2_ERR);
654    	    }
655    	
656    	    //FILTER 3
