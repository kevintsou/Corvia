398    	    else if (XMOD_SEND_POLL_FEEDBACK == xmodem->send_state) {

407    	            xmodem->packet_id++;
408    	
409    	            if (xmodem->send_byte_cnt < xmodem->trans_byte_cnt) {

423    	                     *  Last packet feed back correct, go to valid state.

426    	                }
427    	                else {
428    	                    /*
429    	                     *  Setup last packet EOT, packet_size should be 0.

433    	                    xmodem_send_to_state(xmodem, XMOD_SEND_1ST_TOK);

435    	            }
436    	        }
437    	        else if (XMOD_NAK == xmod_data) {
438    	            /*
439    	             *  send packet fail, go to retry
440    	             */
441    	            xmodem_send_to_state(xmodem, XMOD_SEND_RETRY);
442    	        }
443    	        else {
444    	            Dump_XModem(tprintf("Unknown Token: 0x%x", xmod_data));
445    	            BC_APP_ASSERT(BCODE_ERROR_VUC_XMODEM_UNKNOWN_TOKEN, 0);

447    	    }
