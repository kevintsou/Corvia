416    	            else {
417    	                /*
418    	                 *  Data transmission done, should end the transmission.
419    	                 *  Send EOT and wait for feedback.
420    	                 */
421    	                if (0 == xmodem->packet_size) {
422    	                    /*
423    	                     *  Last packet feed back correct, go to valid state.
424    	                     */
425    	                    xmodem_send_to_state(xmodem, XMOD_SEND_VALID_STATE);
426    	                }
427    	                else {
428    	                    /*
429    	                     *  Setup last packet EOT, packet_size should be 0.
430    	                     */
431    	                    xmodem->packet_size = 0;
432    	                    xmodem->token[FIRST_TOK] = XMOD_EOT;
433    	                    xmodem_send_to_state(xmodem, XMOD_SEND_1ST_TOK);
434    	                }
435    	            }
436    	        }
437    	        else if (XMOD_NAK == xmod_data) {
438    	            /*
439    	             *  send packet fail, go to retry
440    	             */
441    	            xmodem_send_to_state(xmodem, XMOD_SEND_RETRY);
442    	        }
443    	        else {
444    	            Dump_XModem(tprintf("Unknown Token: 0x%x", xmod_data));
445    	            BC_APP_ASSERT(BCODE_ERROR_VUC_XMODEM_UNKNOWN_TOKEN, 0);
(1) Event misra_c_2012_rule_10_1_violation: The expression "0" of non-boolean essential type is being interpreted as a boolean value for the operator "!".
446    	        }
447    	    }
448    	    else {
449    	        if (XMOD_NAK == xmod_data) {
450    	            xmodem_send_to_state(xmodem, XMOD_SEND_RETRY);
451    	        }
452    	    }
453    	}
