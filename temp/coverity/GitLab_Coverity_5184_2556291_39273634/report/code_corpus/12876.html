27    	static void write_buffer_ready(HOST_API_STRUCT_PTR host, U64 lca, U16 nlc)
28    	{
29    	
30    	    HAL_HCMD_STRUCT_PTR hcmd = (HAL_HCMD_STRUCT_PTR)&host->hcmd;
31    	    HAL_APU_WR_BUF_RDY_SQ_HW apu_sq;
32    	    HAL_APU_TRIG_DMA_PARAM write_buf_ready;
33    	
34    	    memset((void *)&write_buf_ready, 0x0U, sizeof(HAL_APU_TRIG_DMA_PARAM));
35    	
36    	    write_buf_ready.ctag                = hcmd->payload.info.user.ctag;
37    	    write_buf_ready.ud                  = 1U; // 1: user data
38    	    write_buf_ready.dir                 = 1U;
39    	    write_buf_ready.fst                 = (nlc == hcmd->payload.info.user.tnlc) ? TRUE : FALSE;
40    	    write_buf_ready.lst                 = (nlc == 1U) ? TRUE : FALSE;
41    	    write_buf_ready.node_tag            = 0U;
42    	    write_buf_ready.buf_addr_l          = (U32)host->buffer_base;
43    	    write_buf_ready.buf_addr_h          = 0;
44    	    write_buf_ready.spr_addr_l          = (U32)host->spr_base;
45    	    write_buf_ready.spr_addr_h          = 0;
46    	
47    	    write_buf_ready.wr_buf_rdy.lcap     = lca;
48    	    write_buf_ready.wr_buf_rdy.lca_h    = 0; // the high part LCA before remap(15bits)
