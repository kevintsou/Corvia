157    	void hal_fip_set_io_init(U8 io_type, U8 rx_mode, U8 tx_mode, U8 edo_en, U8 dis_dqsb, U8 dis_re, U8 high_speed)
158    	{
159    	    U8 io_type_val;
160    	    U8 rx_mode_val;
161    	    U8 ch = 0U;
162    	
163    	    switch (io_type) {
164    	    case HAL_FPL_IO_TYPE_SDR:
165    	        io_type_val = io_type;
166    	        break;
167    	    case HAL_FPL_IO_TYPE_DDR_SINGLE_END:
168    	        io_type_val = io_type;
169    	        break;
170    	    case HAL_FPL_IO_TYPE_DDR_DIFFERENTIAL:
171    	        io_type_val = io_type;
172    	        break;
173    	    default:
174    	        //default fix to BICS8 interface DDR_DIFFERENTIAL
175    	        io_type_val = HAL_FPL_IO_TYPE_DDR_DIFFERENTIAL;
176    	        break;
177    	    }
178    	
179    	    switch (rx_mode) {
180    	    case HAL_FPL_RX_MODE_CTT:
181    	        rx_mode_val = ((1 == high_speed) ? HAL_RX_MODE_CTT_DIFFERENTIAL_VAL : HAL_RX_MODE_CTT_VAL);   //FIP_TEST_TOGGLE_200MT
(1) Event misra_c_2012_rule_10_4_violation: Essential type of the left hand operand "1" (signed) is not the same as that of the right operand "high_speed"(unsigned).
182    	        break;
183    	    case HAL_FPL_RX_MODE_LTTNEQ:
184    	        rx_mode_val = HAL_RX_MODE_LTTNEQ_VAL;
185    	        break;
186    	    case HAL_FPL_RX_MODE_LTTEQ:
187    	        rx_mode_val = HAL_RX_MODE_LTTNEQ_VAL;
188    	        break;
189    	    default:
190    	        //default fix to BICS8 interface CTT MODE
191    	        rx_mode_val = HAL_RX_MODE_CTT_DIFFERENTIAL_VAL;
192    	        break;
193    	    }
194    	
195    	    // set
196    	    for (ch = 0U; ch < HAL_FIP_CH_NUM; ch++) {
197    	        // set io type

199    	        hal_fip_set_io_mode(ch, rx_mode_val, tx_mode);
200    	
201    	        // set edo mode

203

208    	        w_W_WARMUP_CYCLE_EN(ch, 0U);
209    	        w_R_WARMUP_CYCLE_EN(ch, 0U);
210    	    }
