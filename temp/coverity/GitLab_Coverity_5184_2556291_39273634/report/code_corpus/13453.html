594    	    memcpy((void *)calculation_buffer, calculation_buffer_backup, 32 + 32);
595    	}
596    	
597    	/**
598    	 * @brief Check the crc of encrypted command is correct after decryption
599    	 *
600    	 * @param sq_cmd sq pointer
601    	 * @return U32 fail or not
602    	 */
603    	U32 vuc_protect_check_command(VUC_SQ_STRUCT_PTR sq_cmd)
604    	{
605    	    VUC_SQ_STRUCT temp_sq_cmd;
606    	
607    	    U32 crc_cmd;
608    	    U32 crc_calc_result[2];
609    	
610    	    //Set the CRC field to 0 before calculating the CRC
611    	    //The CRC stored is in Big Endian
612    	    //The CRC calculation uses the same one that the XModem uses
613    	    memcpy((void *)&temp_sq_cmd, (void *)sq_cmd, sizeof(temp_sq_cmd));
614    	    crc_cmd = temp_sq_cmd.crc_word;
615    	    crc_cmd = (((crc_cmd & 0xFF00) >> 8) | ((crc_cmd & 0x00FF) << 8));
616    	    temp_sq_cmd.crc_word = 0;
617    	
618    	    //1. Calculate the CRC of the entire SQ without clearing anything
619    	    //This is the old way of calculating CRC. This is left over from the time
620    	    //when the CRC is only used for VUC in UART. It is recommended to be
621    	    //implemented to allow compatibility with older tools that may still fill
622    	    //in B1 and B2
623    	    crc_calc_result[0] = CRC_16_XMODEM((U32)&temp_sq_cmd, sizeof(temp_sq_cmd), CRC_16_XMODEM_INIT);
624    	
625    	    //2. Calculate the CRC of the entire SQ after clearing B1~3, DW1, DW2, DW3, DW4, DW5, DW6, DW7, DW8, DW9, DW11
626    	    //This is the new and correct way of calculating CRC. This is required
627    	    //after VUC Protect is implemented and the CRC is used to check for the
628    	    //VUC's correctness for all interface. The fields that are cleared to 0 are
629    	    //the fields in NVMe that the MP has no control over and may be filled
630    	    //automatically by the driver. The CRC is calculated with these fields set
631    	    //to 0.
632    	    memset((void *)((U32)&temp_sq_cmd + 1), 0x00, (3 + (9 * 4))); //Up to and including DW9
633    	    temp_sq_cmd.rsv_dw11 = 0;
634    	    crc_calc_result[1] = CRC_16_XMODEM((U32)&temp_sq_cmd, sizeof(temp_sq_cmd), CRC_16_XMODEM_INIT);
635    	
636    	    if ((crc_calc_result[0] == crc_cmd) || (crc_calc_result[1] == crc_cmd)) {
637    	#if DEBUG_VUC_PROTECT
638    	        Dump_Debug(tprintf("\n\nAES de crc pass!"));
639    	#endif
640    	        return VUC_PROTECT_CHECK_COMMAND_STATUS_PASS;
641    	    }
642    	    else {
643    	#if DEBUG_VUC_PROTECT
644    	        Dump_Debug(tprintf("\n\nRcv CRC (0x%llx) != CRC[0] (0x%llx) or CRC[1] (0x%llx)\r\n", (U64)crc_cmd, (U64)crc_calc_result[0], (U64)crc_calc_result[1]));
645    	#endif
646    	        return VUC_PROTECT_CHECK_COMMAND_STATUS_CRC_FAIL;
647    	    }
648    	}
