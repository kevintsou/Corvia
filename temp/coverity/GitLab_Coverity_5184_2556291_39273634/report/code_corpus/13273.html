314    	*              - U8 sig[]: byte array containing

319    	int hal_sec_pqc_mldsa_unpack_sig(U8 *c, HAL_SEC_PQC_MLDSA_POLYVECL_T_PTR z, HAL_SEC_PQC_MLDSA_POLYVECK_T_PTR h, U8 *sig, HAL_SEC_PQC_MLDSA_TEMP_BUF_S_PTR temp_buf_ptr)
320    	{
321    	    U32 k;
322    	
323    	    HAL_SEC_PQC_MLDSA_POLY_T_PTR temp = &temp_buf_ptr->unpack_sig.temp;
324    	    U8 *buf = temp_buf_ptr->unpack_sig.buf;
325    	
326    	    (void)memcpy((void *)c, (void *)sig, HAL_SEC_PQC_MLDSA_CTILD_LEN);

339    	    k = 0;

624    	*              Assumes input coefficients were reduced by hal_sec_pqc_mldsa_reduce32().
625    	*
626    	* Arguments:   - HAL_SEC_PQC_MLDSA_POLY_T_PTR a: pointer to polynomial
627    	*              - S32 B: norm bound
628    	*
629    	* Returns 0 if norm is strictly smaller than B <= (Q-1)/8 and 1 otherwise.
630    	**************************************************/
631    	int hal_sec_pqc_mldsa_poly_chknorm(HAL_SEC_PQC_MLDSA_POLY_T_PTR a, S32 B)
632    	{
633    	    S32 t;
634    	
635    	    if (B > ((S32)(HAL_SEC_PQC_MLDSA_Q) - 1) / 8)
636    	    {
637    	        return 1;
638    	    }
639    	
640    	    /* It is ok to leak which coefficient violates the bound since
641    	       the probability for each coefficient is independent of secret
642    	       data but we must not leak the sign of the centralized representative. */
643    	    for (U32 i = 0UL; i < HAL_SEC_PQC_MLDSA_N; ++i)

646    	        // t = a->coeffs[i] >> 31; // org

650    	        if (t >= B)
651    	        {
652    	            return 1;
653    	        }
654    	    }

656    	    return 0;

665    	* Arguments:   - S32 *a: pointer to output array (allocated)
666    	*              - U32 len: number of coefficients to be sampled

668    	*              - U32 buflen: length of array of random bytes
669    	*
670    	* Returns number of sampled coefficients. Can be smaller than len if not enough
671    	* random bytes were given.
672    	**************************************************/
673    	U32 hal_sec_pqc_mldsa_rej_uniform(S32 *a, U32 len, U8 *buf, U32 buflen)
674    	{
675    	    U32 ctr, pos;
676    	    U32 tmp; // avoid misra-c 10.1 (signed int cannot shift) and 10.8 (cast signed to unsigned int)
677    	
678    	    U64 t1, t2, t3;
679    	    ctr = pos = 0UL;
680    	    while ((ctr < len) && (pos + 3UL <= buflen))
681    	    {
682    	        t1 = *(U64 *)((U32)buf + pos);
683    	        t2 = *(U64 *)((U32)buf + pos + 8UL);
684    	        t3 = *(U64 *)((U32)buf + pos + 16UL);
685    	
686    	        if ((t1 & 0x7FFFFFULL) < (U64)(HAL_SEC_PQC_MLDSA_Q))

688    	            tmp = (U32)(t1 & 0x7FFFFFULL);
689    	            a[ctr++] = (S32)tmp;
690    	        }
691    	
692    	        t1 >>= 24ULL;
693    	
694    	        if ((t1 & 0x7FFFFFULL) < (U64)(HAL_SEC_PQC_MLDSA_Q))
695    	        {
696    	            tmp = (U32)(t1 & 0x7FFFFFULL);
697    	            a[ctr++] = (S32)tmp;
698    	        }
699    	
700    	        t1 >>= 24ULL;
701    	        t1 |= (t2 & 0xFFULL) << 16ULL;
702    	
703    	        if ((t1 & 0x7FFFFFULL) < (U64)(HAL_SEC_PQC_MLDSA_Q))
704    	        {
705    	            tmp = (U32)(t1 & 0x7FFFFFULL);
706    	            a[ctr++] = (S32)tmp;
707    	        }
708    	
709    	        t2 >>= 8ULL;
710    	
711    	        if ((t2 & 0x7FFFFFULL) < (U64)(HAL_SEC_PQC_MLDSA_Q))
712    	        {
713    	            tmp = (U32)(t2 & 0x7FFFFFULL);
714    	            a[ctr++] = (S32)tmp;
715    	        }
716    	
717    	        t2 >>= 24;
718    	
719    	        if ((t2 & 0x7FFFFFULL) < (U64)(HAL_SEC_PQC_MLDSA_Q))
720    	        {
721    	            tmp = (U32)(t2 & 0x7FFFFFULL);
722    	            a[ctr++] = (S32)tmp;
723    	        }
724    	
725    	        t2 >>= 24ULL;
726    	        t2 |= (t3 & 0xFFFFULL) << 8ULL;
727    	
728    	        if ((t2 & 0x7FFFFFULL) < (U64)(HAL_SEC_PQC_MLDSA_Q))
729    	        {
730    	            tmp = (U32)(t2 & 0x7FFFFFULL);

751    	    }
752    	
753    	    return ctr;

991    	* Name:        hal_sec_pqc_mldsa_poly_eta_unpack
992    	*
993    	* Description: Unpack polynomial with coefficients in [-ETA,ETA].
994    	*
995    	* Arguments:   - HAL_SEC_PQC_MLDSA_POLY_T_PTR r: pointer to output polynomial
996    	*              - U8 *a: byte array with bit-packed polynomial
997    	**************************************************/
998    	void hal_sec_pqc_mldsa_poly_eta_unpack(HAL_SEC_PQC_MLDSA_POLY_T_PTR r, U8 *a)
999    	{
1000    	    S32 ETA = (S32)(HAL_SEC_PQC_MLDSA_ETA);
1001    	    U8 tmp; // avoid misra-c 10.1 (signed int cannot shift) and 10.8 (cast signed to unsigned int)
1002    	
1003    	    for (U32 i = 0UL; i < (HAL_SEC_PQC_MLDSA_N / 8UL); ++i)
1004    	    {
1005    	        tmp = ((a[3U * i + 0U] >> 0U) & 7U);
1006    	        r->coeffs[8U * i + 0U] = (S32)tmp;
1007    	        tmp = ((a[3U * i + 0U] >> 3U) & 7U);
1008    	        r->coeffs[8U * i + 1U] = (S32)tmp;
1009    	        tmp = (((a[3U * i + 0U] >> 6U) | (a[3U * i + 1U] << 2U)) & 7U);
1010    	        r->coeffs[8U * i + 2U] = (S32)tmp;
1011    	        tmp = ((a[3U * i + 1U] >> 1U) & 7U);
1012    	        r->coeffs[8U * i + 3U] = (S32)tmp;
1013    	        tmp = ((a[3U * i + 1U] >> 4U) & 7U);
1014    	        r->coeffs[8U * i + 4U] = (S32)tmp;
1015    	        tmp = (((a[3U * i + 1U] >> 7U) | (a[3U * i + 2U] << 1U)) & 7U);
1016    	        r->coeffs[8U * i + 5U] = (S32)tmp;
1017    	        tmp = ((a[3U * i + 2U] >> 2U) & 7U);
1018    	        r->coeffs[8U * i + 6U] = (S32)tmp;
1019    	        tmp = ((a[3U * i + 2U] >> 5U) & 7U);
1020    	        r->coeffs[8U * i + 7U] = (S32)tmp;
1021    	
1022    	        r->coeffs[8U * i + 0U] = ETA - r->coeffs[8U * i + 0U];
1023    	        r->coeffs[8U * i + 1U] = ETA - r->coeffs[8U * i + 1U];
1024    	        r->coeffs[8U * i + 2U] = ETA - r->coeffs[8U * i + 2U];

1792    	* Name:        hal_sec_pqc_mldsa_montgomery_reduce
1793    	*
1794    	* Description: For finite field element a with -2^{31}Q <= a <= Q*2^31,
1795    	*              compute r \equiv a*2^{-32} (mod Q) such that -Q < r < Q.
1796    	*
1797    	* Arguments:   - S64: finite field element a
1798    	*
1799    	* Returns r.
1800    	**************************************************/
1801    	S32 hal_sec_pqc_mldsa_montgomery_reduce(S64 a)
1802    	{
1803    	    S32 t;
1804    	    U64 tmp; // avoid misra-c 10.1 (signed int cannot shift) and 10.8 (cast signed to unsigned int)
1805    	
1806    	    tmp = ((U64)a * (U64)58728449);
1807    	    t = (S32)tmp;
1808    	    tmp = (((U64)a - (U64)t * (U64)(HAL_SEC_PQC_MLDSA_Q)) >> 32ULL);
1809    	    t = (S32)tmp;
1810    	
1811    	    return t;
1812    	}

1815    	* Name:        hal_sec_pqc_mldsa_reduce32
1816    	*
1817    	* Description: For finite field element a with a <= 2^{31} - 2^{22} - 1,
1818    	*              compute r \equiv a (mod Q) such that -6283009 <= r <= 6283007.
1819    	*
1820    	* Arguments:   - S32: finite field element a
1821    	*
1822    	* Returns r.
1823    	**************************************************/
1824    	S32 hal_sec_pqc_mldsa_reduce32(S32 a)
1825    	{
1826    	    S32 t;
1827    	    U32 bit22 = (1UL << 22UL);
1828    	    U32 tmp; // avoid misra-c 10.1 (signed int cannot shift) and 10.8 (cast signed to unsigned int)
1829    	
1830    	    // negative number right shift need to fill 1 at MSB bit
1831    	    // t = (a + (1 << 22)) >> 23; // org
1832    	    // t = a - t * HAL_SEC_PQC_MLDSA_Q;
1833    	    tmp = (((a + (S32)bit22) < 0) ? ((((U32)a + bit22) >> 23UL) | 0xFFFFFE00UL) : (((U32)a + bit22) >> 23UL));
1834    	    t = a - (S32)tmp * (S32)(HAL_SEC_PQC_MLDSA_Q);
1835    	
1836    	    return t;
1837    	}

1902    	/*************************************************
1903    	* Name:        hal_sec_pqc_mldsa_decompose
1904    	*
1905    	* Description: For finite field element a, compute high and low bits a0, a1 such
1906    	*              that a mod^+ Q = a1*ALPHA + a0 with -ALPHA/2 < a0 <= ALPHA/2 except
1907    	*              if a1 = (Q-1)/ALPHA where we set a1 = 0 and
1908    	*              -ALPHA/2 <= a0 = a mod^+ Q - Q < 0. Assumes a to be standard
1909    	*              representative.
1910    	*
1911    	* Arguments:   - S32 a: input element
1912    	*              - S32 *a0: pointer to output element a0
1913    	*
1914    	* Returns a1.
1915    	**************************************************/
1916    	S32 hal_sec_pqc_mldsa_decompose(S32 *a0, S32 a)
1917    	{
1918    	    S32 a1;
1919    	    U32 bit21 = (1UL << 21UL);
1920    	    U32 tmp; // avoid misra-c 10.1 (signed int cannot shift) and 10.8 (cast signed to unsigned int)
1921    	
1922    	    // negative number right shift need to fill 1 at MSB bit
1923    	    // a1  = (a + 127) >> 7; // org
1924    	    // a1  = (a1 * 1025 + (1 << 21)) >> 22; // org
1925    	    // a1 &= 15; // org
1926    	    tmp = (((a + 127) < 0) ? ((((U32)a + 127UL) >> 7UL) | 0xFE000000UL) : (((U32)a + 127UL) >> 7UL));
1927    	    a1 = (S32)tmp;
1928    	    tmp = (((a1 * 1025 + (S32)bit21) < 0) ? ((((U32)a1 * 1025UL + bit21) >> 22UL) | 0xFFFFFC00UL) : (((U32)a1 * 1025UL + bit21) >> 22UL));
1929    	    a1 = (S32)tmp;
1930    	    tmp = ((U32)a1 & 15UL);
1931    	    a1 = (S32)tmp;
1932    	
1933    	    tmp = HAL_SEC_PQC_MLDSA_GAMMA2;
1934    	    *a0  = a - a1 * 2 * (S32)tmp;
1935    	    // *a0 -= (((HAL_SEC_PQC_MLDSA_Q - 1) / 2 - *a0) >> 31) & HAL_SEC_PQC_MLDSA_Q; // org
1936    	    tmp = ((((((S32)(HAL_SEC_PQC_MLDSA_Q) - 1) / 2 - *a0) < 0) ? ((((HAL_SEC_PQC_MLDSA_Q - 1UL) / 2UL - (U32)(*a0)) >> 31UL) | 0xFFFFFFFEUL) : (((HAL_SEC_PQC_MLDSA_Q - 1UL) / 2UL - (U32)(*a0)) >> 31UL)) & HAL_SEC_PQC_MLDSA_Q);
1937    	    *a0 -= (S32)tmp;
1938    	    return a1;
1939    	}

1942    	* Name:        hal_sec_pqc_mldsa_make_hint
1943    	*
1944    	* Description: Compute hint bit indicating whether the low bits of the
1945    	*              input element overflow into the high bits.
1946    	*
1947    	* Arguments:   - S32 a0: low bits of input element
1948    	*              - S32 a1: high bits of input element
1949    	*
1950    	* Returns 1 if overflow.
1951    	**************************************************/
1952    	U32 hal_sec_pqc_mldsa_make_hint(S32 a0, S32 a1)

1954    	    U32 gamma2 = HAL_SEC_PQC_MLDSA_GAMMA2;

1957    	    if ((a0 > (S32)gamma2) || (a0 < gamma2_negative) || ((a0 == gamma2_negative) && (a1 != 0)))

1960    	    }
1961    	
1962    	    return 0;
