75    	    drbg_get_try_limit_rng_data(&gRnghealth, sizeof(rng), rng);

77    	    // Compare all efuse key hash
78    	    for (U8 idx = 0; idx < EFUSE_VUC_PROTECT_KEY_INDEX_COUNT; idx++)
(1) Event incr: Incrementing "idx". The value of "idx" is now 1.
(2) Event between: At condition "idx < 2U", the value of "idx" must be between 0 and 1.
(3) Event assignment: Assigning: "idx" = "0".
(4) Event cond_const: Condition "idx < 2U", taking true branch. Now the value of "idx" is equal to 0.
(5) Event misra_c_2012_rule_14_3_violation: The condition "idx < 2U" must be true.
79    	    {

86    	            clear_vuc_key_hash(&gEFuseAPI);

89
