1988    	/*!
1989    	 * @brief trigger mdll tracking
1990    	 *
1991    	 * -
1992    	 *
1993    	 * @return    None
1994    	 * @note
1995    	 *
1996    	 * -
1997    	 *
1998    	 */
1999    	void dal_fip_mdll_tracking_trig(void)
2000    	{
2001    	#define WAIT_CYCLE(x)    {for (i = 0U; i < (x); i++) {r_FLH_DLL_MDLL_TRK_DONE(0U);} }
2002    	
2003    	    U8   dll, dll_num, ch;
2004    	    U32  i, tmp32;
2005    	    U16  ratio = 512U;
2006    	
2007    	    dll_num = DAL_FIP_CH_NUM >> 1U;
2008    	
2009    	    // -----------------------------------------------
2010    	    // Tracking Start
2011    	    // -----------------------------------------------
2012    	    for (dll = 0U; dll < dll_num; dll++) {
2013    	        ch = dll << 1U;
2014    	
2015    	        //step1
2016    	        // TODO: need to adjust by SYS clk
2017    	        w_FLH_DLL_MDLL_TRK_HOLD_CNT(ch, 50);
2018    	
2019    	        //step2
2020    	        w_FLH_DLL_DLL_CH0_W_TRACK_RATIO(ch, ratio);
2021    	        w_FLH_DLL_DLL_CH1_W_TRACK_RATIO(ch, ratio);
2022    	    }
2023    	
2024    	    //step3. mdll training
2025    	    for (dll = 0U; dll < dll_num; dll++) {
2026    	        ch = dll << 1U;
2027    	
2028    	        //step3.1 MSB_DLY_TRN_RSTB = 0;                                            // Reset MDLL MSB Train
2029    	        w_FLH_DLL_MSB_DLY_TRN_RSTB(ch, 0);
2030    	
2031    	        //step3.2 Wait for 40 CLK_FAPHY_TX cycles                                  // Meet FPHY Spec with margin
2032    	        WAIT_CYCLE(40U);
2033    	
2034    	        //step3.3 MSB_DLY_TRN_RSTB = 1;                                            // Reset MDLL MSB Train done
2035    	        w_FLH_DLL_MSB_DLY_TRN_RSTB(ch, 1);
2036    	
2037    	        //step3.4 Wait for 40 CLK_FAPHY_TX cycles                                  // Meet FPHY Spec with margin
2038    	        WAIT_CYCLE(40U);
2039    	
2040    	        //step3.5 MSB_DLY_TRN_EN = 1;                                              // Enable MDLL MSB Train
2041    	        w_FLH_DLL_MSB_DLY_TRN_EN(ch, 1);
2042    	
2043    	        //step3.6 while (MSB_DLY_TRN_RDY == 0) with Timeout Threshold = 1ms;       // Wait MSB Train done
2044    	        while (r_FLH_DLL_MSB_DLY_TRN_RDY(ch) == 0U) {
2045    	            // None
2046    	        }
2047    	
2048    	        //step3.7 MSB_DLY_CFG_IN = MSB_DLY_TRN_OUT;                                // Load register with Train result
2049    	        tmp32 = r_FLH_DLL_MSB_DLY_TRN_OUT(ch);
2050    	        w_FLH_DLL_MSB_DLY_CFG_IN(ch, tmp32);
2051    	
2052    	        //step3.8 Wait for 40 CLK_FAPHY_TX cycles                                  // Meet FPHY Spec with margin
2053    	        WAIT_CYCLE(10U);
2054    	
2055    	        //step3.9 MSB_DLY_TRN_EN = 0;                                              // Disable MDLL MSB Train
2056    	        w_FLH_DLL_MSB_DLY_TRN_EN(ch, 0);
2057    	
2058    	        //step3.10 Wait for 40 CLK_FAPHY_TX cycles                                  // Meet FPHY Spec with margin
2059    	        WAIT_CYCLE(10U);
2060    	
2061    	        //step3.11 MSB_DLY_TRN_RSTB = 0;                                            // Reset MDLL MSB Train
2062    	        w_FLH_DLL_MSB_DLY_TRN_RSTB(ch, 0);
2063    	    }
2064    	
2065    	    for (dll = 0U; dll < dll_num; dll++) {
2066    	        ch = dll << 1U;
2067    	
2068    	        // MDLL_TRK_EN = 0;                                                 // Clear MDLL tracking state
2069    	        w_FLH_DLL_MDLL_TRK_EN(ch, 0);
2070    	
2071    	        //step4
2072    	        w_FLH_DLL_MDLL_DIR_MACRO_RSTB(ch, 0);
2073    	        w_FLH_DLL_MDLL_DIR_MACRO_EN(ch, 0);
2074    	
2075    	        //step5 FW MDLL-MSB + HW MDLL-LSB Tracking
2076    	        //ref:http://192.168.1.137/svn/PS5028/A02_HW/B03_IP_Specification/C01_Digital_IP/flh_ip/app note/PS5031_MDLL_Application_Note.docx
2077    	        w_FLH_DLL_MDLL_DIR_MACRO_RSTB(ch, 1);
2078    	        w_FLH_DLL_MDLL_DIR_MODE(ch, 0);
2079    	        w_FLH_DLL_MDLL_TRK_SRCH_MD(ch, 0); //E28 doesn't have MSB, use srch_mode = 0
2080    	
2081    	        w_FLH_DLL_MDLL_TRK_EN(ch, 1);
2082    	    }
2083    	#undef WAIT_CYCLE
2084    	}
