545    	    U16 cmp_rtn       = HAL_SEC_DMSCA_FAIL;

574    	                break;

965    	/*!
966    	 @brief execute frequency test
967    	 @param[in]  n             - data length, unit of bit
968    	 @param[in]  epsilon       - source data buffer of bit array
969    	 @param[out] ret           - test result
970    	 @return test result
971    	 @return 0U - p_value < 0.01
972    	 @return 1U - p_value >= 0.01
973    	 @note
974    	
975    	 - None
976    	
977    	*/
978    	U8 hal_sec_frequency_test(U8 n, U8 *epsilon)
979    	{
980    	    double f, s_obs, p_value, sum = 0.0;
981    	    double sqrt2 = 1.41421356237309504880;
982    	
983    	    for (U8 i = 0U; i < n; i++)
984    	    {
985    	        sum += (double)2.0 * (double)epsilon[i] - (double)1.0;
986    	    }
987    	
988    	    s_obs = (double)(fabs(sum)) / (sqrt((double)n));
989    	    f = s_obs / sqrt2;
990    	    p_value = erfc(f);
991    	
992    	    if (p_value < 0.01)
993    	    {
994    	        //RSA_Printf("frequency p_value < 0.01\n");
995    	        return SEC_NO;
996    	    }
997    	    else
998    	    {
999    	        //RSA_Printf("frequency p_value >= 0.01\n");
1000    	        return SEC_YES;
1001    	    }
1002    	}

1004    	/*!
1005    	 @brief execute self correlation test
1006    	 @param[in]  d             - number of digits to shift left
1007    	 @param[in]  n             - data length, unit of bit
1008    	 @param[in]  epsilon       - source data buffer of bit array
1009    	 @param[out] ret           - test result
1010    	 @return test result
1011    	 @return 0U - p_value < 0.01
1012    	 @return 1U - p_value >= 0.01
1013    	 @note
1014    	
1015    	 - None
1016    	
1017    	*/
1018    	U8 hal_sec_self_correlation_test(U8 d, U8 n, U8 epsilon[])
1019    	{
1020    	    // epsilon is input pattern
1021    	    U8 n_d = n - d;
1022    	    U32 Ad = 0UL;
1023    	    double V, p_value = 0.0;
1024    	    double sqrt2 = 1.41421356237309504880;
1025    	
1026    	    for (U8 i = 0U; i < (n_d - 1U); ++i)
1027    	    {
1028    	        Ad += ((U32)(epsilon[i]) ^ (U32)(epsilon[i + d]));
1029    	    }
1030    	
1031    	    V = 2.0 * ((double)Ad - ((double)n_d / (double)2.0)) / (sqrt((double)n_d));
1032    	
1033    	    p_value = erfc(fabs(V) / sqrt2);
1034    	
1035    	    if (p_value < 0.01)
1036    	    {
1037    	        //RSA_Printf("self_correlation p_value < 0.01\n");
1038    	        return SEC_NO;
1039    	    }
1040    	    else
1041    	    {
1042    	        //RSA_Printf("self_correlation p_value >= 0.01\n");
1043    	        return SEC_YES;
1044    	    }
1045    	}

1047    	/*!
1048    	 @brief execute binary derivate test
1049    	 @param[in]  k              - the times of the sequence doing binary derivate test
1050    	 @param[in]  n              - sequence length, unit of bit
1051    	 @param[in]  epsilon_l      - epsilon length, unit of bit
1052    	 @param[in]  epsilon        - source data buffer of bit array
1053    	 @param[in]  temp_epsilon   - temp data buffer of bit array
1054    	 @param[out] ret            - test result
1055    	 @return test result
1056    	 @return 0U - p_value < 0.01
1057    	 @return 1U - p_value >= 0.01
1058    	 @note
1059    	
1060    	 - None
1061    	
1062    	*/
1063    	U8 hal_sec_binary_derivate_test(U32 k, U8 n, U8 epsilon_l, U8 epsilon[], U8 temp_epsilon[])
1064    	{
1065    	    // epsilon is input pattern
1066    	    U8 Sn_k = 0U;
1067    	    U8 n_k = n - (U8)k;
1068    	    double V = 0.0, p_value = 0.0, sqrt2 = 1.41421356237309504880;
1069    	
1070    	    for (U8 i = 0U; i < k; ++i)
1071    	    {
1072    	        for (U8 j = 0U; j < (epsilon_l - 1U); ++j)
1073    	        {
1074    	            temp_epsilon[j] = epsilon[j] ^ epsilon[j + 1U];
1075    	        }
1076    	    }
1077    	
1078    	    for (U8 i = 0U; i < n_k; ++i)
1079    	    {
1080    	        Sn_k += (2U * temp_epsilon[i]) - 1U;
1081    	    }
1082    	
1083    	    V = (double)(fabs((double)Sn_k)) / (sqrt((double)n_k));
1084    	
1085    	
1086    	    p_value = erfc(fabs(V) / sqrt2);
1087    	
1088    	    if (p_value < 0.01)
1089    	    {
1090    	        //RSA_Printf("binary_derivate p_value < 0.01\n");
1091    	        return SEC_NO;
1092    	    }
1093    	    else
1094    	    {
1095    	        //RSA_Printf("binary_derivate p_value >= 0.01\n");
1096    	        return SEC_YES;
1097    	    }
1098    	}
