22    	/**
23    	 * Handles the Landmark data structure by performing the indicated number of entries within the Landmark.
24    	 * The Landmark is used as a way to change certain registers to affect the behavior of Boot Code.
25    	 * The Landmark is not limited to the one within the ID Page.
26    	 * Landmark Entry functionalities:
27    	 *  Polling - Polls for a bit at the indicated memory location to become a 1 or 0. A timeout is
28    	 *              also included to prevent indefinite polling.
29    	 *  Idling  - Idles for the amount of time specified.
30    	 *  Setting - Sets a 32Bit value that shod be 4 byte aligned to a value indicated in the entry.
31    	 *
32    	 * @param id_table_landmark_ptr
33    	 *      The address to the first entry within the Landmark data structure
34    	 * @param landmark_len
35    	 *      The number of entries within the Landmark data structure
36    	 * @return
37    	 *      Returns the number of times polling has timed out.
38    	 */
39    	U32 id_table_general_landmark_handle(ID_TABLE_LANDMARK_PTR id_table_landmark_ptr, U32 landmark_len)
40    	{
41    	    U32 landmark_entry = 0;
42    	    volatile U32 *address_base;
43    	    U32 reg_data, poll_time_us, poll_status;
44    	    U32 idle_unit_cnt;
45    	    U32 fail_cnt;
46    	
47    	    U64 poll_time_us_tmp;   // for coverity
48    	
49    	    Dump_LM(tprintf("Handling %d landmarks", landmark_len));
50    	    for (landmark_entry = 0; landmark_entry < landmark_len; landmark_entry++) {
51    	        Dump_LM(tprintf("BD %d: %x %x", landmark_entry, id_table_landmark_ptr[landmark_entry].reg.long_address, id_table_landmark_ptr[landmark_entry].data));
52    	    }
53    	
54    	    fail_cnt = 0;
55    	    for (landmark_entry = 0; landmark_entry < landmark_len; landmark_entry++) {
56    	        Dump_LM(tprintf("BD %d: %x %x ", landmark_entry, id_table_landmark_ptr[landmark_entry].reg.long_address, id_table_landmark_ptr[landmark_entry].data));
57    	        Dump_LM(tprintf("(Op %x Rsv %x Poll Status %x Address %x)", id_table_landmark_ptr[landmark_entry].reg.op, id_table_landmark_ptr[landmark_entry].reg.rsv, id_table_landmark_ptr[landmark_entry].reg.poll_status, id_table_landmark_ptr[landmark_entry].reg.address));
58    	
59    	        //Polling: The Landmark Entry is for Polling a bit
60    	        if (0xFFF == (id_table_landmark_ptr[landmark_entry].reg.op)) {
61    	            Dump_LM(tprintf("Polling - "));
62    	            address_base = (U32 *)((id_table_landmark_ptr[landmark_entry].data & 0xFFFF0000) | (id_table_landmark_ptr[landmark_entry].reg.address));
63    	            reg_data = id_table_landmark_ptr[landmark_entry].data & 0xFF;
64    	            Dump_LM(tprintf(" Addr %x Bit %d ", (U32)address_base, reg_data));
65    	            Dump_LM(tprintf("(Value %x)", *address_base));
66    	
67    	            poll_time_us = ((id_table_landmark_ptr[landmark_entry].data >> 8) & 0xFF) * 50; //update the spec
68    	            if (0 == poll_time_us) {
69    	                poll_time_us = 0x0100 * 50;
70    	            }
71    	            poll_status = (id_table_landmark_ptr[landmark_entry].reg.poll_status);
72    	
73    	            Dump_LM(tprintf(" for %db, time %dus", poll_status, poll_time_us));
74    	            poll_time_us_tmp = poll_time_us;
75    	            U32 mabiao_addr = MABIAO_SET_US(MABIAO_DB_LANDMARK, poll_time_us_tmp);
76    	            while ((((*address_base) >> reg_data) & 0x01) != poll_status) {
77    	                if ((TRUE == MABIAO_CHK(mabiao_addr)) && (poll_time_us != 0)) {
78    	                    Dump_LM(tprintf(" Timeout"));
79    	                    fail_cnt++;
80    	                    break;
81    	                }
82    	            }
83    	            Dump_LM(tprintf(" Done"));
84    	        }
85    	        //Idling: The Landmark Fill Entry is for idling for an amount of time
86    	        else if (0xFFE == (id_table_landmark_ptr[landmark_entry].reg.op)) {
87    	            Dump_LM(tprintf("Idling - "));
88    	            idle_unit_cnt = id_table_landmark_ptr[landmark_entry].data;
89    	            Dump_LM(tprintf("%d", idle_unit_cnt));
90    	            switch (id_table_landmark_ptr[landmark_entry].reg.address) {
91    	            case 0x0000:
92    	                Dump_LM(tprintf("PC"));
93    	                idle_pc(idle_unit_cnt);
94    	                break;
95    	            case 0x0001:
96    	                Dump_LM(tprintf("uS"));
97    	                MABIAO_IDLE_US(idle_unit_cnt);
98    	                break;
99    	            case 0x0002:
100    	                Dump_LM(tprintf("mS"));
101    	                MABIAO_IDLE_MS(idle_unit_cnt);
102    	                break;
103    	            default:
104    	                Dump_LM(tprintf("S"));
105    	                MABIAO_IDLE_S(idle_unit_cnt);
106    	                break;
107    	            }
108    	        }
109    	        //Setting: The Landmark Fill Entry is for Setting a 32-bit value
110    	        else {
111    	            Dump_LM(tprintf("Filling - "));
112    	            address_base = (U32 *)(id_table_landmark_ptr[landmark_entry].reg.long_address);

116    	        }
117    	    }
118    	
119    	    Dump_LM(tprintf("Landmark Handle at %x for %d fail %d times", (U32)id_table_landmark_ptr, landmark_len, fail_cnt));
120    	    return fail_cnt;
121    	}

169    	static void efuse_bootcode_landmark_handle(EFUSE_BOOTCODE_LANDMARK_PTR landmark)
170    	{
171    	    volatile U8 *address_base = 0;
172    	    U32 reg_data, poll_time_us, poll_status;
173    	    U32 idle_unit_cnt;
174    	    U32 fail_cnt = 0;
175    	
176    	    Dump_LM(tprintf("Data %x, Addr %x, Ctrl %x \r\n", landmark->data, landmark->addr, landmark->ctrl));
177    	
178    	    if (landmark->ctrl == 0x0) { //nop
179    	        return;
180    	    }
181    	    if (landmark->ctrl == 0x1) {
182    	
183    	        Dump_LM(tprintf("assign:"));
184    	        address_base = (U8 *)(landmark->addr + 0x2000000);
185    	        reg_data = landmark->data;
186    	        Dump_LM(tprintf("Addr 0x%x Data 0x%x (Orig 0x%x)", address_base, reg_data, *address_base));
187    	        *address_base = reg_data;
188    	    }
189    	    else if (landmark->ctrl == 0x2) {
190    	        Dump_LM(tprintf("and:"));
191    	        address_base = (U8 *)(landmark->addr + 0x2000000);
192    	        reg_data = landmark->data;
193    	        Dump_LM(tprintf("Addr 0x%x Data 0x%x (Orig 0x%x)", address_base, reg_data, *address_base));
194    	        *address_base &= reg_data;
195    	    }
196    	    else if (landmark->ctrl == 0x3) {
197    	        Dump_LM(tprintf("or:"));
198    	        address_base = (U8 *)(landmark->addr + 0x2000000);
199    	        reg_data = landmark->data;
200    	        Dump_LM(tprintf("Addr 0x%x Data 0x%x (Orig 0x%x)", address_base, reg_data, *address_base));
201    	        *address_base |= reg_data;
202    	    }
203    	    else if (landmark->ctrl == 0x4) {
204    	        Dump_LM(tprintf("xor:"));
205    	        address_base = (U8 *)(landmark->addr + 0x2000000);
206    	        reg_data = landmark->data;
207    	        Dump_LM(tprintf("Addr 0x%x Data 0x%x (Orig 0x%x)", address_base, reg_data, *address_base));
208    	        *address_base ^= reg_data;
209    	    }
210    	    else if (landmark->ctrl == 0x5) { //idle
211    	        Dump_LM(tprintf("idle:"));
212    	        idle_unit_cnt = landmark->addr;
213    	        Dump_LM(tprintf("idle %d ", idle_unit_cnt));
214    	
215    	        switch (landmark->data) {
216    	        case 0x0000:
217    	            Dump_LM(tprintf("PC"));
218    	            idle_pc(idle_unit_cnt);
219    	            break;
220    	
221    	        case 0x0001:
222    	            Dump_LM(tprintf("uS"));
223    	            MABIAO_IDLE_US(idle_unit_cnt);
224    	            break;
225    	
226    	        case 0x0002:
227    	            Dump_LM(tprintf("mS"));
228    	            MABIAO_IDLE_MS(idle_unit_cnt);
229    	            break;
230    	
231    	        default:
232    	            Dump_LM(tprintf("S"));
233    	            MABIAO_IDLE_S(idle_unit_cnt);
234    	            break;
235    	        }
236    	    }
237    	    else {      //polling
238    	        if ((landmark->ctrl & 0x8) > 0) { // polling bit = 1
239    	            poll_status = 1;
240    	        }
241    	        else {
242    	            poll_status = 0;
243    	        }
244    	
245    	        Dump_LM(tprintf("Polling:"));
246    	
247    	        address_base = (U8 *)(landmark->addr + 0x2000000);
248    	        reg_data = landmark->data;
249    	
250    	        Dump_LM(tprintf("Addr %x Bit %x ", (U32)address_base, reg_data));
251    	        Dump_LM(tprintf("(Value %x)", *address_base));
252    	
253    	        poll_time_us = landmark->data * 50; //update the spec
254    	        if (0 == poll_time_us) {
255    	            poll_time_us = 0x0100 * 50;
256    	        }
257    	
258    	        Dump_LM(tprintf("for %db, time %dus", poll_status, poll_time_us));
259    	
260    	        U32 mabiao_addr = MABIAO_SET_MS(MABIAO_DB_LANDMARK, 50);
261    	        while ((((*address_base) >> reg_data) & 0x01) != poll_status) {
262    	            if (MABIAO_CHK(mabiao_addr)) {
263    	                Dump_LM(tprintf("Timeout"));
264    	                fail_cnt++;
265    	                break;
266    	            }
267    	        }
268    	        Dump_LM(tprintf("Done"));
269    	    }
270    	
271    	    Dump_LM(tprintf("fail_cnt is %d", fail_cnt));
272    	}
