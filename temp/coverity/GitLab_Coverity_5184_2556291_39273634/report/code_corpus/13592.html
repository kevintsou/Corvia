79    	U32 crc_calc(U32 data_base, U32 data_size, U32 crc, U32 polynomial, U32 xor, U32 bit_width, U32 reverse)
80    	{
81    	    U32 data_offset;
82    	    U32 data;
83    	    U32 bit_offset;
84    	
85    	    U32 temp;
86    	
87    	    data = 0;
88    	    for (data_offset = 0; data_offset < data_size; (data_offset += 1)) {
89    	        data = ((U8 *)data_base)[data_offset];
90    	
91    	        //Calculate the CRC
92    	        for (bit_offset = 0; bit_offset < 8; bit_offset++) {
93    	            if (reverse & CRC_CALC_REVERSE_INPUT) {
94    	                temp = (crc >> (bit_width - 1)) ^ (data >> bit_offset);
95    	            }
96    	            else {
97    	                temp = (crc >> (bit_width - 1)) ^ (data >> (8 - 1 - bit_offset));
98    	            }
99    	            crc = (crc << 1) ^ ((temp & BIT0) * polynomial);
100    	        }
101    	    }
102    	
103    	    if (reverse & CRC_CALC_REVERSE_OUTPUT) {
104    	        {
105    	            U32 output;
106    	
107    	            output = 0;
108    	            for (bit_offset = 0; bit_offset < (bit_width / 2); bit_offset++) {
109    	                output |=
110    	                    (((crc >> bit_offset) & 1) << (bit_width - 1 - bit_offset)) |
111    	                    (((crc >> (bit_width - 1 - bit_offset)) & 1) << bit_offset);
112    	            }
113    	            crc = output;
114    	        }
115    	    }
116    	
117    	    //Return the calculated CRC value, remove the extra bits
118    	    return ((crc ^ xor) & (0xFFFFFFFF >> (32 - bit_width)));
119    	}
