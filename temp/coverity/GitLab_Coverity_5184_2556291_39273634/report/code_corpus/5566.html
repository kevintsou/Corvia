66    	/*!
67    	   @brief      Caculate CRC
68    	
69    	   @param      ubDataByte        data byte
70    	   @param      ubRemainder       remainder
71    	   @param      ubFinal           final
72    	   @return     U8
73    	 */
74    	static U8 SMBSCalculateCRC(U8 ubDataByte, U8 ubRemainder, U8 ubFinal)
75    	{
76    	#define SMB_CRC8_POLYNOMIAL         (0x07)      //  x^8 + x^2 + x + 1, 1_0000_0111
77    	#define SMB_CRC8_BIT_COUNT          (0x08)
78    	
79    	    if (ubFinal)
80    	    {
81    	        ubDataByte = 0;
82    	    }
83    	
84    	    for (U8 ubIdx = 0; ubIdx < SMB_CRC8_BIT_COUNT; ubIdx++)
(1) Event misra_c_2012_rule_10_4_violation: Essential type of the left hand operand "ubIdx" (unsigned) is not the same as that of the right operand "8"(signed).
85    	    {
86    	        if (ubRemainder & BIT(SMB_CRC8_BIT_COUNT - 1)) { //it is equal to Shift left the augmented message until a 1 comes out
87    	            //XOR the contents of the register with the low eight bits of the polynomial
88    	            ubRemainder = ((ubRemainder << 1) + (ubDataByte >> (SMB_CRC8_BIT_COUNT - 1))) ^ SMB_CRC8_POLYNOMIAL;
89    	        }
90    	        else {
91    	            ubRemainder = ((ubRemainder << 1) + (ubDataByte >> (SMB_CRC8_BIT_COUNT - 1))); //no 1 comes out, only shift
92    	        }
93    	        ubDataByte = ubDataByte << 1;
94    	    }
95    	
96    	    return ubRemainder;
97    	}
