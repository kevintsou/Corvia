27    	void fip_inst_gen_read_cmd(FIP_CMD_INFO_PTR cmd_info_ptr, FIP_READ_PARAM_STRUCT_PTR fip_read_param_struct_ptr, U8 *retry_tbl)
28    	{
29    	    U8  *inst_buf = (U8 *)DBUF_API_BOOTCODE_TEMP_INST_BUFFER_BASE;
30    	    U8  *buf = (U8 *)DBUF_API_BOOTCODE_TEMP_INST_BUFFER_BASE;
31    	    U32 sq_cmd_size = 0;
32    	    U16 mr_target_id;
33    	    U16 mr_source_id;
34    	    U8  fta_addr[FLH_FTA_ADDR_MAX];
35    	    U8  die = 0;
36    	    U8  plane = 0;
37    	    mr_target_id = MR_COP0_0_0_MVP(4U * fip_read_param_struct_ptr->nand_addr.ch);
38    	    mr_source_id = BE_MR_TID_CPU0_CQ_FIP_CH0;
39    	
40    	    memset((void *)fta_addr, 0, sizeof(fta_addr));
41    	    //memset((void *)inst_buf, 0x00, DBUF_API_BOOTCODE_TEMP_INST_BUFFER_SIZE);
42    	    fip_set_fta_addr(fta_addr, fip_read_param_struct_ptr, FLH_FTA_ROW_ADDR_MAX);
43    	
44    	    buf += sizeof(MR_HDR_STRUCT);
45    	
46    	    buf = (U8 *)fip_inst_api_insert_hd(buf, 0, 0, 0, 1);
47    	
48    	    buf = (U8 *)fip_inst_gen_ce(buf, ((U8)1U << fip_read_param_struct_ptr->nand_addr.bank));
49    	
50    	    buf = (U8 *)fip_inst_gen_dly(buf, gFipInstStruct.ac_timing[FLH_DLY_BUS_SWITCH]);
51    	
52    	    buf = (U8 *)fip_inst_gen_dly(buf, gFipInstStruct.ac_timing[FLH_DLY_TCS]);
53    	
54    	    buf = (U8 *)fip_inst_pasring(cmd_info_ptr, buf, fta_addr, die, plane, retry_tbl);
55    	
56    	    buf = (U8 *)fip_inst_gen_dly(buf, gFipInstStruct.ac_timing[FLH_DLY_TCH]);
57    	
58    	    buf = (U8 *)fip_inst_gen_ce(buf, 0U);
59    	
60    	    buf = (U8 *)fip_inst_gen_end(buf);
61    	
62    	    sq_cmd_size = buf - inst_buf;
63    	
64    	    U32 sq_cmd_size_align = fip_align_nbit(sq_cmd_size, DOORBELL_ALIGN_NUM);
65    	
66    	    for (U32 i = 0; i < (sq_cmd_size_align - sq_cmd_size) ; i++)
67    	    {
68    	        inst_buf[i + sq_cmd_size]  = 0;
69    	    }
70    	
71    	    hal_flh_scq_mr_header(inst_buf, mr_target_id, mr_source_id, sq_cmd_size_align);
72    	
73    	    hal_db_push_cmd_ext((void *)inst_buf, HOST_MR_QID_CPU0_SQ_DMAC_NORMAL);
74    	
75    	}

77    	U32 fip_inst_api_insert_hd(U8 *inst, U8 ce, U8 lun, U8 plane, U8 dma_rel)
78    	{
79    	    U8 is_sca = gFplApiStruct.p_efuse_nand_setting->sca_mode_en;
80    	    HAL_FIP_OP_UNION_PTR p = (HAL_FIP_OP_UNION_PTR)(void *)inst;
81    	
82    	    p->u32_val[0U]         = HAL_FIP_INST_OP_HD;
83    	
84    	    p->hd.ce                = ce;
85    	    p->hd.lun               = lun;
86    	    p->hd.pln               = (is_sca == 1) ? plane : 0U;
87    	    p->hd.dma_rel           = dma_rel;
88    	    p->hd.ignore_phi_rvalid = 0U;
89    	    p->hd.sync_bypass       = (is_sca == 1) ? 1U : 0U;
90    	
91    	    return (U32)&inst[sizeof(HAL_FIP_OP_HD_STRUCT)];
92    	}

126    	U32 fip_inst_gen_cmd(U8 *inst, U8 cmd_addr, U8 last, U8 dqs_h)
127    	{
128    	    //tprintf("fip_inst_gen_cmd cmd = %x", cmd_addr);
129    	    HAL_FIP_OP_UNION_PTR p;
130    	    U32 dqs_oe;
131    	
132    	    p  = (HAL_FIP_OP_UNION_PTR)(void *)inst;
133    	    dqs_oe  = (dqs_h == 0) ? (HAL_FIP_OP_CMD_DQS_LOW) : (HAL_FIP_OP_CMD_DQS_HIGH);
134    	
135    	    p->u32_val[0U] = (HAL_FIP_OP_CMD | ((last == TRUE) ? (HAL_FIP_OP_LAST_CLE_ALE) : (0U)) | dqs_oe);
136    	    p->cmd.cmd    = cmd_addr;
137    	
138    	    return (U32)&inst[sizeof(HAL_FIP_OP_CMD_STRUCT)];
139    	}

141    	U32 fip_inst_gen_prew(U8 *inst, U32 tag, U32 scr_seed, U64 buf_addr, U64 spr_addr, U64 crc_seed)
142    	{
143    	    HAL_FIP_OP_UNION_PTR p = (HAL_FIP_OP_UNION_PTR)(void *)inst;
144    	
145    	    p->u32_val[0U]          = HAL_FIP_OP_PREW;
146    	    p->prew.scr_en          = FIP_DMAR_SCR_EN;
147    	    p->prew.spr_addr_vld    = 1U;
148    	
149    	    p->prew.scr_seed        = scr_seed;
150    	    p->prew.crc_seed_l      = (U32)crc_seed;
151    	    p->prew.buf_addr_l      = (U32)buf_addr;
152    	    p->prew.spr_addr_l      = (U32)spr_addr;
153    	    p->prew.crc_seed_h      = (U8)(crc_seed >> 32U);
154    	    p->prew.buf_addr_h      = (U8)(buf_addr >> 32U);
155    	    p->prew.spr_addr_h      = (U8)(spr_addr >> 32U);
156    	
157    	    p->prew.tag             = tag;
158    	    p->prew.ins_mode        = FIP_DMAR_INST_MODE;
159    	    p->prew.frm_mode        = FIP_DMAR_FRM_MODE;
160    	
161    	    return (U32)&inst[sizeof(HAL_FIP_OP_PREW_STRUCT) - 4U];
162    	}

164    	U32 fip_inst_gen_datr(U8 *inst, U32 tag_ofst)
165    	{
166    	    HAL_FIP_OP_UNION_PTR p = (HAL_FIP_OP_UNION_PTR)(void *)inst;
167    	
168    	    p->u32_val[0U] = (HAL_FIP_OP_DATR | HAL_FIP_OP_FIRST_DAT | HAL_FIP_OP_LAST_DAT);

171    	
172    	    return (U32)&inst[sizeof(HAL_FIP_OP_DATR_STRUCT)];
173    	}

175    	U32 fip_inst_pasring(FIP_CMD_INFO_PTR cmd_info_ptr, U8 *buf, U8 *addr, U8 die, U8 plane, U8 *retry_tbl)
176    	{
177    	    U64 dly;
178    	    U8 status;
179    	    U8 col_addr_num;
180    	    U8 row_addr_num;
181    	    U8 retry_addr_num;
182    	    U8 retry_addr[EFUSE_PARAM_READ_RTY_DAT_SIZE] = {0U};
183    	
184    	    status                = TRUE;
185    	    dly                   = 0U;
186    	    cmd_info_ptr->op_id   = 0U;
187    	    cmd_info_ptr->data_id = 0U;
188    	    col_addr_num          = 0U;
189    	    row_addr_num          = 0U;
190    	    retry_addr_num        = 0U;
191    	
192    	    while (status)
193    	    {
194    	        switch (cmd_info_ptr->op[cmd_info_ptr->op_id])
195    	        {
196    	            case INST_CMD:    //x0x01
197    	                buf = (U8 *)fip_inst_gen_cmd(buf, cmd_info_ptr->data[cmd_info_ptr->data_id], 1U, gFplApiStruct.p_efuse_nand_setting->cmd_dqs_l);
198    	                cmd_info_ptr->op_id ++;
199    	                cmd_info_ptr->data_id ++;
200    	                break;
201    	
202    	            case INST_ADR_FTA_C:    //0x06
203    	                if ((cmd_info_ptr->op[cmd_info_ptr->op_id + 1U]  & ADR_FTA_COL_ADR_MASK) != FALSE)
204    	                {
205    	                    col_addr_num = ADR_FTA_TWO_CYCLE_COL_ADR;
206    	                }
207    	                else
208    	                {
209    	                    col_addr_num = ADR_FTA_ZERO_CYCLE_COL_ADR;
210    	                }
211    	                row_addr_num = (cmd_info_ptr->op[cmd_info_ptr->op_id + 1U] & ADR_FTA_ROW_ADR_MASK);
212    	                if ((col_addr_num == 0U) && (row_addr_num != 0U))
213    	                {
214    	                    buf = (U8 *)fip_inst_gen_addr(buf, &addr[ADR_FTA_ROW_ADR_OFST], row_addr_num);
215    	                }
216    	                else
217    	                {
218    	                    buf = (U8 *)fip_inst_gen_addr(buf, addr, (col_addr_num + row_addr_num));
219    	                }
220    	                cmd_info_ptr->op_id += 2U;
221    	                break;
222    	
223    	            case INST_ADR:      //0x02
224    	            case INST_ADR_RTY:  //0x0A
225    	                buf = (U8 *)fip_inst_gen_addr(buf, &(cmd_info_ptr->data[cmd_info_ptr->data_id]), 1U);
226    	                cmd_info_ptr->op_id ++;
227    	                cmd_info_ptr->data_id ++;
228    	                break;
229    	
230    	            case INST_PLANE_ADR_RTY:    //0x0C
231    	                buf = (U8 *)fip_inst_gen_addr(buf, &gFplApiStruct.plane_adr_tbl[plane], 1U);
232    	                cmd_info_ptr->op_id ++;
233    	                break;
234    	
235    	            case INST_DIE_ADR:      //0x0F
236    	                buf = (U8 *)fip_inst_gen_addr(buf, &die, 1U);
237    	                cmd_info_ptr->op_id ++;
238    	                break;
239    	
240    	            case INST_ACRR_ADR:     //0x0B
241    	                memset((void *)retry_addr, 0U, sizeof(retry_addr));
242    	                retry_addr_num = 1U + ((cmd_info_ptr->op[cmd_info_ptr->op_id + 1U]) & ACRR_ADR_RTY_CYCLE_MASK);
243    	                if (retry_tbl == NULL)   // NOR READ SEQ
244    	                {
245    	                    if (gFplApiStruct.acrr_enable == TRUE)
246    	                    {
247    	                        buf = (U8 *)fip_inst_gen_addr(buf, retry_addr, retry_addr_num);    // micron 1 addr = 0x00 , hynix V6 4 addr = 0x00
248    	                    }
249    	                }
250    	                else    // RETRY SEQ
251    	                {
252    	                    if (gFplApiStruct.adr_rty_en == FALSE)
253    	                    {
254    	                        memcpy((void*)retry_addr, (void*)retry_tbl, EFUSE_PARAM_READ_RTY_DAT_SIZE);
255    	                    }
256    	                    buf = (U8 *)fip_inst_gen_addr(buf, retry_addr, retry_addr_num);
257    	                }
258    	                cmd_info_ptr->op_id += 2U;
259    	                break;
260    	
261    	            case INST_ADR_RTY_C:        //0x09
262    	                gFplApiStruct.adr_rty_en = TRUE;
263    	                memset((void *)retry_addr, 0U, sizeof(retry_addr));
264    	                retry_addr_num = 1U + ((cmd_info_ptr->op[cmd_info_ptr->op_id + 1U]) & ACRR_ADR_RTY_CYCLE_MASK);
265    	                if (retry_tbl != NULL)
266    	                {
267    	                    memcpy((void*)retry_addr, (void*)retry_tbl, EFUSE_PARAM_READ_RTY_DAT_SIZE);
268    	                }
269    	                buf = (U8 *)fip_inst_gen_addr(buf, retry_addr, retry_addr_num);
270    	                cmd_info_ptr->op_id += 2U;
271    	                break;
272    	
273    	            case INST_CMD_00h:      //0x10
274    	            case INST_CMD_30h:      //0x11
275    	            case INST_CMD_05h:      //0x12
276    	            case INST_CMD_06h:      //0x13
277    	            case INST_CMD_E0h:      //0x14
278    	            case INST_CMD_A2h:      //0x15
279    	            case INST_CMD_3Bh:      //0x16
280    	            case INST_CMD_2Eh:      //0x17
281    	            case INST_CMD_DAh:      //0x18
282    	            case INST_CMD_5Dh:      //0x19
283    	            case INST_CMD_36h:      //0x1C
284    	            case INST_CMD_37h:      //0x1D
285    	            case INST_CMD_16h:      //0x1E
286    	                buf = (U8 *)fip_inst_gen_cmd(buf, gFipInstStruct.cmd_remap_table[(cmd_info_ptr->op[cmd_info_ptr->op_id] - INST_CMD_00h)], 1U, gFplApiStruct.p_efuse_nand_setting->cmd_dqs_l);
287    	                cmd_info_ptr->op_id ++;
288    	                break;
289    	
290    	            case INST_END:      //0x00
291    	                status = FALSE;
292    	                break;
293    	
294    	            default:
295    	                //ERROR
296    	                Dump_Debug(tprintf("Error, Undefined cmd info : %x", cmd_info_ptr->op[cmd_info_ptr->op_id]));
297    	                break;
298    	        }
299    	    }
300    	    return (U32)&buf[0U];
301    	}

303    	U32 fip_inst_gen_datw(U8 *inst, U8 *data, U32 len)
304    	{
305    	    Dump_Debug(tprintf("fip_inst_gen_datw exec"));
306    	    HAL_FIP_OP_UNION_PTR p = (HAL_FIP_OP_UNION_PTR)(void *)inst;
307    	    U32 i = 0U;
308    	
309    	    while (i < len)
310    	    {
311    	        if (i == 0U)
312    	        {
313    	            p->u32_val[i]  = (HAL_FIP_OP_DATW | HAL_FIP_OP_FIRST_DAT | (((U32)data[i]) << 16U));
314    	        }
315    	        else
316    	        {
317    	            p->u32_val[i] = (HAL_FIP_OP_DATW | (((U32)data[i]) << 16U));
318    	        }
319    	
320    	        i ++;
321    	    }
322    	
323    	    if (i > 0U)
324    	    {
325    	        p->u32_val[i - 1U] |= HAL_FIP_OP_LAST_DAT;
326    	    }
327
