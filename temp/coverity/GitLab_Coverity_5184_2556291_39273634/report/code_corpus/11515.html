748    	void fpl_set_addr_rule(PACK_ADDR_RULE_STRUCT_PTR rule, PACK_ADDR_RULE_STRUCT_PTR lsb_rule, U32 value)
749    	{
750    	    // shift
751    	    if (lsb_rule) {
752    	        rule->shift = lsb_rule->shift + lsb_rule->bit_no;
753    	    }
754    	    else {
755    	        rule->shift = 0;
756    	    }
757    	
758    	    // bit_no
759    	    rule->bit_no = my_log2(value);
760    	    if (((U32)1 << rule->bit_no) < value) {
761    	        rule->bit_no++;
762    	    }
763    	
764    	    // mask
765    	    rule->mask = (1 << rule->bit_no) - 1;
766    	}
