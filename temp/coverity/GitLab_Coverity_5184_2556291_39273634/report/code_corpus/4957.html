63    	{
64    	    if (0U == p_efuse_api->efuse_nand_setting_read) {
65    	        p_efuse_api->efuse_nand_setting_offset = EFUSE_NAND_SETTING_COUNT;
66    	        //Read out 1st NAND Setting
67    	        if (0U == efuse_api_read_bank(

71    	                EF_NAND_SIZE,
72    	                &(p_efuse_api->a_efuse_bank_read_state[EF_NAND_BANK]))) {
