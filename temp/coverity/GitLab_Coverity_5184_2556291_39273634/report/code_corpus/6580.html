818    	                fip_inst_gen_read_cmd(&(gFipInstStruct.inst[INST_ID_PREFIX_FOR_RETRY]), fip_read_param_struct_ptr, gFplApiStruct.read_rty_tbl[(read_op - 1U)]);
819    	            }
820    	            else {
821    	                do {
822    	                    status |= fip_set_feature_retry(&(gFipInstStruct.inst[INST_ID_SET_FEAT_FOR_RETRY]), gFplApiStruct.read_rty_tbl[(read_op - 1U)], ch, bank, plane, die, set_feat_dat);

826

829    	                                break;
830    	                            }
831    	                        }
832    	                    }
833    	                    cnt++;
834    	                } while ((cnt < FLH_SET_FEAT_CNT_MAX) && (status == 0U));
835    	
836    	                if (cnt == FLH_SET_FEAT_CNT_MAX) {
837    	                    status |= HAL_STATUS_SET_FEAT_DAT_CMP_FAIL;
838    	                }
839    	
840    	                if (status == 0U) {
841    	                    fip_inst_gen_read_cmd(&(gFipInstStruct.inst[INST_ID_READ_CMD]), fip_read_param_struct_ptr, NULL);
842    	                }
843    	            }
844    	        }
845    	        else {
846    	            //Normal Read
847    	            if (gFplApiStruct.retry_by_prefix == TRUE) // TBD, need efuse ??
(1) Event misra_c_2012_rule_10_4_violation: Essential type of the left hand operand "gFplApiStruct.retry_by_prefix" (unsigned) is not the same as that of the right operand "1"(boolean).
848    	            {
849    	                U8 default_read_level[EFUSE_PARAM_READ_RTY_DAT_SIZE] = {0, 0, 0, 0};
850    	                fip_inst_gen_read_cmd(&(gFipInstStruct.inst[INST_ID_PREFIX_FOR_RETRY]), fip_read_param_struct_ptr, default_read_level);
851    	            }
852    	            else
853    	            {
854    	                fip_inst_gen_read_cmd(&(gFipInstStruct.inst[INST_ID_READ_CMD]), fip_read_param_struct_ptr, NULL);
855    	            }
856    	        }
857    	    }
858    	
859    	    return status;
860    	}
