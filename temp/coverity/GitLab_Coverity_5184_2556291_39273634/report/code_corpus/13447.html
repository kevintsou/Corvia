104    	void fpl_ecc_param_init(FPL_API_STRUCT_PTR fpl_ptr)
105    	{
106    	    U8  spare_size;
107    	    U8  ldpc_2_size;
108    	    U8  ubCH;
109    	    U16 ldpc_size;
110    	    U32 frame_size;
111    	
112    	    U8  spare_mode_size[]  = {0U, 0U, 0U, 16U, 20U, 24U, 28U, 32U, 36U, 40U, 44U, 48U};
113    	
114    	    U16 ldpc_mode_size[]   = {1120U, 1088U, 1056U, 1024U, 992U, 960U, 928U, 896U, 864U, 832U,
115    	                            800U, 768U, 736U, 704U, 672U, 640U, 608U, 576U, 544U, 512U,
116    	                            480U, 448U, 416U, 384U, 352U, 320U};
117    	
118    	    U8  ldpc_2_mode_size[] = {0U, 4U, 8U, 12U, 16U, 20U};
119    	
120    	    ldpc_size    = ldpc_mode_size[fpl_ptr->p_efuse_nand_setting->ldpc_mode];
121    	    ldpc_2_size  = ldpc_2_mode_size[fpl_ptr->p_efuse_nand_setting->puncture_mode];
122    	    spare_size   = spare_mode_size[HAL_FIP_SPARE_MODE];
123    	
124    	    frame_size  = LC_BYTE_SIZE + spare_size + ldpc_size - ldpc_2_size + fpl_ptr->p_efuse_nand_setting->dummy_data_length;
125    	
126    	    Dump_Debug(tprintf("frame_size = %d", frame_size));
127    	
128    	    fpl_ptr->byte_per_node = frame_size;
129    	
130    	
131    	    for (ubCH = 0U; ubCH < DAL_FIP_CH_NUM; ubCH++)
132    	    {
133    	        fip_set_data_size_mode(ubCH,(U8)fpl_ptr->p_efuse_nand_setting->ldpc_mode, (U8)fpl_ptr->p_efuse_nand_setting->puncture_mode);
134    	        fip_set_frm_size(ubCH, frame_size);
135    	    }
136    	}

163    	void fpl_redriver_hw_reset(FPL_API_STRUCT_PTR fpl_ptr)
164    	{
165    	    U8 ubCE = 0;
166    	    for(U8 ubCh = 0; ubCh < FLH_MAX_CH_NUM; ubCh++)
167    	    {
168    	        r32_FLH_REG_TX(ubCh)[r32_FCH_FCE_ENB] |= (U32)(BIT(ubCE));
169    	        w_DQ_BIT_REV_IDX(ubCh, ubCE);
170    	        dal_fip_delay(fpl_ptr->ac_timing[HAL_DLY_TCS]);
171    	
172    	        w_PIO_CER_LAST(ubCh, 0U);
173    	        for(U8 ubi = 0; ubi < 20; ubi++)
174    	        {
175    	            if(19 == ubi)
176    	            {
177    	                w_PIO_CER_LAST(ubCh, 1U);
178    	            }
179    	            w_PIO_CER(ubCh, 0x1F);
180    	        }
181    	        dal_fip_delay(fpl_ptr->ac_timing[HAL_DLY_TCH]);
182    	        r32_FLH_REG_TX(ubCh)[r32_FCH_FCE_ENB] &= (U32)(~(BIT(ubCE)));
183    	    }

187    	void fpl_redriver_initial(FPL_API_STRUCT_PTR fpl_ptr)
188    	{
189    	    if(0 != fpl_ptr->p_efuse_nand_setting->redriver_startup_time)
190    	    {
191    	        Dump_Debug(tprintf("redriver startup delay time"));
192    	        U16 uwDelayTimeUS[4] = {0, 250, 500, 1000};
193    	        idle_us(uwDelayTimeUS[fpl_ptr->p_efuse_nand_setting->redriver_startup_time]);
194    	    }
195    	
196    	    if (gpFwSimCtrl1Reg->redriver_reset_cmd && gpFwSimCtrl1Reg->jump_to_fw)
197    	    {
198    	        Dump_Debug(tprintf("redriver hw reset"));
199    	        fpl_redriver_hw_reset(fpl_ptr);
200    	    }
201    	
202    	    if(0 != fpl_ptr->p_efuse_nand_setting->redriver_reg_wr_option)
203    	    {
204    	        Dump_Debug(tprintf("redriver write reg"));
205    	        U8 ubCE = 0;
206    	        U8 ubData[4] = {0, 0x1, 0x2, 0x3};
207    	        for(U8 ubCh = 0; ubCh < FLH_MAX_CH_NUM; ubCh++)
208    	        {
209    	            r32_FLH_REG_TX(ubCh)[r32_FCH_FCE_ENB] |= (U32)(BIT(ubCE));
210    	            w_DQ_BIT_REV_IDX(ubCh, ubCE);
211    	            dal_fip_delay(fpl_ptr->ac_timing[HAL_DLY_TCS]);
212    	
213    	            w_PIO_CER_LAST(ubCh, 1U);
214    	            w_PIO_CER(ubCh, 0x03);
215    	            w_PIO_ADR(ubCh, 0xF1);
216    	            M_FIP_CH_SET_PIO_DAT(ubCh, ubData[fpl_ptr->p_efuse_nand_setting->redriver_reg_wr_option]);
217    	
218    	            dal_fip_delay(fpl_ptr->ac_timing[HAL_DLY_TCH]);
219    	            r32_FLH_REG_TX(ubCh)[r32_FCH_FCE_ENB] &= (U32)(~(BIT(ubCE)));
220    	        }
221    	    }
222    	
223    	    if(0 != fpl_ptr->p_efuse_nand_setting->redriver_reg_rd_option)
224    	    {
225    	        Dump_Debug(tprintf("redriver read reg"));
226    	        U8 ubCE = 0, ubReadData = 0xFF, ubReadCnt = 0;
227    	        U8 ubData[4] = {0, 0x1, 0x2, 0x3};
228    	        for(U8 ubCh = 0; ubCh < FLH_MAX_CH_NUM; ubCh++)
229    	        {
230    	            r32_FLH_REG_TX(ubCh)[r32_FCH_FCE_ENB] |= (U32)(BIT(ubCE));
231    	            w_DQ_BIT_REV_IDX(ubCh, ubCE);
232    	            dal_fip_delay(fpl_ptr->ac_timing[HAL_DLY_TCS]);
233    	
234    	            while(ubData[fpl_ptr->p_efuse_nand_setting->redriver_reg_rd_option] != ubReadData)
235    	            {
236    	                if (100 == ubReadCnt)
237    	                {
238    	                    break;
239    	                }
240    	                w_PIO_CER_LAST(ubCh, 1U);
241    	                w_PIO_CER(ubCh, 0x03);
242    	                w_PIO_ADR(ubCh, 0xF1);
243    	                ubReadData = ((U8)r32_r_FCH_PIO_DAT(ubCh));
244    	                ubReadCnt++;
245    	            }
246    	
247    	            dal_fip_delay(fpl_ptr->ac_timing[HAL_DLY_TCH]);
248    	            r32_FLH_REG_TX(ubCh)[r32_FCH_FCE_ENB] &= (U32)(~(BIT(ubCE)));
249    	        }
250    	    }
251    	}

253    	void fpl_set_dq_revert(FPL_API_STRUCT_PTR fpl_ptr)
254    	{
255    	    U8 ch;
256    	    U8 bank;
257    	
258    	    bank = 0U;
259    	    for (ch = 0U; ch < FLH_MAX_PHYS_CH; ch++) {
260    	        if (BITS_GET(fpl_ptr->dq_revert_bitmap, ch, 1U) == TRUE) {
261    	            hal_fip_set_dq_reverse(ch, bank, (BOOL)TRUE);
262    	        }
263    	        FIP_DBG_INFO_PRINTF("[dq_revert]DQ_REV_BMP[%d] = 0x%02X\r\n", \
264    	            ch, \
265    	            fpl_ptr->dq_revert_bitmap);
266    	    }

269    	void fpl_fip_clock_init(FPL_API_STRUCT_PTR fpl_ptr)
270    	{
271    	    /* X3 FIP functional divider clock setting table (FLH_OSC_CLK = 800 MHz)
272    	       Toggle :
273    	       * - 000b: div 2
274    	       * - 001b: div 4
275    	       * - 010b: div 8
276    	       * - 011b: div 16
277    	       * - 100b: div 32
278    	       * - 101b: div 64
279    	       * - 110b: div 128
280    	       * - 111b: div 128
281    	    */
282    	
283    	    //clear clock divider
284    	    hal_fip_set_ca_clk_div_en(FALSE, 0U);
285    	    hal_fip_set_clk_div_en(FALSE, 0U);
286    	
287    	    //set clock divider
288    	    if (fpl_ptr->p_efuse_nand_setting->sca_mode_en == EFUSE_PARAM_PROTOCOL_CONVENTIONAL) {
289    	        //CONV MODE
290    	        if (fpl_ptr->flash_if == HAL_FIP_OP_FLH_IF_LEGACY) {
291    	            if (fpl_ptr->p_efuse_nand_setting->fip_dq_clock_divider == FIP_TEST_LEGACY_50MT) {
292    	                hal_fip_set_clk_div_en(TRUE, 0U);
293    	            }
294    	            else if (fpl_ptr->p_efuse_nand_setting->fip_dq_clock_divider == FIP_TEST_LEGACY_25MT) {
295    	                hal_fip_set_clk_div_en(TRUE, 1U);
296    	            }
297    	            else {
298    	                hal_fip_set_clk_div_en(TRUE, 2U); // FIP_TEST_LEGACY_12_5MT
299    	            }
300    	        }
301    	        else {
302    	            if (fpl_ptr->p_efuse_nand_setting->fip_dq_clock_divider == FIP_TEST_TOGGLE_400MT) {
303    	                hal_fip_set_clk_div_en(TRUE, 0U);
304    	            }
305    	            else if (fpl_ptr->p_efuse_nand_setting->fip_dq_clock_divider == FIP_TEST_TOGGLE_200MT) {
306    	                hal_fip_set_clk_div_en(TRUE, 1U);
307    	            }
308    	            else if (fpl_ptr->p_efuse_nand_setting->fip_dq_clock_divider == FIP_TEST_TOGGLE_50MT) {
309    	                hal_fip_set_clk_div_en(TRUE, 3U);
310    	            }
311    	            else {
312    	                hal_fip_set_clk_div_en(TRUE, 2U); // FIP_TEST_TOGGLE_100MT
313    	            }
314    	        }
315    	    }
316    	    else {
317    	        //SCA MODE
318    	        //CA BUS
319    	        // HW recommand not to use in X3
320    	//        hal_fip_set_ca_clk_div_en(FALSE, 0U);
321    	
322    	        //DQ BUS
323    	        if (fpl_ptr->p_efuse_nand_setting->fip_dq_clock_divider == FIP_TEST_TOGGLE_400MT) {
324    	            hal_fip_set_clk_div_en(TRUE, 0U);
325    	        }
326    	        else if (fpl_ptr->p_efuse_nand_setting->fip_dq_clock_divider == FIP_TEST_TOGGLE_200MT) {
327    	            hal_fip_set_clk_div_en(TRUE, 1U);
