419    	    dal_fip_delay(ac_timing[HAL_DLY_TCH]);
420    	    //dal_fip_pio_ce_disable(ch, ce);
421    	    r32_FLH_REG_TX(ch)[r32_FCH_FCE_ENB] &= (U32)(~(BIT(ce)));
422    	
423    	    FIP_DBG_INFO_PRINTF("[read_status_by_lun]ch = %d, ce = %d, read status - %x\r\n", ch, ce, *data_buf);
424    	
425    	    return status;
