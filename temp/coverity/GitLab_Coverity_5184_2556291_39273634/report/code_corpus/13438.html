404    	  @brief reverse an array, big endian<-> little endian
405    	  @param[in] by_s big number you want to reverse.
406    	  @param[in] dwlen means the length of by_s.
407    	  @return NONE
408    	  @note
409    	    - NONE
410    	*/
411    	void hal_sec_pqc_kaz_big_num_invert(U8 *by_s, U32 dwlen)
412    	{
413    	    U32 dw_ix, dw_iy;
414    	    U8 by_t;
415    	
416    	    dw_ix = 0UL;
417    	    dw_iy = dwlen - 1UL;
418    	    while (dw_ix < dw_iy)
419    	    {
420    	        by_t     = by_s[dw_ix];
421    	        by_s[dw_ix] = by_s[dw_iy];
422    	        by_s[dw_iy] = by_t;
423    	        ++dw_ix;
424    	        --dw_iy;
425    	    }

628

630

632    	    hal_sec_pke_lib_zero(&pke_cmd, 15U, tmp_c1_addr.sel, tmp_c1_addr.adr, (sizen >> 1U) + 1U);
633    	
634    	    // 18 TMPA1 = 1
635    	    hal_sec_pke_lib_sltb(&pke_cmd, 15U, ((U16)sizen >> 1U) + 1U, tmp_c1_addr.sel, tmp_c1_addr.adr, tmp_a1_addr.sel, tmp_a1_addr.adr, 15U, 15U);
636    	
637    	    // ---- ---- Out = mmul(Out,TMPA1)mod N ---- ---- //
638    	
639    	
640    	    // 19 TMPC1 = mmul(TMPA1,Out)
641    	    hal_sec_pke_lib_mmul64(&pke_cmd, tmp_c1_addr.sel, tmp_c1_addr.adr, tmp_a1_addr.sel, tmp_a1_addr.adr, out_addr.sel, out_addr.adr, r_addr.adr, n_addr.adr, (sizen64 + 1U), 0U, 0U);
642    	
643    	    // 20 Out = TMPC1
644    	    hal_sec_pke_lib_shf(&pke_cmd, 15U, ((U16)sizen >> 1U) + 1U, tmp_c1_addr.sel, tmp_c1_addr.adr, out_addr.sel, out_addr.adr, HAL_SEC_PKE_SHIFT_NONE, 0U);
645    	
646    	    // ---- ---- Other ---- ---- //
647    	    hal_sec_pke_init_reg_tb_val(HAL_SEC_BLOCKING_TIME_VAL);
648    	
649    	    hal_sec_pke_set_en();
650    	
651    	    return SEC_YES;
652    	}

803    	U64 hal_sec_kaz_mod_exp_of_two(U64 a, U64 b, U32 n_exp)
804    	{
805    	    U64 result = 1ULL;
806    	    U64 n = 1ULL << (U64)n_exp;
807    	    a = a & (n - 1ULL);  // a = a % n
808    	
809    	    while (b > 0ULL)
810    	    {
811    	        if (b % 2ULL == 1ULL)
812    	        {
813    	            result = (result * a) & (n - 1ULL);
814    	        }
815    	        b = b >> 1ULL;
816    	        a = (a * a) & (n - 1ULL);  // a = a^2 % n
817    	    }
818    	    return result;
819    	}
