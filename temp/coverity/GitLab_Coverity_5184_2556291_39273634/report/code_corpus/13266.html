669    	        (void)memcpy((void *)tmp, (void *)xpos, last);
670    	        (void)memset((void *)(tmp + last), 0, sizeof(tmp) - last);
671    	
672    	        /* Y_i = (Y^(i-1) XOR X_i) dot H */
673    	        hal_xor_block(y, tmp);
674    	
675    	        /* dot operation:
676    	         * multiplication operation for binary Galois (finite) field of
677    	         * 2^128 elements */
678    	        hal_gf_mult(y, h, tmp);
679    	        (void)memcpy((void *)y, (void *)tmp, 16);
680    	    }
681    	
682    	    /* Return Y_m */
683    	}

1109    	    aesofl.comm.setkey.key_ptr     = params->setkey.key_ptr;
1110    	    aesofl.comm.setkey.aes_mode    = HAL_SEC_AES_ECB_MODE;
1111    	    aesofl.comm.setkey.key_type    = params->setkey.key_type;
1112    	    aesofl.comm.setkey.is_otf_type = params->setkey.is_otf_type;
1113    	    aesofl.comm.setkey.is_otf_core = params->setkey.is_otf_core;
1114    	    aesofl.comm.setkey.gen_key_en  = SEC_NO;
1115    	    aesofl.comm.setkey.timeout_val = HAL_SEC_BLOCKING_TIME_VAL;
1116    	
1117    	    //Calculate intermediate values.
1118    	    /*
1119    	        * For j = 0 to 5
1120    	        *     For i=1 to n
1121    	        *         B = AES(K, A | R[i])
1122    	        *         A = MSB(64, B) ^ t where t = (n*j)+i
1123    	        *         R[i] = LSB(64, B)
1124    	        *
1125    	     */
1126    	    for (U8 j = 0U; j <= 5U; j++)
1127    	    {
1128    	        r = ctx + (8U >> 2U);
1129    	
1130    	        for (U32 i = 1U; i <= n_blk; i++)
1131    	        {
1132    	            (void)memcpy((void *)b, (void *)a, 8U);
1133    	            (void)memcpy((void *)(b + 2U), (void *)r, 8U);
1134    	
1135    	            ASM_DSB();
1136    	
1137    	            ret = phal_sec_aes_ofl_blocking(&aesofl);
1138    	
1139    	            if (ret == SEC_NO)
1140    	            {
1141    	                break;
1142    	            }
1143    	
1144    	            for (U32 k = 0U; k < (8U >> 2U); k++)
1145    	            {
1146    	                hal_sec_api_reverse_endian(&b[k], &b[k], 4UL);
1147    	            }
1148    	
1149    	            (void)memcpy((void *)a, (void *)b, 8U);
1150    	
1151    	            tmp = ((U64)((a)[0])) << 32U | (U64)(a[1]);
1152    	            //tmp ^= (n_blk * j + i);
1153    	
1154    	            a[0] = (U32)(tmp >> 32U);
1155    	            //a[1] = (U32)(tmp);
1156    	            a[1] = (U32)(tmp) ^ (n_blk * j + i);
1157    	
1158    	            for (U32 k = 0U; k < (8U >> 2U); k++)
1159    	            {
1160    	                hal_sec_api_reverse_endian(&a[k], &a[k], 4U);
1161    	            }
1162    	
1163    	            (void)memcpy((void *)r, (void *)(b + 2U), 8U);
1164    	
1165    	            r += 2U;
1166    	        }
1167    	
1168    	        if (ret == SEC_NO)
1169    	        {
1170    	            break;
1171    	        }
1172    	    }
1173    	
1174    	    return ret;
1175    	}
